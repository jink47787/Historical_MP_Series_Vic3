	first_name = Otto
	last_name = von_Bismarck
	historical = yes
	ig_leader = yes
	interest_group = ig_landowners
	female = no
	birth_date = 1815.4.1
	dna = dna_otto_von_bismarck
	ideology = ideology_royalist
	traits = {
		persistent
		ambitious
		masterful_diplomat
		experienced_political_operator
		demagogue
	}

    interest_group_leader_usage = {
    	country_trigger = {
    		OR = {
				c:PRU ?= this
				was_formed_from = PRU
			}
			has_law = law_type:law_monarchy
			has_technology_researched = pan-nationalism
			NOT = {
				has_global_variable = bismarck_variable
			}
    	}
    	interest_group_trigger = {
    	    is_interest_group_type = ig_landowners
    	}
    	earliest_usage_date = 1846.1.1
    	latest_usage_date = 1890.3.18
    	chance = 100
    }

	on_created = {
		set_global_variable = bismarck_variable
		owner = {
			trigger_event = { id = german_unification.3 }
		}
	}
}
