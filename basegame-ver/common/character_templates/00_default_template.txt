# Must have one entry of the "default" template, the template itself can be modified but one instance of it MUST exist with all parameters filled out

# As of 1.12, there's a few different ways to specify roles, all of these can be mixed together so long as the roles are compatible
# ruler = yes, heir = yes, executive = yes etc - old method, works as before as a shortcut to assign the archetype and its associated auto-assigned roles
# role = ruler/heir/agitator/politician/executive/general/admiral/other - assigns the archetype and its associated auto-assigned roles
# role = <database key> - assigns role from database + any archetype auto-assigned roles

default = {
	first_name = culture 		# Picks a random first and last name from the culture, combines those together to make one name
	last_name = culture
	historical = no
	religion = random_accepted # Valid inputs: random_accepted/random/specific key. random & random_accepted weight based on share of political strength within the character's culture. If home region is specified, random is based on home state, otherwise home country.
	culture = ig_before_random_accepted # Valid inputs: primary_culture/random_accepted/random/ig_before_primary_culture/ig_before_random_accepted/ig_before_random/specific key. random/random_accepted weight based on share of political strength within the country. If home region is specified, random is based on home state, otherwise home country.
	female = no
	home_region = random		# valid inputs: state region key, state scope, random
	holding_type = random		# valid inputs: building type key, building type scope, random
	dna = random 				# Generate random dna from a random ethnicity in the culture
	age = default 				# A random age base on AGE_GENERATION_STARTVAL + ( random_int % ( AGE_GENERATION_DICE_ROLL_MAGNITUDE + 1 ) )
	interest_group = random 	# Select a random interest group based on clout
	commander_rank = default	# commander rank if this character would be a military commander
	trait_generation = { 		# this effect will run as soon as the character is created (should only be used for traits)
		if = {
			limit = {
				trait_value > 0
			}
			set_variable = has_historical_traits
			add_experience = trait_value
		}
		else = {
			add_experience = 4
		}

		if = {
			limit = {
				is_adult = no
			}
			add_trait = trait_child
		}
		if = {
			limit = {
				is_adult = yes
				trait_value = 0
			}
			add_random_trait = personality
			if = {
				limit = {
					OR = {
						NOT = { has_variable = has_historical_traits }
						trait_value <= 3
					}
				}
				add_random_trait = skill
			}
		}

		if = {
			limit = {
				age >= 45
				trait_value < 8
			}
			random_list = {
				33 = {}
				33 = { add_experience = 1 }
				33 = {
					add_experience = 2
					add_random_trait = skill
				}
			}
		}
		if = {
			limit = {
				age >= 30
				NOT = { has_variable = has_historical_traits }
			}
			random_list = {
				95 = {}
				5 = { add_random_trait = condition }
			}
		}
	}
}
