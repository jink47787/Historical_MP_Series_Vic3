character_has_been_patronized = {
    trigger_if = {
        limit = {
            AND = {
                has_role_of_type = agitator
                OR = {
                    has_role_of_type = general
                    has_role_of_type = admiral
                }
            }
        }
        has_modifier = modifier_agitator_commander_political_patronage
    }
    trigger_else_if = {
        limit = {
            has_role_of_type = agitator
        }
        has_modifier = modifier_agitator_political_patronage
    }
    trigger_else_if = {
        limit = {
            NOT = {
                has_role_of_type = politician
            }
            OR = {
                has_role_of_type = general
                has_role_of_type = admiral
            }
        }
        has_modifier = modifier_commander_political_patronage
    }
    trigger_else = {
        has_modifier = modifier_political_patronage
    }
}

character_has_valid_reason_for_coup = {
    OR = {
        owner = { # Case 1 - owner is enacting a law the character does not approve of
			exists = owner.currently_enacting_law
            OR = {
                PREV = {
                    law_enactment_stance = { # Refers to any_scope_character
                        law = PREV.currently_enacting_law.type # Refers to owner
                        value < neutral # Disapproves of the law
                    }
                }
                AND = {
                    PREV ?= {
                        law_enactment_stance = { # Refers to any_scope_character
                            law = PREV.currently_enacting_law.type # Refers to owner
                            value < approve # Character does not approve of the law
                        }
                        interest_group ?= {
                            law_enactment_stance = { # Refers to any_scope_character's interest group
                                law = PREV.owner.currently_enacting_law.type # Refers to owner
                                value < neutral # Interest group opposes the law
                            }
                        }
                    }
                }
            }
        }
        this.interest_group = {
            OR = {
                has_modifier = modifier_failed_to_meet_petition # Case 2 - owner has failed a petition for this.interest_group
                has_modifier = negotiation_failed_quest # Case 3 - owner has failed a negotiation for this.interest_group
            }
        }
        OR = {
            has_trait = ambitious # Case 4 - this just has a coupiful personality
            has_trait = imperious
            has_trait = special_character_blanqui_is_free
            has_trait = special_character_napoleonic_return
        }
        # Not in this trigger, but exposed elsewhere:
            # Case 5 - Foreign-Backed Coup
            # Case 6 - Foreign-Backed Coup
    }
}

japanese_character_is_marked_for_death_chance = {
    OR = {
        AND = { # The Ninko Emperor dies quite young
            has_template = JAP_ninko_yamato
            game_date >= 1846.1.1
        }
        AND = { # Ienari mustn't live past 1842
            has_template = JAP_ienari_tokugawa
            game_date >= 1841.1.1
        }
        AND = { # Iesada had serious health issues
            has_template = JAP_iesada_tokugawa
            is_ruler_of_own_country = yes
            var:year_in_power >= 4
        }
        AND = { # Iemochi also dies very young
            has_template = JAP_iemochi_tokugawa
            is_ruler_of_own_country = yes
            age >= 20
        }
        AND = { # The Komei Emperor will die at an extremely inconvenient time
            has_template = JAP_komei_yamato
            game_date >= 1865.1.1
            owner ?= {
                any_political_movement = {
                    OR = {
                        is_political_movement_type = movement_meiji_restorationist
                        is_political_movement_type = movement_land_reform
                        movement_is_liberal = yes
                        movement_is_socialist = yes
                    }
                    political_movement_radicalism > define:NPolitics|RADICALISM_PROTESTING_THRESHOLD
                }
            }
        }
    }
}
