# Treaty Shorthands
has_friendly_diplomatic_pact = {
	OR = {
		any_scope_treaty = {
			binds = $TARGET$
			any_scope_article = {
				has_type = trade_privilege
				source_country = $TARGET$
			}
		}
		any_scope_treaty = {
			binds = $TARGET$
			any_scope_article = {
				has_type = alliance
			}
		}
		any_scope_treaty = {
			binds = $TARGET$
			any_scope_article = {
				has_type = defensive_pact
			}
		}
		any_scope_treaty = {
			binds = $TARGET$
			any_scope_article = {
				has_type = foreign_investment_rights
				source_country = $TARGET$
			}
		}
	}
}
