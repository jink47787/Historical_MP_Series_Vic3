average_german_fervor = {
    value = 0
    add = cu:north_german.culture_current_fervor
    add = cu:south_german.culture_current_fervor
    divide = 2
}

average_south_slavic_fervor = {
    value = 0
    add = cu:serb.culture_current_fervor
    add = cu:croat.culture_current_fervor
    add = cu:bosniak.culture_current_fervor
    add = cu:slovene.culture_current_fervor
    if = {
        limit = {
            AND = {
                exists = c:BUL
                c:BUL ?= {
                    has_variable = bul_pursued_greater_yugoslavism
                }
            }
        }
        add = cu:bulgarian.culture_current_fervor
        divide = 5
    }
    else = {
        divide = 4
    }
}
