absolute_state_daimyo_loyalty = { # Refers to var:cached_daimyo_loyalty for performance reasons - used only in movement tooltips
	if = {
		limit = {
			NOT = {
				has_variable = cached_daimyo_loyalty
			}
		}
		value = 0.5 # Neutrality
	}
	else = { # Transforms to decimal
		value = var:cached_daimyo_loyalty
		divide = 100
		min = 0
		max = 1
	}
}

state_daimyo_loyalty = { # Refers to var:cached_daimyo_loyalty for performance reasons
	if = {
		limit = {
			NOT = {
				has_variable = cached_daimyo_loyalty
			}
		}
		value = 0.5 # Neutrality
	}
	else_if = { # If the owner does not have Bakufu, inverts
		limit = {
			owner ?= {
				country_has_emperor_in_power = yes
			}
		}
		value = 100
		subtract = var:cached_daimyo_loyalty
		divide = 100
		min = 0
		max = 1
	}
	else = { # Transforms to decimal
		value = var:cached_daimyo_loyalty
		divide = 100
		min = 0
		max = 1
	}
}

country_levels_owned_by_zaibatsu = {
	value = 0
	if = {
		limit = {
			any_company = {
				company_is_zaibatsu = yes
			}
		}
		every_company = {
			limit = {
				company_is_zaibatsu = yes
			}
			add = this.company_owned_levels
		}
	}
}

fraction_of_country_levels_owned_by_zaibatsu = {
	if = {
		limit = {
			country_levels_owned_by_zaibatsu > 0
		}
		value = this.country_levels_owned_by_zaibatsu
		divide = this.building_levels
	}
	else = {
		value = 0
	}
}
