goa_decision = {
	is_shown = {
		has_dlc_feature = ip4_content
		OR = {
			c:POR ?= ROOT
			c:IBE ?= ROOT
		}
		country_has_state_religion = rel:catholic
		any_scope_state = {
			this = p:x51F0A0.state
		}
		is_subject = no
		NOT = {
			has_variable = goa_decision_initiated
		}
	}

	possible = {
		p:x51F0A0.state = {
			culture_percent_state = {
				target = cu:portuguese
				value > 0.05
			}
			religion_percent_state = {
				target = rel:catholic
				value > 0.9
			}
		}
	}

	when_taken = {
		trigger_event = {
			id = portuguese_colonialism.15
		}
		set_variable = goa_decision_initiated
	}

	ai_chance = {
		value = 1
	}
}

macao_decision = {
	is_shown = {
		has_dlc_feature = ip4_content
		OR = {
			c:POR ?= ROOT
			c:IBE ?= ROOT
		}
		any_scope_state = {
			this = p:xB031E0.state
			is_treaty_port = yes
		}
		is_subject = no
		NOT = {
			has_variable = macao_decision_initiated
		}
	}

	possible = {
		p:xB031E0.state = {
			gdp > 250000
		}
		has_technology_researched = central_banking
	}

	when_taken = {
		trigger_event = {
			id = portuguese_colonialism.16
		}
		custom_tooltip = macao_decision_tt
		set_variable = macao_decision_initiated
	}

	ai_chance = {
		value = 1
	}
}
