tramsfer_ionian_islands_decision = {
	is_shown = {
		hidden_trigger = {
			NOT = {
				has_variable = ionian_islands_decision_var
			}		
		}
		c:GRE ?= this
		is_subject = no
		exists = c:ION
		c:ION ?= {
			is_subject = yes
			is_player = no
			NOT = { is_subject_of = ROOT }
			top_overlord = {
				is_player = no
			}
		}
	}

	possible = {
		is_at_war = no
		c:ION ?= {
			is_at_war = no
			relations:root >= relations_threshold:amicable
			top_overlord = {
				custom_tooltip = {
					text = ionian_island_overlord_not_at_war_with_root_tt
					NOT = { has_war_with = ROOT }
				}
				relations:root >= relations_threshold:amicable
			}
		}
	}

	when_taken = {
		custom_tooltip = {
			text = we_annex_ionian_islands_tt
			set_variable = ionian_islands_decision_var
		}
		c:ION ?= {
			top_overlord = {
				trigger_event = {
					id = greece.6
					days = 0
					popup = yes
				}
			}
		}
	}

	ai_chance = {
		value = 10
	}
}
