pink_map = {
	is_shown = {
		has_dlc_feature = ip4_content
		OR = {
			c:POR ?= this
			c:IBE ?= this
		}
		is_subject = no
		NOT = {
			has_variable = pink_map_sent
		}
		has_variable = portuguese_colonialism_je_completed
	}

	possible = {
		any_scope_state = {
			OR = {
				state_region = s:STATE_KAZEMBE
				state_region = s:STATE_ZAMBIA
				state_region = s:STATE_ZAMBEZI
			}
			is_under_colonization = yes
		}
	}

	when_taken = {
		set_variable = pink_map_seeking
		if = {
			limit = {
				exists = c:GBR
			}
			c:GBR ?= {
				trigger_event = {
					id = portuguese_colonialism.4
				}
			}
			custom_tooltip = pink_map_tt2
		}
		else = {
			add_journal_entry = {
				type = je_the_pink_map
			}
			s:STATE_KAZEMBE = {
				add_claim = ROOT
			}
			s:STATE_ZAMBIA = {
				add_claim = ROOT
			}
			s:STATE_ZAMBEZI = {
				add_claim = ROOT
			}
		}
		set_variable = pink_map_sent
	}

	ai_chance = {
		value = 1
	}
}
