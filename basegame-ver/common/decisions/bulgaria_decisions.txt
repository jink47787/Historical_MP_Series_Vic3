bulgaria_proclaim_empire = {
	is_shown = {
		has_dlc_feature = ip3_content
		c:BUL ?= this
		NOT = { has_variable = bul_proclaimed_bulgarian_empire }
		has_variable = bul_pursued_bulgarian_empire
	}

	possible = {
		has_law_or_variant = law_type:law_monarchy
		owns_entire_state_region = STATE_EASTERN_THRACE
	}

	when_taken = {
		custom_tooltip = bulgarian_empire_tt
		set_variable = bul_proclaimed_bulgarian_empire
		if = {
			limit = {
				s:STATE_EASTERN_THRACE.region_state:BUL ?= {
					is_incorporated = no
				}
			}
			s:STATE_EASTERN_THRACE.region_state:BUL ?= {
				set_state_type = incorporated
				set_hub_name = {
					type = city
					name = HUB_NAME_STATE_EASTERN_THRACE_city_bulgarian_tsarigrad
				}
			}
		}
		if = {
			limit = {
				capital.state_region = {
					NOT = { this = s:STATE_EASTERN_THRACE }
				}
			}
			set_capital = STATE_EASTERN_THRACE
		}

		if = {
			limit = {
				exists = c:GRE
			}
			change_relations = {
				country = c:GRE
				value = -30
			}
		}
		if = {
			limit = {
				exists = c:RUS
			}
			change_relations = {
				country = c:RUS
				value = -30
			}
		}
		if = {
			limit = {
				exists = c:TUR
			}
			change_relations = {
				country = c:TUR
				value = -30
			}
		}
	}


	ai_chance = {
		value = 100
	}
}
