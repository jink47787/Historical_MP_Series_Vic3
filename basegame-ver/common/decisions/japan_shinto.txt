shinto_decision = {
	is_shown = {
		exists = c:JAP
		c:JAP = THIS
		NOT = { country_has_state_religion = rel:shinto }
		NOT = { has_dlc_feature = ep2_content }
	}

	possible = {
		custom_tooltip = {
			text = ruler_is_japanese_emperor_tt
			has_variable = japan_emperor_restored
		}
		ig:ig_devout ?= {
			ig_counts_as_marginal = no
		}
		country_has_monarchy_law = yes
		NOT = { has_law_or_variant = law_type:law_state_religion }
	}

	when_taken = {
		set_state_religion = rel:shinto
		state_religion_switch_effect = yes
		ig:ig_devout ?= {
			hidden_effect = { set_interest_group_name = ig_shinto_monks }
			remove_ideology = ideology_buddhist_moralist
			add_ideology = ideology_shinto_moralist
			set_ig_trait = ig_trait:ig_trait_haibutsu_kishaku
			set_ig_trait = ig_trait:ig_trait_secular_shrine_theory
			set_ig_trait = ig_trait:ig_trait_heavenly_sovereign
			leader ?= {
				change_character_religion = rel:shinto
			}
		}
		random_scope_state = {
			limit = {
				owner = ROOT
				is_capital = yes
			}
			convert_population = {
				target = rel:shinto
				value = 0.4
			}
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				is_incorporated = yes
				NOT = { is_capital = yes }
			}
			convert_population = {
				target = rel:shinto
				value = 0.2
			}
		}
	}

	ai_chance = {
		value = 10
	}
}

# EP2 Religious JE Decisions

shinbutsu_bunri_decision = {
	is_shown = {
		exists = c:JAP
		c:JAP = THIS
		has_dlc_feature = ep2_content
		has_variable = japan_restoration_complete
		has_variable = japan_emperor_restored
		NOT = { has_variable = shinbutsu_bunri_decision_taken }
	}

	possible = {
		ig:ig_devout = { is_powerful = no }
		has_law_or_variant = law_type:law_freedom_of_conscience
	}

	when_taken = {
		add_journal_entry = { type = je_shinbutsu_bunri }
		show_as_tooltip = {
			every_scope_homeland_state = {
				convert_population = {
					target = rel:shinto
					value = 0.10
				}
			}
		}
		set_variable = shinbutsu_bunri_decision_taken
	}

	ai_chance = {
		value = 50
	}
}

elevate_buddhism_decision = {
	is_shown = {
		exists = c:JAP
		c:JAP = THIS
		has_dlc_feature = ep2_content
		has_variable = japan_restoration_complete
		NOT = { has_variable = japan_emperor_restored }
		NOT = { has_variable = elevate_buddhism_decision_taken }
	}

	possible = {
		ig:ig_devout = { is_marginal = no }
		NOT = { has_law_or_variant = law_type:law_total_separation }
		NOT = { has_law_or_variant = law_type:law_state_atheism }
	}

	when_taken = {
		add_journal_entry = { type = je_elevate_buddhism }
		set_variable = elevate_buddhism_decision_taken
	}

	ai_chance = {
		value = 0

		if = {
			limit = { has_ruling_interest_group = ig_devout }
			add = 20
		}
	}
}
