amendment_taikunate = {

	parent = law_autocracy # Bakufu

	allowed_laws = {
		law_landed_voting
	}

	modifier = {
		interest_group_ig_landowners_pol_str_mult = 0.10
		country_legitimacy_govt_total_clout_add = 25
	}

	possible = {
		country_has_primary_culture = cu:japanese
		has_variable = japan_taikunate_var
	}

	would_sponsor = {
		ROOT = {
			is_interest_group_type = ig_landowners
			leader = {
				has_variable = house_tokugawa
			}
		}
	}

	on_activate = {
		hidden_effect = {
			country_assess_daimyo_alignments = yes
		}
	}

	on_deactivate = {
		hidden_effect = {
			country_assess_daimyo_alignments = yes
		}
		remove_variable = japan_taikunate_var
	}

	ai_will_revoke = {
		any_interest_group = {
			is_in_government = yes
			law_stance = {
				law = law_type:law_landed_voting
				value < neutral
			}
		}
	}

	can_repeal = {
		legitimacy >= legitimacy_to_repeal_amendment
	}

	amendment_activism_multiplier = 1
}

amendment_zaibatsu_oligopoly_1 = {

	parent = law_oligarchy

	allowed_laws = { # All economic laws
		law_traditionalism
		law_agrarianism
		law_interventionism
		law_industry_banned
		law_laissez_faire
		law_cooperative_ownership
		law_command_economy
		law_extraction_economy
	}

	modifier = {
		country_loan_interest_rate_mult = -0.2
		country_company_construction_efficiency_bonus_add = 0.15
		state_capitalists_investment_pool_efficiency_mult = 0.25
		country_voting_power_mult = -0.15
	}

	possible = {
		OR = {
			c:JAP ?= this
			was_formed_from = JAP
		}
		has_variable = zaibatsu_victory_var
		NOT = {
			any_active_law = {
				has_amendment = amendment_type:amendment_zaibatsu_oligopoly_2
			}
		}
	}

	would_sponsor = {
		AND = {
			exists = scope:approval
			scope:approval > 0
		}
	}

	ai_will_revoke = {
		always = no
	}

	can_repeal = {
		legitimacy >= legitimacy_to_repeal_amendment
	}

	amendment_activism_multiplier = 1
}
