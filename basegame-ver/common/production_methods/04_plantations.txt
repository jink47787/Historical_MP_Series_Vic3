	}

		workforce_scaled = {
			goods_output_sugar_add = 40
		}

		unscaled = {
			building_laborers_mortality_mult = 0.35
			building_slaves_mortality_mult = 0.35 #sugar is the one slave mortality modifier we kept, for the deadliness
			building_laborers_job_attractiveness_mult = -0.9 #no one wants to work in sugar as a laborer
			### 1.3-1.4 mortality works out to be around replacement rate (births equal deaths), depending on final SOL of the slave. Historically, sugar slaves were expected to live around 7 years before being replaced, with very few children surviving at all, let alone making adulthood. This mortality level is actually no where near as bad as history, but is much more accurate than previous, and safer for game stability.
			building_slaves_standard_of_living_add = -3
			building_minimum_wage_mult = -0.15 #you have to pay sugar workers better than other plantations to get them to do this labor
		}
	}
}
