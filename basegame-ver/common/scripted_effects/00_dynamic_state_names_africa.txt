# North Africa Region

STATE_TUNISIA_state_name_assign = {
    if = {
        limit = {
            owner = {
                any_primary_culture = { has_discrimination_trait = language_italian }
            }
        }

        set_state_name = STATE_TUNISIA_italian
        set_hub_names = italian

    }
    else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
        reset_state_name = yes
        reset_hub_names = yes
    }
}

# Niger Region

STATE_NORTH_CAMEROON_state_name_assign = {
    if = {
        limit = {
            owner = {
                any_primary_culture = { has_discrimination_trait = language_germanophone }
                NOT = { country_has_primary_culture = cu:french }
            }
        }

        set_state_name = STATE_NORTH_CAMEROON_german

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_NORTH_CAMEROON_city_german
        }
    }
    else_if = {
        limit = {
            owner = {
                NOT = { any_primary_culture = { has_discrimination_trait = language_germanophone } }
                country_has_primary_culture = cu:french
            }
        }

        set_state_name = STATE_NORTH_CAMEROON_french
    }
    else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
        reset_state_name = yes
        reset_hub_names = yes
    }
}

# Congo Region

STATE_BAS_CONGO_state_name_assign = {
	if = {
        limit = {
            owner = {
                OR = {
                    c:BEL ?= this
                    AND = {
                        exists = top_overlord
                        exists = c:BEL
                        top_overlord = c:BEL
                    }
                }
            }
        }

        set_state_name = STATE_BAS_CONGO_french
        set_hub_names = french

    }
    else_if = {
        limit = {
            owner = {
                OR = {
                    country_has_primary_culture = cu:dutch
                    country_has_primary_culture = cu:flemish
                    country_has_primary_culture = cu:boer
                    AND = {
                        c:BEL ?= this
                        NOT = {
                            any_scope_state = {
                                state_region = s:STATE_WALLONIA
                            }
                        }
                    }
                }
            }
        }

        set_state_name = STATE_BAS_CONGO_dutch

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_BAS_CONGO_city_dutch
        }
    }
	else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
		reset_state_name = yes
		reset_hub_names = yes
	}
}
