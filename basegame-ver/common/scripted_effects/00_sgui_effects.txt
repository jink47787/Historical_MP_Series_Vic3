# These are used in je_meiji_restoration as well as in gui, fire these effects whenever you manipulate, kill or replace the emperor or one of the daimyos for good measure
je_meiji_restoration_update_japanese_emperor = {
    if = {
        limit = {
            has_variable = emperor_var
        }
        remove_variable = emperor_var
    }
    if = {
        limit = {
            any_scope_character = {
                has_trait = special_character_japanese_emperor
            }
        }
        random_scope_character = {
            limit = {
                has_trait = special_character_japanese_emperor
            }
            save_temporary_scope_as = emperor_var_scope
        }
        set_variable = {
            name = emperor_var
            value = scope:emperor_var_scope
        }
    }
    else_if = {
        limit = {
            is_revolutionary = yes
            country_definition = c:JAP.country_definition
            c:JAP ?= {
                any_scope_character = {
                    has_trait = special_character_japanese_emperor
                }
            }
        }
        c:JAP ?= {
            random_scope_character = {
                limit = {
                    has_trait = special_character_japanese_emperor
                }
                save_temporary_scope_as = emperor_var_scope
            }
        }
        set_variable = {
            name = emperor_var
            value = scope:emperor_var_scope
        }
    }
}

je_meiji_restoration_update_daimyos = {
    if = {
        limit = {
            has_variable_list = daimyos_list
        }
        clear_variable_list = daimyos_list
    }
    if = {
        limit = {
            any_scope_character = {
                has_variable = daimyo_var
            }
        }
        every_scope_character = {
            limit = {
                has_variable = daimyo_var
            }
            root = {
                add_to_variable_list = {
                    name = daimyos_list
                    target = prev
                }
            }
        }
    }
}
