### IP3 Monuments ###

# Kaiserforum

destroy_building_kaiserforum_1 = {
	if = {
		limit = {
			any_scope_state = {
				AND = {
					has_building = building_kaiserforum_1
					any_scope_building = {
						is_building_type = building_kaiserforum_2
						is_under_construction = yes
					}
				}
			}
		}
		random_scope_state = {
			limit = {
				AND = {
					has_building = building_kaiserforum_1
					any_scope_building = {
						is_building_type = building_kaiserforum_2
						is_under_construction = yes
					}
				}

			}
			remove_building = building_kaiserforum_1
		}
	}
}

destroy_building_kaiserforum_2 = {
	if = {
		limit = {
			any_scope_state = {
				AND = {
					has_building = building_kaiserforum_2
					any_scope_building = {
						OR = {
							is_building_type = building_kaiserforum_3
							is_building_type = building_kaiserforum_4
						}
						is_under_construction = yes
					}
				}
			}
		}
		random_scope_state = {
			limit = {
				AND = {
					has_building = building_kaiserforum_2
					any_scope_building = {
						OR = {
							is_building_type = building_kaiserforum_3
							is_building_type = building_kaiserforum_4
						}
						is_under_construction = yes
					}
				}

			}
			remove_building = building_kaiserforum_2
		}
	}
}
