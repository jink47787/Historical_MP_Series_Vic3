country_restore_japanese_emperor = {
	if = {
		limit = {
			any_scope_character = {
				has_trait = special_character_japanese_emperor
			}
		}
		if = {
			limit = {
				has_gov_regency = yes
			}
			heir ?= {
				remove_character_role = heir # Child-Shoguns get no shot
			}
			remove_regency_modifier = yes
		}
		else = {
			heir ?= {
				retire_character = yes
			}
			ruler ?= {
				if = {
					limit = {
						OR = {
							has_role = character_role_general
							has_role = character_role_admiral
						}
					}
					set_commander_rank = 3
				}
				if = {
					limit = {
						NOT = {
							has_role = character_role_politician
						}
					}
					add_character_role = character_role_politician
				}
				remove_character_role = ruler
				set_variable = former_shogun_var
				set_as_interest_group_leader = yes # Landowners leader, typically
				hidden_effect = { set_character_immortal = yes }
				set_variable = { name = temporary_immortality_var value = 36 }
			}
		}
		random_scope_character = {
			limit = {
				has_trait = special_character_japanese_emperor
			}
			set_character_as_ruler = yes
		}
		japan_generate_scripted_heirs = yes
	}
	else = { # Should never happen, but just in case
		if = {
			limit = {
				has_gov_regency = yes
			}
			heir ?= {
				remove_character_role = heir # Child-Shoguns get no shot
			}
			remove_regency_modifier = yes
		}
		else = {
			heir ?= {
				retire_character = yes
			}
			ruler ?= {
				remove_character_role = ruler
				set_variable = former_shogun_var
				set_as_interest_group_leader = yes # Landowners leader, typically
				set_character_immortal = yes
				set_variable = { name = temporary_immortality_var value = 36 }
			}
		}
		country_generate_japanese_emperor = yes
	}
}

japan_generate_imperial_princess_effect = {
	if = { # Moriko
		limit = {
			year >= 1838
			year < 1842
		}
		create_character = {
			template = JAP_moriko_yamato
		}
	}
	if = { # Katsura
		limit = {
			year >= 1842
			year < 1862
		}
		create_character = {
			template = JAP_sumiko_yamato
		}
	}
	if = { # Chikako
		limit = {
			year >= 1862
			year < 1877
		}
		create_character = {
			template = JAP_chikako_yamato
		}
	}
	else = { # Random
		create_character = {
			last_name = "Yamato"
			female = yes
			noble = yes
			interest_group = ig_landowners
			on_created = {
				set_variable = house_yamato
				set_variable = japan_imperial_princess
			}
		}
	}
}

JAP_character_generate_satsuma_daimyo = { # Owner = country, ROOT = predecessor
	owner = {
		if = {
			limit = {
				s:STATE_KYUSHU ?= {
					var:daimyo_chain_id = 0
				}
				game_date >= 1809.4.28
			}
			create_character = {
				template = JAP_nariakira_shimazu
			}
		}
		else_if = {
			limit = {
				s:STATE_KYUSHU ?= {
					var:daimyo_chain_id = 1
				}
				game_date >= 1817.11.28
			}
			create_character = {
				template = JAP_hisamitsu_shimazu
			}
		}
		else_if = {
			limit = {
				s:STATE_KYUSHU ?= {
					var:daimyo_chain_id = 2
				}
				game_date >= 1840.5.22
			}
			create_character = {
				template = JAP_tadayoshi_shimazu
			}
		}
		else_if = {
			limit = {
				s:STATE_KYUSHU ?= {
					var:daimyo_chain_id = 3
				}
				game_date >= 1886.10.20
			}
			create_character = {
				template = JAP_tadashige_shimazu
			}
		}
		else = { # Satsuma people are typically Industrialists
			create_character = {
				last_name = Shimazu
				noble = yes
				culture = cu:japanese
				role = character_role_magnate
				role = character_role_politician
				home_region = STATE_KYUSHU
				interest_group = ig_industrialists
				on_created = {
					if = {
						limit = {
							NOT = {
								has_global_variable = japan_daimyo_abolished
							}
						}
						set_variable = { name = daimyo_var value = s:STATE_KYUSHU }
					}
					designate_as_tozama_daimyo = yes
				}
			}
		}
		s:STATE_KYUSHU ?= {
			change_variable = { name = daimyo_chain_id add = 1 }
		}
	}
}

iwakura_mission_save_learning = {
	var:expedition_leader_storage_var ?= {
		save_scope_as = iwakura_scope
	}
	var:current_expedition_location_var ?= {
		save_scope_as = current_expedition_location_scope
	}
	random_list = {
		40 = {
			var:current_expedition_location_var ?= {
				random_active_law = {
					limit = {
						save_temporary_scope_as = temp_law_scope
						type.progressiveness >= 0 # Will not learn about the glories of Traditionalism
						law_is_governance_principles_law = no # I don't love Meiji! I'm in love with... America!
						NOT = { # Japan does not have this law
							c:JAP = {
								has_law = scope:temp_law_scope.type
							}
						}
						c:JAP ?= { # Japan can enact this law
							any_law = {
								scope:temp_law_scope.type ?= this.type
								law_is_available = yes
							}
						}
					}
					save_scope_as = iwakura_law_learning
					ROOT = {
						add_to_variable_list = { name = iwakura_mission_law_learnings target = prev }
						post_notification = iwakura_mission_learns_laws
					}
				}
			}
		}
		30 = {
			trigger = {
				var:current_expedition_location_var.techs_researched > ROOT.techs_researched
			}
			change_variable = {
				name = iwakura_mission_tech_learnings
				add = {
					value = var:current_expedition_location_var.techs_researched
					subtract = ROOT.techs_researched
					divide = 2  # This can get ridiculous quickly
				}
			}
			post_notification = iwakura_mission_learns_techs
		}
		30 = {
			# Nothing ever happens
		}
	}
}

		set_variable = { name = iwakura_world_political_index value = 0 }
		every_in_list = {
			variable = iwakura_mission_law_learnings
			ROOT = {

					add = prev.type.progressiveness

					name = iwakura_world_political_index

				}
			}
		}

	}

character_get_and_save_coup_reason = {
	owner = {
		set_variable = { name = coup_reason_var value = 0 } # If 0, something is broken
	}
    if = {
		limit = {
			owner ?= {
				any_political_lobby = {
					has_variable = coup_lobby
				}
			}
		}
		owner = {
			set_variable = { name = coup_reason_var value = 5 } # Case 5 - Foreign-Backed Coup
		}
	}
    else_if = {
		limit = {
			has_modifier = modifier_dynastic_coup
		}
		owner = {
			set_variable = { name = coup_reason_var value = 6 } # Case 6 - Dynastic Coup
		}
	}
    else_if = {
		limit = {
			owner ?= { # Case 1 - owner is enacting a law the character does not approve of
				exists = owner.currently_enacting_law
				OR = {
					PREV = {
						law_enactment_stance = { # Refers to any_scope_character
							law = PREV.currently_enacting_law.type # Refers to owner
							value < neutral # Disapproves of the law
						}
					}
					AND = {
						PREV ?= {
							law_enactment_stance = { # Refers to any_scope_character
								law = PREV.currently_enacting_law.type # Refers to owner
								value < approve # Character does not approve of the law
							}
							interest_group ?= {
								law_enactment_stance = { # Refers to any_scope_character's interest group
									law = PREV.currently_enacting_law.type # Refers to owner
									value < neutral # Interest group opposes the law
								}
							}
						}
					}
				}
			}
		}
		owner = {
			set_variable = { name = coup_reason_var value = 1 }
		}
	}
	else_if = {
		limit = {
			this.interest_group = {
				has_modifier = modifier_failed_to_meet_petition # Case 2 - owner has failed a petition for this.interest_group
			}
		}
		owner = {
			set_variable = { name = coup_reason_var value = 2 }
		}
	}
	else_if = {
		limit = {
			this.interest_group ?= {
				has_modifier = negotiation_failed_quest # Case 3 - owner has failed a negotiation for this.interest_group
			}
		}
		owner = {
			set_variable = { name = coup_reason_var value = 3 }
		}
	}
	else_if = {
		limit = {
			OR = {
				has_trait = ambitious # Case 4 - this just has a coupiful personality
				has_trait = imperious
				has_trait = special_character_napoleonic_return
			}
		}
		owner = {
			set_variable = { name = coup_reason_var value = 4 }
		}
	}
	else_if = {
		limit = {
			has_trait = special_character_blanqui_is_free
		}
		owner = {
			set_variable = { name = coup_reason_var value = 7 } # Case 7 - Louis Auguste Blanqui
		}
	}
}

			limit = {
				has_ideology = ideology:ideology_constitutionalist
			}
			remove_ideology = ideology_constitutionalist
			add_ideology = ideology_republican
		}
	}
}

				}
			}
		}

		remove_character_role = character_role_politician
	}
	else_if = {
		limit = {
			has_role = character_role_emperor_of_japan
			NOT = { has_variable = emperor_kicked_out_of_ig_leadership }
		}
		set_variable = emperor_kicked_out_of_ig_leadership # Infinite loop safety variable

	}
