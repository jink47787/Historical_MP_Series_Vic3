# New Spain
set_up_new_spain_default = {
	if = {
		limit = {
			any_subject_or_below = {
				NOT = {
					c:SC1 ?= this
				}
				capital = {
					OR = { # North America, Mexico, and the Central America region
						is_in_geographic_region = geographic_region_colonial_new_spain	
						state_region = s:STATE_PANAMA
					}
				}
			}
		}
		every_subject_or_below = {
			limit = {
				NOT = {
					c:SC1 ?= this
				}
				capital = {
					OR = { # North America, Mexico, and the Central America region
						is_in_geographic_region = geographic_region_colonial_new_spain	
						state_region = s:STATE_PANAMA
					}
				}
			}
			c:SC1 = {
				annex = prev
			}
		}
	}
	custom_tooltip = {
		text = set_up_new_spain_default_tt
		if = {
			limit = {
				any_scope_state = {
					OR = { # North America, Mexico, and the Central America region
						is_in_geographic_region = geographic_region_colonial_new_spain	
						state_region = s:STATE_PANAMA
					}					
				}
			}
			every_scope_state = {
				limit = {
					OR = { # North America, Mexico, and the Central America region
						is_in_geographic_region = geographic_region_colonial_new_spain	
						state_region = s:STATE_PANAMA
					}			
				}
				set_state_owner = c:SC1
			}
		}
		if = {
			limit = {
				any_subject_or_below = {
					NOT = {
						c:SC1 ?= this
					}
					any_scope_state = {
						OR = { # North America, Mexico, and the Central America region
							is_in_geographic_region = geographic_region_colonial_new_spain	
							state_region = s:STATE_PANAMA
						}					
					}
				}
			}
			every_subject_or_below = {
				limit = {
					NOT = {
						c:SC1 ?= this
					}
					any_scope_state = {
						OR = { # North America, Mexico, and the Central America region
							is_in_geographic_region = geographic_region_colonial_new_spain	
							state_region = s:STATE_PANAMA
						}						
					}
				}
				every_scope_state = {
					limit = {
						OR = { # North America, Mexico, and the Central America region
							is_in_geographic_region = geographic_region_colonial_new_spain	
							state_region = s:STATE_PANAMA
						}					
					}
					set_state_owner = c:SC1
