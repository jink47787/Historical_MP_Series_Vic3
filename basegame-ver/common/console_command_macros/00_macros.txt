debug_macro = {
	args = 0
	commands = {
		"fastresearch"
		"fastbuild"
		"tag FRA"
	}
	description = "turn on fastresearch and fastbuild and tag switch to France"
}

debug_money = {
	args = 1
	commands = {
		"money $1"
	}
	description = "give us X money"
}


# they can of course be defined in terms of each other:
debug_args = {
	args = 2
	commands = {
		"debug_macro"
		"debug_money $1"
		"debug_money $2"
	}
	description = "run debug_macro and give us $1 and $2 debug_money"
}

#trivially detectable recursion is prevented, but not cyclic chains - be careful!
