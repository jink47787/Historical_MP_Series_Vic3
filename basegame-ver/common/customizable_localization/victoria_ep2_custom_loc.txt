brz_republic_or_state = {
	type = country
	text = {
		trigger = {
			has_law = law_type:law_social_monarchy
		}
		localization_key = brz_state
	}
	text = {
		trigger = {
			NOT = {
				has_law = law_type:law_social_monarchy
			}
		}
		localization_key = brz_republic
	}
	text = {
		trigger = {
			always = yes
		}
		localization_key = brz_state
	}
}
