je_meiji_restoration_get_faction_sgui = {
    scope = state
    saved_scopes = { frame }

    is_shown = {
        owner ?= {
            has_journal_entry = je_meiji_restoration
        }
    }

    is_valid = {
        owner ?= {
            has_journal_entry = je_meiji_restoration
        }
        trigger_if = {
            limit = {
                scope:frame = 0 # Tokugawa
            }
            OR = {
                root.state_region = s:STATE_KANTO
                root.state_region = s:STATE_TOKAI
                root.state_region = s:STATE_KANSAI
            }
        }
        trigger_else_if = {
            limit = {
                scope:frame = 1 # Date
            }
            root.state_region = s:STATE_TOHOKU
        }
        trigger_else_if = {
            limit = {
                scope:frame = 2 # Hachisuka
            }
            root.state_region = s:STATE_SHIKOKU
        }
        trigger_else_if = {
            limit = {
                scope:frame = 3 # ii
            }
            root.state_region = s:STATE_KYOTO
        }
        trigger_else_if = {
            limit = {
                scope:frame = 4 # Maeda
            }
            root.state_region = s:STATE_HOKUSHINETSU
        }
        trigger_else_if = {
            limit = {
                scope:frame = 5 # Matsumae
            }
            root.state_region = s:STATE_HOKKAIDO
        }
        trigger_else_if = {
            limit = {
                scope:frame = 6 # Mori
            }
            root.state_region = s:STATE_CHUGOKU
        }
        trigger_else_if = {
            limit = {
                scope:frame = 7 # Shimazu
            }
            root.state_region = s:STATE_KYUSHU
        }
        trigger_else = {
            always = no
        }
    }
}
