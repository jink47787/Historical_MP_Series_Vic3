# For je_ryukyu_rivalry

je_ryukyu_rivalry_score_bar = {
	name = "je_ryukyu_rivalry_score_bar"
	desc = "je_ryukyu_rivalry_score_bar_desc"
	second_desc = "je_ryukyu_rivalry_score_bar_second_desc"

	double_sided_bad = yes

	start_value = 0
	min_value = -40
	max_value = 40

	monthly_progress = {
		c:CHI ?= {
			c:JAP ?= {
				c:RYU ?= {
				# Naval Power Projection
					if = {
						limit = {
							c:CHI.naval_power_projection > c:JAP.naval_power_projection
						}
						add = {
							desc = "ryukyu_influence_progress_china_dominant_navy"
							value = -1
						}
					}
					if = {
						limit = {
							c:CHI.naval_power_projection < c:JAP.naval_power_projection
						}
						add = {
							desc = "ryukyu_influence_progress_japan_dominant_navy"
							value = 1
						}
					}
					# Bad relations
					if = {
						limit = {
							c:RYU = {
								"relations(c:JAP)" <= 49
							}
						}
						add = {
							value = -1
							desc = "ryukyu_influence_progress_japan_unfriendly_relation"

						}
					}
					if = {
						limit = {
							c:RYU = {
								"relations(c:CHI)" <= 49
							}
						}
						add = {
							value = 1
							desc = "ryukyu_influence_progress_china_unfriendly_relation"

						}
					}
					# Good relations
					if = {
						limit = {
							c:RYU = {
								"relations(c:JAP)" >= 50
								"relations(c:JAP)" <= 79
							}
						}
						add = {
							value = 1
							desc = "ryukyu_influence_progress_japan_amicable_relation"

						}
					}
					if = {
						limit = {
							c:RYU = {
								"relations(c:JAP)" >= 80
							}
						}
						add = {
							value = 2
							desc = "ryukyu_influence_progress_japan_friendly_relation"

						}
					}
					if = {
						limit = {
							c:RYU = {
								"relations(c:CHI)" >= 50
								"relations(c:CHI)" <= 79
							}
						}
						add = {
							value = -1
							desc = "ryukyu_influence_progress_china_amicable_relation"

						}
					}
					if = {
						limit = {
							c:RYU = {
								"relations(c:CHI)" >= 80
							}
						}
						add = {
							value = -2
							desc = "ryukyu_influence_progress_china_friendly_relation"

						}
					}
					# Money Transfer
					if = {
						limit = {
							c:CHI ?= {
								any_scope_treaty = {
									any_scope_article = {
										has_type = money_transfer
										input_quantity >= 100
										target_country = c:RYU
									}
								}
							}
						}
						add = {
							desc = "ryukyu_influence_progress_china_money_transfer"
							c:CHI ?= {
								every_scope_treaty = {
									limit = {
										any_scope_article = {
											has_type = money_transfer
											target_country = c:RYU
											input_quantity >= 100
										}
									}
									every_scope_article = {
										limit = {
											has_type = money_transfer
											target_country = c:RYU
										}
										add = this.input_quantity
										divide = 1000
										multiply = -1
										max = 0
										min = -2
									}
								}
							}
						}
					}
					if = {
						limit = {
							c:JAP ?= {
								any_scope_treaty = {
									any_scope_article = {
										has_type = money_transfer
										input_quantity >= 100
										target_country = c:RYU
									}
								}
							}
						}
						add = {
							desc = "ryukyu_influence_progress_japan_money_transfer"
							c:JAP ?= {
								every_scope_treaty = {
									limit = {
										any_scope_article = {
											has_type = money_transfer
											target_country = c:RYU
											input_quantity >= 100
										}
									}
									every_scope_article = {
										limit = {
											has_type = money_transfer
											target_country = c:RYU
										}
										add = this.input_quantity
										divide = 1000
										max = 2
										min = 0
									}
								}
							}
						}
					}
					# Actions
					if = {
						limit = {
							exists = global_var:ryukyu_rivalry_reaffirm_suzerainty_jap
						}
						add = {
						desc = "ryukyu_influence_progress_japan_reaffirm_suzerainty"
						value = 3
						}
					}
					if = {
						limit = {
							exists = global_var:ryukyu_rivalry_reaffirm_suzerainty_chi
						}
						add = {
						desc = "ryukyu_influence_progress_china_reaffirm_suzerainty"
						value = -3
						}
					}
					if = {
						limit = {
							c:JAP ?= {
								has_modifier = modifier_ryukyu_rivalry_officer_training
							}
						}
						add = {
						desc = "ryukyu_influence_progress_japan_inspect_defenses"
						value = 3
						}
					}
					if = {
						limit = {
							c:CHI ?= {
								has_modifier = modifier_ryukyu_rivalry_officer_training
							}
						}
						add = {
						desc = "ryukyu_influence_progress_china_inspect_defenses"
						value = -3
						}
					}
					if = {
						limit = {
							c:JAP ?= {
								ig:ig_devout = {
									has_modifier = modifier_ryukyu_rivalry_honor_kikoeogimi
								}
							}
						}
						add = {
						desc = "ryukyu_influence_progress_japan_honor_kikoeogimi"
						value = 3
						}
					}
					if = {
						limit = {
							c:CHI ?= {
								ig:ig_devout = {
									has_modifier = modifier_ryukyu_rivalry_honor_kikoeogimi
								}
							}
						}
						add = {
						desc = "ryukyu_influence_progress_china_honor_kikoeogimi"
						value = -3
						}
					}
					# Obligation
					if = {
						limit = {
							c:RYU ?= { owes_obligation_to = c:JAP }
						}
						add = {
						desc = "ryukyu_influence_progress_japan_owe_obligation"
						value = 1
						}
					}
					if = {
						limit = {
							c:RYU ?= { owes_obligation_to = c:CHI }
						}
						add = {
						desc = "ryukyu_influence_progress_china_owe_obligation"
						value = -1
						}
					}
					# Ryukyu actions
					if = {
						limit = {
							c:RYU = {
								prestige >= 30
							}
						}
						if = {
							limit = {
								scope:journal_entry = {
									"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" < -1
								}
							}
							add = {
							desc = "ryukyu_influence_progress_bar_ryukyu_prestige_high"
							value = 2
							}
						}
						else_if = {
							limit = {
								scope:journal_entry = {
									"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" > 1
								}
							}
							add = {
							desc = "ryukyu_influence_progress_bar_ryukyu_prestige_high"
							value = -2
							}
						}
						else = {
							add = {
							desc = "ryukyu_influence_progress_bar_ryukyu_prestige_low"
							value = 0
							}
						}
					}
					else_if = {
						limit = {
							c:RYU ?= {
								AND = {
									prestige < 30
									prestige >= 15
								}

							}
						}
						if = {
							limit = {
								scope:journal_entry = {
									"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" < -1
								}
							}
							add = {
							desc = "ryukyu_influence_progress_bar_ryukyu_prestige"
							value = 1
							}
						}
						else_if = {
							limit = {
								scope:journal_entry = {
									"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" > 1
								}
							}
							add = {
							desc = "ryukyu_influence_progress_bar_ryukyu_prestige"
							value = -1
							}
						}
						else = {
							add = {
							desc = "ryukyu_influence_progress_bar_ryukyu_prestige_low"
							value = 0
							}
						}
					}
					else = {
						add = {
						desc = "ryukyu_influence_progress_bar_ryukyu_prestige_low"
						value = 0
						}
					}
				}
			}
		}
	}
}

je_ryukyu_rivalry_progress_bar = {
	name = "je_ryukyu_rivalry_progress_bar"
	desc = "je_ryukyu_rivalry_progress_bar_desc"

	default_bad = yes

	start_value = 0
	min_value = 0
	max_value = 100

	monthly_progress = {
		c:RYU ?= {
			if = {
				limit = {
					scope:journal_entry = {
						AND = {
							"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" >= -20
							"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" <= 20
						}
					}
				}
				add = {
					value = 0.5
					desc = "ryukyu_influence_progress_neither_dominant"

				}
			}
		}
	}
}
