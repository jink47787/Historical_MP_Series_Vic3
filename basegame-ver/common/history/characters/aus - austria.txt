					set_character_immortal = yes
					# This is abhorrent spaghetti script, and I will now explain why.
					# When the game starts, all interest groups have randomly generated leaders.
					# Historical leaders replace the randomly generated leader after "global" generation.
					# Making this journal entry available at game start, thus, means Metternich doesn't exist at time of initialisation.
					# Attempting to save Metternich as a scope in the immediate then either returns null or the randomly generated leader prior to him.
					# This is why we add the journal entry here rather than in the country or global file, because for some reason this is the only way for this to not happen.
					# Most journal entries get around this through either scoping to ruler or being added several ticks into the game.
					# If we ever revise generation steps or find a better way to do this, we will fix this.
					# - Victoria
					owner = {
						add_journal_entry = { type = je_metternich }
					}
				}
