gov_colonial_administration_gov_in_chief = {
	transfer_of_power = dictatorial
	new_leader_on_reform_government = no

	male_ruler = "RULER_TITLE_GOVERNOR_IN_CHIEF"
	female_ruler = "RULER_TITLE_GOVERNOR_IN_CHIEF"

	possible = {
		has_law = law_type:law_colonial_administration
		OR = {
			is_subject_type = subject_type_colony
			is_subject_type = subject_type_dominion
		}
		c:SIL ?= this
		c:SIL ?= {
			any_scope_state = {
				NOT = {
					state_region = s:STATE_SIERRA_LEONE
				}
			}
		}
	}

	on_government_type_change = {
		change_to_dictatorial = yes
	}
	on_post_government_type_change = {
		post_change_to_dictatorial = yes
	}
}

gov_colonial_administration = {
	transfer_of_power = dictatorial
	new_leader_on_reform_government = no

	male_ruler = "RULER_TITLE_GOVERNOR_GENERAL"
	female_ruler = "RULER_TITLE_GOVERNOR_GENERAL"

	possible = {
		has_law = law_type:law_colonial_administration
		OR = {
			is_subject_type = subject_type_colony
			is_subject_type = subject_type_dominion
		}
		OR = {
			AND = {
				AND = { exists = c:NET is_subject_of = c:NET }
				AND = { exists = c:DEI c:DEI = ROOT }
				is_subject_type = subject_type_dominion
			}
			NOT = { #handled in gov_colonial_administration_spa
				overlord ?= {
					c:SPA ?= this
				}
				is_subject_type = subject_type_colony
				is_country_type = colonial
				top_overlord = { NOT = { is_country_type = colonial } }
				any_primary_culture = {
					has_discrimination_trait_group = heritage_group_european
				}
				is_not_in_geographic_region = geographic_region_europe
				is_not_in_geographic_region = geographic_region_africa
			}
			NOT = { #handled in gov_colonial_administration_spa
				is_subject_type = subject_type_colony
				is_country_type = colonial
				top_overlord = { NOT = { is_country_type = colonial } }
				top_overlord = {
					was_formed_from = SPA
				}
				any_primary_culture = {
					has_discrimination_trait_group = heritage_group_european
				}
				is_not_in_geographic_region = geographic_region_europe
				is_not_in_geographic_region = geographic_region_africa
			}
			AND = {
				has_variable = senegal_subject_var
				is_subject = yes
			}
			AND = {
				has_variable = niger_subject_var
				is_subject = yes
			}
			AND = {
				has_variable = ethiopia_subject_var
				is_subject = yes
			}
			AND = {
				has_variable = zanj_subject_var
				is_subject = yes
			}
			AND = {
				has_variable = congo_subject_var
				is_subject = yes
			}
		}
	}

	on_government_type_change = {
		change_to_dictatorial = yes
	}
	on_post_government_type_change = {
		post_change_to_dictatorial = yes
	}
}
