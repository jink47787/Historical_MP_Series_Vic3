character_role_magnate = {
	texture = "gfx/interface/icons/character_role_icons/magnate.dds"

	type = magnate
	priority = 10

	auto_assigned = yes
	
	title_format = custom
	title_preposition = PREP_IN
	
	spawn_characters_to_pool = yes
	
	pool_spawn_target_country_wide = {
		value = country_rank # (0 = decentralized, 7 = great power)
		
		# Subtracting by 1 and dividing by 2 results in:
		# Great Power = 3 magnates
		# Major Power/Unrecognized Major Power = 2 magnates
		# Minor Power/Unrecognized Regional Power = 1 magnate
		# Others = no magnates
		subtract = 1
		divide = 2 
	}
	
	# root = country
	holding_scores = {
		holding_manor_houses_score = {
			type = "building_manor_house"
			score = {
				value = 100
				if = {
					limit = {
						scope:is_monarch = yes
					}
					add = 10000
				}
				else_if = {
					limit = {
						scope:interest_group ?= { 
							OR = {
								is_interest_group_type = ig_rural_folk
								is_interest_group_type = ig_landowners	
							}
						}
					}
					add = 1000
				}
				else_if = {
					limit = {
						scope:interest_group ?= { 
							OR = {
								is_interest_group_type = ig_devout
								is_interest_group_type = ig_armed_forces
							}
						}
					}
					add = 200
				}
				
				if = {
					limit = {
						scope:state ?= { NOT = { has_building = building_manor_house } }
					}
					multiply = 0
				}						
			}
		}
		holding_financial_districts_score = {
			type = "building_financial_district"
			score = {
				value = 100				
				if = {
					limit = {
						scope:interest_group ?= { is_interest_group_type = ig_industrialists }
					}
					add = 1000
				}
				else_if = {
					limit = {
						scope:interest_group ?= { 
							OR = {
								is_interest_group_type = ig_petty_bourgeoisie
								is_interest_group_type = ig_trade_unions
							}
						}
					}
					add = 200
				}	

				if = {
					limit = {
						scope:state ?= { NOT = { has_building = building_financial_district } }
					}
					multiply = 0
				}				
			}
		}			
	}
}
