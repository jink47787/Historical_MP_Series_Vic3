je_taming_the_north = {
    icon = "gfx/interface/icons/event_icons/event_map.dds"

    group = je_group_historical_content

    scripted_button = button_je_taming_the_north_establish_colonies
    scripted_button = button_je_taming_the_north_integrate_ainu
    scripted_button = button_je_taming_the_north_sapporo_agricultural_college
    scripted_button = button_je_taming_the_north_agriculture_potentials
    scripted_button = button_je_taming_the_north_agriculture_arable_land

    is_shown_when_inactive = {
        has_dlc_feature = ep2_content
        c:JAP ?= this
        s:STATE_HOKKAIDO = {
            any_scope_state = {
                owner = root
            }
        }
    }

    possible = {
        owns_entire_state_region = STATE_HOKKAIDO
        s:STATE_HOKKAIDO.region_state:JAP = {
            is_incorporated = yes
            any_scope_building = {
                is_building_type = building_urban_center
                level >= 1
            }
        }
    }

    immediate = {
        trigger_event = {
            id = hokkaido_events.1
            popup = yes
        }
        s:STATE_HOKKAIDO = {
            random_scope_state = {
                limit = {
                    owner = root
                }
                save_scope_as = hokkaido_state_scope
            }
        }
        s:STATE_SAKHALIN = {
            save_scope_as = sakhalin_state_region_scope
        }
        set_variable = {
            name = hokkaido_agriculture_potentials_counter_var
            value = 0
        }
        set_variable = {
            name = hokkaido_agriculture_arable_counter_var
            value = 0
        }
        set_variable = {
            name = ainu_friendship_var
            value = 0
        }
    }

    on_monthly_pulse = {
    }

    on_yearly_pulse = {
        random_events = {
        }
    }

    complete = {
        s:STATE_HOKKAIDO.region_state:JAP = {
            state_population > 500000
            gdp > 1000000
        }
        custom_tooltip = {
            text = taming_the_north_agriculture_potentials_goal_tt
            exists = var:hokkaido_agriculture_potentials_counter_var
            var:hokkaido_agriculture_potentials_counter_var >= 3
        }
        custom_tooltip = {
            text = taming_the_north_agriculture_arable_goal_tt
            exists = var:hokkaido_agriculture_arable_counter_var
            var:hokkaido_agriculture_arable_counter_var >= 3
        }
        custom_tooltip = {
            text = taming_the_north_colonies_goal_tt
            has_variable = button_je_taming_the_north_establish_colonies_var
        }
        
    }

    on_complete = {
        trigger_event = {
            id = hokkaido_events.7
            popup = yes
        }
        show_as_tooltip = {
            add_modifier = {
                name = modifier_northern_learnings
                days = very_long_modifier_time
            }
            s:STATE_HOKKAIDO.region_state:JAP ?= {
                add_modifier = {
                    name = modifier_tamed_north
                    days = very_long_modifier_time
                }
            }
            if = {
                limit = {
                    NOT = {
                        has_journal_entry = je_taming_the_north
                    }
                }
                custom_tooltip = {
                    text = je_taming_the_north_additional_goal_pre
                }
            }
            else_if = {
                limit = {
                    NOT = {
                        owns_entire_state_region = STATE_SAKHALIN
                    }
                }
                custom_tooltip = {
                    text = je_taming_the_north_additional_goal
                }
            }
            else_if = {
                limit = {
                    owns_entire_state_region = STATE_SAKHALIN
                }
                custom_tooltip = je_taming_the_north_additional_goal_completed
                s:STATE_SAKHALIN.region_state:JAP ?= {
                    add_modifier = {
                        name = modifier_tamed_north
                        days = very_long_modifier_time
                    }
                }
            }
        }
        remove_variable = hokkaido_agriculture_potentials_counter_var
        remove_variable = hokkaido_agriculture_arable_counter_var
        remove_variable = button_je_taming_the_north_establish_colonies_var
    }

    event_outcome_completed_effect_desc = {
        header = hokkaido_events.7.a
        effect = {
            add_modifier = {
                name = modifier_external_applications
                days = long_modifier_time
            }
        }
    }

    event_outcome_completed_effect_desc = {
        header = hokkaido_events.7.b
        effect = {
            add_modifier = {
                name = modifier_domestic_applications
                days = long_modifier_time
            }
        }
    }

    fail = {
        NOT = {
            has_state_in_state_region = STATE_HOKKAIDO
        }
    }

    on_fail = {
        trigger_event = {
            id = hokkaido_events.8
            popup = yes
        }
    }

    event_outcome_failed_effect_desc = {
        header = hokkaido_events.8.a
        effect = {
            add_radicals = {
                value = medium_radicals
                culture = cu:japanese
            }
            ig:ig_industrialists = {
                add_modifier = {
                    name = modifier_lost_northern_investments
                    days = normal_modifier_time
                }
            }
            ig:ig_petty_bourgeoisie = {
                add_modifier = {
                    name = modifier_lost_northern_investments
                    days = normal_modifier_time
                }
            }
        }
    }

    invalid = {
    }

    weight = 5000

      
    should_be_pinned_by_default_uninvolved_or_context = yes
}
