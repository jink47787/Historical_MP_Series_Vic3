je_iwakura_mission = { # I am possibly in Spain, but only if it's doing really well.
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_historical_content

   # Added by event/decision

   scripted_button = je_iwakura_mission_recall_mission

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = iwakura_mission_status_loc_returning_home
				trigger = {
					exists = var:expedition_leader_storage_var
					exists = var:current_expedition_location_var
					OR = {
						has_variable = recalled_iwakura_mission
						has_variable = iwakura_mission_en_route_home
					}
				}
			}
			triggered_desc = {
				desc = iwakura_mission_status_loc_returned_home
				trigger = {
					has_variable = iwakura_mission_returned_home
					exists = var:expedition_leader_storage_var
				}
			}
			triggered_desc = {
				desc = blank_tt
				trigger = {
					NOT = { exists = var:current_expedition_location_var }
					NOT = { exists = var:expedition_leader_storage_var }
				}
			}
			triggered_desc = {
				desc = iwakura_mission_status_loc_leader
				trigger = {
					NOT = { exists = var:current_expedition_location_var }
					exists = var:expedition_leader_storage_var
				}
			}
			triggered_desc = {
				desc = iwakura_mission_status_loc_leader_and_location
				trigger = {
					exists = var:current_expedition_location_var
					exists = var:expedition_leader_storage_var
				}
			}
			triggered_desc = {
				desc = iwakura_mission_status_loc_location
				trigger = {
					exists = var:current_expedition_location_var
					NOT = { exists = var:expedition_leader_storage_var }
				}
			}
		}
	}

	immediate = {
		set_variable = { name = iwakura_mission_tech_learnings value = 0 }
		random_scope_character = {
			limit = {
				has_variable = expedition_leader_var
			}
			save_scope_as = expedition_leader
		}
		trigger_event = { id = iwakura_mission.4 days = 25 popup = yes }
	}

	on_monthly_pulse = {
        effect = {
			if = {
				limit = {
					has_variable = current_expedition_location_var
				}
				iwakura_mission_save_learning = yes
			}
        }
	}

	on_weekly_pulse = {
        effect = {
			if = {
				limit = {
					has_variable = iwakura_mission_in_continental_europe
				}
				random_list = {
					25 = {
						trigger_event = iwakura_mission.7
					}
					50 = {
						# Nothing ever happens
					}
				}
			}
        }
	}

	custom_completion_header = je_iwakura_mission_complete_header

	complete = {
		custom_tooltip = {
			text = iwakura_mission_returned_to_japan_tt
			AND = {
				has_variable = iwakura_mission_returned_home
				has_variable = complete_iwakura_mission
			}
		}
	}

	custom_on_completion_header = je_iwakura_mission_on_complete_header

	on_complete = {
		trigger_event = { id = iwakura_mission.2 popup = yes }
		custom_tooltip = iwakura_mission_leader_change_ideology_tt
		custom_tooltip = iwakura_mission_leader_traits_tt
		custom_tooltip = effects_depend_on_visited_countries_tt
	}

	weight = 5000

    should_be_pinned_by_default_involved = yes
	should_be_pinned_by_default_uninvolved_or_context = yes
}
