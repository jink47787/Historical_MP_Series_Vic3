# Purveyor of Progress JE
je_india_railway = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	group = je_group_british_india

	is_shown_when_inactive = {
		has_dlc_feature = ip2_content
		is_player = yes
		NOT = { has_global_variable = india_railway_done }
		is_in_geographic_region = geographic_region_india
	}

	possible = {
		hidden_trigger = {
			OR = {
				c:BIC ?= root
				c:RAJ ?= root
				is_in_geographic_region = geographic_region_india
			}
		}
		AND = { # Country has a presence near all state regions for the line
			sr:region_north_india = { root_owns_state_in_region = yes }
			sr:region_south_india = { root_owns_state_in_region = yes }
		}
		has_technology_researched = railways
	}

	complete = {
		# Completed the line. Now it doesn't matter who owns the railway.
		AND = {
			trigger_if = {
				limit = {
					any_state_region_in_delhi_calcutta_line = {
						has_building = building_railway
						count = all
					}
				}
				custom_tooltip = {
					text = delhi_calcutta_line_complete_tt
					any_state_region_in_delhi_calcutta_line = {
						has_building = building_railway
						count = all
					}
				}
			}
			trigger_else = {
				custom_tooltip = {
					text = delhi_calcutta_line_status_tt
					any_state_region_in_delhi_calcutta_line = {
						has_building = building_railway
						count = all
					}
				}
			}
			trigger_if = {
				limit = {
					any_state_region_in_allahabad_bombay_line = {
						has_building = building_railway
						count = all
					}
				}
				custom_tooltip = {
					text = allahabad_bombay_line_complete_tt
					any_state_region_in_allahabad_bombay_line = {
						has_building = building_railway
						count = all
					}
				}
			}
			trigger_else = {
				custom_tooltip = {
					text = allahabad_bombay_line_status_tt
					any_state_region_in_allahabad_bombay_line = {
						has_building = building_railway
						count = all
					}
				}
			}
			trigger_if = {
				limit = {
					any_state_region_in_bombay_madras_line = {
						has_building = building_railway
						count = all
					}
				}
				custom_tooltip = {
					text = bombay_madras_line_complete_tt
					any_state_region_in_bombay_madras_line = {
						has_building = building_railway
						count = all
					}
				}
			}
			trigger_else = {
				custom_tooltip = {
					text = bombay_madras_line_status_tt
					any_state_region_in_bombay_madras_line = {
						has_building = building_railway
						count = all
					}
				}
			}
			trigger_if = {
				limit = {
					any_state_region_in_gujarat_bombay_line = {
						has_building = building_railway
						count = all
					}
				}
				custom_tooltip = {
					text = gujarat_bombay_line_complete_tt
					any_state_region_in_gujarat_bombay_line = {
						has_building = building_railway
						count = all
					}
				}
			}
			trigger_else = {
				custom_tooltip = {
					text = gujarat_bombay_line_status_tt
					any_state_region_in_gujarat_bombay_line = {
						has_building = building_railway
						count = all
					}
				}
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_BOMBAY
			}
			save_scope_as = bombay_scope #for loc
		}
		every_state_region_in_delhi_calcutta_line = {
			root = {
				add_to_variable_list = {
					name = delhi_calcutta_states_list
					target = prev
				}
			}
		}
		every_state_region_in_allahabad_bombay_line = {
			root = {
				add_to_variable_list = {
					name = allahabad_bombay_states_list
					target = prev
				}
			}
		}
		every_state_region_in_bombay_madras_line = {
			root = {
				add_to_variable_list = {
					name = bombay_madras_states_list
					target = prev
				}
			}
		}
		every_state_region_in_gujarat_bombay_line = {
			root = {
				add_to_variable_list = {
					name = gujarat_bombay_states_list
					target = prev
				}
			}
		}
	}

	on_complete = {
		set_global_variable = india_railway_done
		trigger_event = { id = india_railway.5 }
	}

    event_outcome_completed_desc = {
        first_valid = {
            triggered_desc = {
                desc = event_effects_india_railway.5_tt
            }
        }
    }

	invalid = {
		NOT = {
			any_scope_state = {
				is_in_geographic_region = geographic_region_india
			}
		}
	}

	on_yearly_pulse = {
		effect = {

		}
		random_events = {
			100 = 0
			10 = india_railway.1
			10 = india_railway.2
			10 = india_railway.3
			10 = india_railway.4
		}
	}

	weight = 50
}
