je_shinbutsu_bunri = {
	icon = "gfx/interface/icons/religion_icons/shinto.dds"

	group = je_group_historical_content

	possible = {
		# triggered by shinbutsu_bunri_decision
	}

	scripted_button = je_taikyo_proclamation_button

	immediate = {
		every_scope_homeland_state = {
			convert_population = {
				target = rel:shinto
				value = 0.10
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			random_scope_state = {
				limit = {
					state_region = { is_homeland = cu:japanese }
					religion_percent_state = {  
						target = rel:shinto
						value <= 0.5
					}
				}
				convert_population = {
					target = rel:shinto
					value = 0.02
				}
			}
		}
		random_events = {
			100 = 0
			10 = japan_religion.1
			10 = japan_religion.2
			10 = japan_religion.3
			10 = japan_religion.4
			10 = japan_religion.5
			10 = japan_religion.6
		}
	}

	complete = {
		any_scope_homeland_state = {
			religion_percent_state = {
				target = rel:shinto
				value >= 0.6
			}
			count = all
		}
	}

	on_complete = {
		add_modifier = {
			name = arahitogami
		}
		# TODO (Success event?)
	}

	fail = {
		OR = {
			has_law_or_variant = law_type:law_total_separation
			has_law_or_variant = law_type:law_state_atheism
			AND = {
				NOT = { country_has_state_religion = rel:shinto }
				NOT = { has_law_or_variant = law_type:law_freedom_of_conscience }
			}
		}
	}

	invalid = {
	}

	weight = 5000

	timeout = 9125

	should_be_pinned_by_default_uninvolved_or_context = yes
}

je_elevate_buddhism = {
	icon = "gfx/interface/icons/religion_icons/mahayana.dds"

	group = je_group_historical_content

	possible = {
		# triggered by elevate_buddhism_decision
	}

	immediate = {
		set_variable = {
			name = je_elevate_buddhism_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					ig:ig_devout = { is_powerful = yes }
				}
				change_variable = {
					name = je_elevate_buddhism_var
					add = 1
				}
			}
		}
		random_events = {
			100 = 0
			10 = japan_religion.7
			10 = japan_religion.8
			10 = japan_religion.9
			10 = japan_religion.6
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		ig:ig_devout = { is_powerful = yes }
	}

	on_complete = {
		add_modifier = {
			name = extended_danka_system
		}
		# TODO (Success event?)
	}

	fail = {
		OR = {
			ig:ig_devout = { is_marginal = yes }
			has_law_or_variant = law_type:law_total_separation
			has_law_or_variant = law_type:law_state_atheism
		}
	}

	invalid = {
	}

	current_value = {
		value = root.var:je_elevate_buddhism_var
	}

	goal_add_value = {
		add = 60
	}

	progressbar = yes

	weight = 5000

	timeout = 7300

	should_be_pinned_by_default_uninvolved_or_context = yes
}
