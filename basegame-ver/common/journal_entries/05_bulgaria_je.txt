je_prussia_of_the_balkans = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_historical_content

	scripted_progress_bar = bulgaria_military_progress_bar

	immediate = {
		set_variable = {
			Name = army_projection_target
			Value = 1000
		}
	}

	is_shown_when_inactive = {
		has_dlc_feature = ip3_content
		OR = {
			c:BUL ?= this
			AND = {
				NOT = { exists = c:BUL }
				c:TUR ?= this
				game_date <= 1840.1.1
			}
		}
	}

	possible = {
		trigger_if = {
			limit = {
				c:TUR ?= this
			}
			custom_tooltip = {
				text = TUR_how_to_play_bulgaria_tt
				always = no
			}
		}
		trigger_else_if = {
			c:BUL ?= THIS
			is_subject = no
		}
	}

	on_weekly_pulse = {
		effect = {
			scope:journal_entry = {
				set_bar_progress = {
					value = ROOT.army_power_projection
					name = bulgaria_military_progress_bar
				}
			}
		}
	}

	modifiers_while_active = {
		modifier_balgaria_arming
	}

	complete = { #This is going to take the player some time. A long term goal. They already have modifiers to help.
		trigger_if = {
			limit = {
				c:BUL ?= this
			}
			custom_tooltip = {
				text = bularia_military_progress_tt
				ROOT.army_power_projection >= var:army_projection_target
			}
		}
	}

	on_complete = {
		add_modifier = {
			name = modifier_prussia_of_the_balkans
		}
		add_modifier = {
			name = modifier_bul_militarisation_complete
			days = very_long_modifier_time
			is_decaying = yes
		}
		trigger_event = { id = bulgaria_events.1 popup = yes}
	}

	event_outcome_completed_effect_desc = {
		header = bulgaria_events.1.a
		effect = {
			every_state = {
				limit = {
					is_homeland_of_country_cultures = root
					NOT = {
						owner = {
							this = root
						}
					}
				}
				state_region = {
					add_claim = root
				}
			}
			if = {
				limit = {
					NOT = {
						any_scope_state = {
							state_region = s:STATE_EASTERN_SERBIA
						}
					}
				}
				s:STATE_EASTERN_SERBIA = {
					add_claim = root
				}
			}
			if = {
				limit = {
					NOT = {
						any_scope_state = {
							state_region = s:STATE_EASTERN_THRACE
						}
					}
				}
				s:STATE_EASTERN_THRACE = {
					add_claim = root
				}
			}
			if = {
				limit = {
					NOT = {
						any_scope_state = {
							state_region = s:STATE_MACEDONIA
						}
					}
				}
				s:STATE_MACEDONIA = {
					add_claim = root
				}
			}
			custom_tooltip = bulgarian_empire_homeland_tt
		}
	}

	event_outcome_completed_effect_desc = {
		header = bulgaria_events.1.b
		effect = {
			custom_tooltip = bulgaria_unlock_yugoslavia_tt
			set_variable = bul_pursued_greater_yugoslavism

			add_modifier = {
				name = modifier_bul_pursuing_greater_yugoslavism
				days = very_long_modifier_time
				is_decaying = yes
			}
		}
	}

	event_outcome_completed_effect_desc = {
		header = bulgaria_events.1.c
		effect = {
			add_modifier = {
				name = modifier_bulgarian_renaissance
				days = very_long_modifier_time
				is_decaying = yes
			}
		}
	}

	invalid = {
	}

	on_invalid = {
	}

	fail = {
		is_subject = yes
	}

	on_fail = {
	}

	weight = 100
	can_revolution_inherit = yes
	transferable = no
	should_be_pinned_by_default = yes
}
