	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_historical_content

			limit = {
				is_political_movement_type = movement_tokugawa_loyalist
			}
			save_scope_as = movement_tokugawa_loyalist_scope

	on_complete = {

			id = tenpo_events.3
			popup = yes
		}
		hidden_effect = {
			remove_variable = tenpo_gdp_goal

	event_outcome_completed_effect_desc = {
		header = je_tenpo_crisis_success_outcome_1_tt
		effect = {
			add_modifier = {
				name = modifier_tenpo_crisis_reward_hardliner
				days = long_modifier_time
			}
			add_loyalists = {
				value = medium_radicals
				strata = upper
			}
		}
	}

	event_outcome_completed_effect_desc = {
		header = je_tenpo_crisis_success_outcome_2_tt
		effect = {
			add_modifier = {

				days = long_modifier_time

		header = je_tenpo_crisis_success_outcome_3_tt
		effect = {
			add_modifier = {
				name = modifier_tenpo_crisis_reward_balanced
				days = long_modifier_time

				value = medium_radicals
				strata = middle

	on_timeout = {
		trigger_event = {
			id = tenpo_events.4
			popup = yes
		}
		hidden_effect = {
			remove_variable = tenpo_gdp_goal
		}
	}

	event_outcome_timeout_effect_desc = {
        header = tenpo_events.4.a
        effect = {
			add_modifier = {
				name = modifier_tenpo_fail
				days = normal_modifier_time

    event_outcome_timeout_effect_desc = {
        header = tenpo_events.4.b

        }
