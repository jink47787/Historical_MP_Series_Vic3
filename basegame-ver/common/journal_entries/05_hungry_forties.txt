	group = je_group_crises
	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	should_be_pinned_by_default = yes
	weight = 10000

	is_shown_when_inactive = {
		country_is_in_europe = yes
		game_date <= 1860.1.1
	}

	immediate = {
		trigger_event = { id = hungry_forties.1 popup = yes }
	}

	possible = {
		custom_tooltip = {
			text = hungry_forties_triggered_tt
			has_global_variable = hungry_forties_var
		}
	}

	on_timeout = {
		trigger_event = { id = hungry_forties.2 popup = yes }
		remove_global_variable = hungry_forties_var
	}

	event_outcome_timeout_desc = {
		first_valid = {
			triggered_desc = {
				desc = event_effects_hungry_forties.2_tt
			}
		}
	}
}
