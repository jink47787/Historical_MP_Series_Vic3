je_balkan_national_awakenings = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		has_dlc_feature = ip3_content
		capital = {
			OR = {
				region = sr:region_balkans
				region = sr:region_danubia
			}
		}
	}

	possible = {
		has_technology_researched = nationalism
		NOT = {
			has_law_or_variant = law_type:law_subjecthood
		}
		OR = {
			is_subject_type = subject_type_protectorate
			is_subject_type = subject_type_personal_union
			is_subject = no
		}
	}

	immediate = {
		random_primary_culture = {
			save_scope_as = primary_culture_scope
		}
		if = {
			limit = {
				scope:primary_culture_scope ?= {
					culture_national_awakening_occurred = no
				}
			}
			start_national_awakening = {
				culture = scope:primary_culture_scope
				months = 120
				state_region = capital.state_region
			}
		}
		set_variable = {
			name = fervour_bar_var
			value = 0
		}
		ig:ig_armed_forces ?= {
			save_scope_as = armed_forces_ig
		}
		je:je_balkan_national_awakenings ?= {
			add_modifier = {
				name = support_separatism_nationalist_propaganda
				multiplier = 5
			}
		}
	}

	on_weekly_pulse = {
		effect = {
			set_variable = {
				name = fervour_bar_var
				value = scope:primary_culture_scope.culture_current_fervor
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			60 = 0
			1 = national_awakening.2
			1 = national_awakening.3
			1 = national_awakening.4
			1 = national_awakening.5
			1 = national_awakening.6
			1 = national_awakening.7
		}
	}

	complete = {
		is_subject = no
		NOT = {
			has_law_or_variant = law_type:law_subjecthood
		}
		scope:primary_culture_scope ?= { culture_current_fervor >= 75 }
	}

	on_complete = {
		trigger_event = {
			id = national_awakening.1
		}
	}

	event_outcome_completed_desc = {
		first_valid = {
			triggered_desc = {
				desc = event_effects_national_awakening.1_tt
			}
		}
	}

	fail = {
		trigger_if = {
			limit = {
				OR = {
					country_has_primary_culture = cu:slovene
					country_has_primary_culture = cu:croat
					country_has_primary_culture = cu:serb
					country_has_primary_culture = cu:bosniak
					country_has_primary_culture = cu:bulgarian
				}
			}
			custom_tooltip = {
				text = has_formed_yugoslavia_tt
				c:YUG ?= ROOT
			}
		}
	}

	current_value = {
		value = var:fervour_bar_var
	}

	goal_add_value = {
		add = 100
	}

	progressbar = yes

	weight = 100

	should_be_pinned_by_default = yes
}
