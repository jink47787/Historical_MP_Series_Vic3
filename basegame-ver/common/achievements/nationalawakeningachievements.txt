achievement_david_slays_goliath = {
	possible = {
		c:MON ?= this
	}
	happened = {
		custom_tooltip = {
			text = achievement_david_slays_goliath_2_tt
			OR = {
				c:MON ?= this
				has_variable = serbian_montenegro
				AND = {
					c:YUG ?= this
					OR = {
						was_formed_from = SER
						was_formed_from = MON
					}
				}
			}
		}
		custom_tooltip = {
			text = achievement_david_slays_goliath_tt
			NOT = {
				exists = c:TUR
			}
		}
	}
}

achievement_prussia_of_the_balkans = {
	possible = {
		has_dlc_feature = ip3_content
	}
	happened = {
		custom_tooltip = {
			text = achievement_prussia_of_the_balkans_3_tt
			country_has_primary_culture = cu:bulgarian
		}
		OR = {
			NOR = {
				exists = c:PRU
				exists = c:GER
			}
			OR = {
				c:PRU ?= {
					custom_tooltip = {
						text = achievement_prussia_of_the_balkans_2_tt
						country_rank < ROOT.country_rank
					}
				}
				c:GER ?= {
					custom_tooltip = {
						text = achievement_prussia_of_the_balkans_2_tt
						country_rank < ROOT.country_rank
					}
				}
			}
		}
		custom_tooltip = {
			text = achievement_prussia_of_the_balkans_tt
			has_modifier = modifier_prussia_of_the_balkans
		}
	}
}

achievement_sorbia_is_serbia = {

			OR = {
				c:SER ?= this
				has_variable = serbian_montenegro
				AND = {
					c:YUG ?= this
					OR = {
						was_formed_from = SER
						was_formed_from = MON
					}
				}
			}
		}
		custom_tooltip = {
			text = achievement_sorbia_is_serbia_2_tt
			owns_entire_state_region = STATE_SAXONY
		}
	}
}
