# possible - filters out achivements on game start so we dont have to check everything all the time
# happened - checks if achivement happened

achievement_mikasa_es_su_casa = {
	possible = {
		has_dlc_feature = ep2_content
		country_has_primary_culture = cu:japanese
	}
	happened = {
		country_has_primary_culture = cu:japanese
		custom_tooltip = {
			text = mikasa_es_su_casa_tt
			any_scope_treaty = {
				any_scope_article = {
					has_type = ship_transfer
					target_country ?= {
						any_primary_culture = {
							has_discrimination_trait = language_hispanophone
						}
					}
					input_ship ?= { 
						is_ship_type = ship_type:ship_type_pre_dreadnought
					}
				}
			}
		}
	}
}

achievement_golden_spirit = {
	possible = {
		OR = {
			country_has_primary_culture = cu:ainu
			AND = {
				c:JAP ?= this
				has_dlc_feature = ep2_content
			}
		}
	}
	happened = {
		country_has_primary_culture = cu:ainu
		owns_entire_state_region = STATE_HOKKAIDO
		custom_tooltip = {
			text = four_gold_mines_in_hokkaido_tt
			s:STATE_HOKKAIDO ?= {
				any_scope_state = {
					owner = ROOT
					any_scope_building = {
						is_building_type = building_gold_mine
						level >= 4
					}
				}
			}
		}
	}
}
