commander_rank_1 = {
	rank_value = 1
	texture = "gfx/interface/icons/commander_rank_icons/commander_rank_1.dds"

	character_modifier = {
		character_prominence_add = 10
		character_max_offensive_battles_add = 2
	}

	general_modifier = {
		character_command_limit_add = 30
	}

	admiral_modifier = {
		character_command_limit_add = 20
		character_naval_mission_area_add = 2
	}

	country_modifier = {
		country_bureaucracy_add = -10
	}

	title = {
		first_valid = {
			triggered_desc = {
				desc = commander_rank_1_general
				trigger = {
					has_role_of_type = general
				}
			}
			triggered_desc = {
				desc = commander_rank_1_admiral
				trigger = {
					has_role_of_type = admiral
				}
			}
			desc = commander_rank_1_fallback
		}
	}
}
