# Triggered by on_state_owner_change
# Root = state (the state that changed owner)
# Removes "State of Anarchy" modifier from Morocco JE when state is conquered by another country
ip4_lands_of_anarchy_remove_on_state_lost = {
	trigger = {
		has_dlc_feature = ip4_content
		has_modifier = state_of_anarchy_scaled_mod
		state_region = {
			OR = {
				this = s:STATE_AL_RIF
				this = s:STATE_MARRAKECH
				this = s:STATE_INNER_MOROCCO
			}
		}
	}
	effect = {
		remove_modifier = state_of_anarchy_scaled_mod
	}
}

# Triggered by on_battle_won
#
# Root = attacker or defender country
# scope:enemy_country = enemy of root
# scope:battle = battle
# scope:attacker = attacking commander
# scope:defender = defending commander
# scope:state = location state
carlist_war_battle_score = {
	trigger = {
		is_involved_in_journal_entry = je_the_first_carlist_war
		has_global_variable = carlist_war_skirmish_phase
		scope:enemy_country ?= {
			is_involved_in_journal_entry = je_the_first_carlist_war
		}
	}

	effect = {
		if = {
			limit = {
				c:SPC ?= this
			}
			change_global_variable = {
				name = spc_carlist_war_score
				add = 1
			}
			je:je_the_first_carlist_war ?= {
				add_progress = { value = -1 name = je_the_first_carlist_war_score_bar }
			}
		}
		else_if = {
			limit = {
				c:SPA ?= this
			}
			change_global_variable = {
				name = spa_carlist_war_score
				add = 1
			}
			je:je_the_first_carlist_war ?= {
				add_progress = { value = 1 name = je_the_first_carlist_war_score_bar }
			}
		}
	}
}
