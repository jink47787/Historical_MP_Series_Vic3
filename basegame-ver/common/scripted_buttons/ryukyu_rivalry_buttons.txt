je_ryukyu_rivalry_reaffirm_suzerainty_button = {
	name = "je_ryukyu_rivalry_reaffirm_suzerainty_button"
	desc = "je_ryukyu_rivalry_reaffirm_suzerainty_button_desc"

	visible = {
		OR = {
			c:JAP ?= this
			c:CHI ?= this
		}
	}

	possible = {
		custom_tooltip = {
			text = ryukyu_rivalry_reaffirm_suzerainty_button_tt
			OR = {
				AND = {
					scope:journal_entry ?= {
						"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" < 0
					}
					c:JAP ?= this
				}
				AND = {
					scope:journal_entry ?= {
						"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" > 0
					}
					c:CHI ?= this
				}
			}
		}
		c:RYU = {
			NOT = {
				has_attitude = {
					who = ROOT
					attitude = wary
				}
			}
		}
		hidden_trigger = {
			NOT = { has_variable = ryukyu_rivalry_reaffirm_suzerainty }
		}
	}

	selected = {
		has_variable = ryukyu_rivalry_reaffirm_suzerainty
	}

	effect = {
		if = {
			limit = {
				c:JAP ?= this
			}
			add_modifier = {
				name = modifier_ryukyu_rivalry_overreach
				multiplier = {
					value = 1
					if = {
                        limit = {
                            scope:journal_entry ?= {
                                "scripted_bar_progress(je_ryukyu_rivalry_score_bar)" <= -5
                            }
                        }
                        multiply = 2
                    }
					if = {
                        limit = {
                            scope:journal_entry ?= {
                                "scripted_bar_progress(je_ryukyu_rivalry_score_bar)" <= -10
                            }
                        }
                        multiply = 3
                    }
				}
				days = normal_modifier_time
				is_decaying = yes
			}
			set_global_variable = {
				name = ryukyu_rivalry_reaffirm_suzerainty_jap
			}
			hidden_effect = { c:CHI ?= { trigger_event = { id = ryukyu_rivalry.1 days = 3 popup = yes } } }
		}
		else = {
			add_modifier = {
				name = modifier_ryukyu_rivalry_overreach
				multiplier = {
					value = 1
					if = {
                        limit = {
                            scope:journal_entry ?= {
                                "scripted_bar_progress(je_ryukyu_rivalry_score_bar)" >= 5
                            }
                        }
                        multiply = 2
                    }
					if = {
                        limit = {
                            scope:journal_entry ?= {
                                "scripted_bar_progress(je_ryukyu_rivalry_score_bar)" >= 10
                            }
                        }
                        multiply = 3
                    }
				}
				days = normal_modifier_time
				is_decaying = yes
			}
			set_global_variable = {
				name = ryukyu_rivalry_reaffirm_suzerainty_chi
			}
			hidden_effect = { c:JAP ?= { trigger_event = { id = ryukyu_rivalry.1 days = 3 popup = yes } } }
		}
		ig:ig_landowners = {
			add_modifier = {
				name = modifier_ryukyu_rivalry_landowner_discontent
				days = normal_modifier_time
			}
		}
		custom_tooltip = ryukyu_rivalry_large_influence_gain_self_tt

		set_variable = ryukyu_rivalry_reaffirm_suzerainty
	}

	ai_chance = {
		value = 0

		if = {
			limit = {
				AND = {
					scope:journal_entry = {
						"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" >= 5
					}
					c:CHI ?= this
				}
			}
			add = 10
		}
		if = {
			limit = {
				AND = {
					scope:journal_entry = {
						"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" >= 10
					}
					c:CHI ?= this
				}
			}
			add = 10
		}
		if = {
			limit = {
				AND = {
					scope:journal_entry = {
						"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" >= 15
					}
					c:CHI ?= this
				}
			}
			add = 10
		}
		if = {
			limit = {
				AND = {
					scope:journal_entry = {
						"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" <= -5
					}
					c:CHI ?= this
				}
			}
			add = 10
		}
		if = {
			limit = {
				AND = {
					scope:journal_entry = {
						"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" <= -10
					}
					c:JAP ?= this
				}
			}
			add = 10
		}
		if = {
			limit = {
				AND = {
					scope:journal_entry = {
						"scripted_bar_progress(je_ryukyu_rivalry_score_bar)" <= -15
					}
					c:JAP ?= this
				}
			}
			add = 10
		}
	}
}

    name = "je_ryukyu_rivalry_inspect_defenses_button"
    desc = "je_ryukyu_rivalry_inspect_defenses_button_desc"

    visible = {
        OR = {
            c:JAP ?= this
            c:CHI ?= this
        }
    }

    possible = {
		OR = {
			custom_tooltip = {
				text = ryukyu_rivalry_inspect_defenses_button_tt
				any_scope_treaty = {
					any_scope_article = {
						has_type = military_access
						target_country = ROOT
						source_country = c:RYU
					}
				}
			}
			c:RYU ?= {
				any_scope_state = {
					is_capital = yes
					any_scope_building = {
						is_building_type = building_shipyard
						is_under_construction = no
						levels_owned_by_country = {
							target = ROOT
							value >= 1
						}
					}
				}
			}	
		}
		c:RYU = {
			any_scope_state = {
				is_capital = yes
				any_scope_building = {
					is_building_type = building_naval_fortification
					is_under_construction = no
				}
			}
		}
		any_scope_general = {
			is_ruler_of_own_country = no
			is_heir_of_own_country = no
		}
		NOT = { has_modifier = modifier_ryukyu_rivalry_officer_training }
    }

    selected = {
		has_modifier = modifier_ryukyu_rivalry_officer_training
    }

    effect = {
		custom_tooltip = {
			text = ryukyu_rivalry_send_general_tt
			random_scope_general = {
				limit = {
					is_ruler_of_own_country = no
					is_heir_of_own_country = no
				}
				save_scope_as = ryukyu_general_scope
				transfer_character = c:RYU
			}
		}
		if = {
			limit = {
				c:JAP ?= this
			}
			c:JAP = {
				save_scope_as = from_scope
			}
			hidden_effect = { c:RYU ?= { trigger_event = { id = ryukyu_rivalry.4 days = 1 popup = yes } } }
			hidden_effect = { c:CHI ?= { trigger_event = { id = ryukyu_rivalry.3 days = 1 popup = yes } } }
		}
		else = {
			c:CHI = {
				save_scope_as = from_scope
			}
			hidden_effect = { c:RYU ?= { trigger_event = { id = ryukyu_rivalry.4 days = 1 popup = yes } } }
			hidden_effect = { c:JAP ?= { trigger_event = { id = ryukyu_rivalry.3 days = 1 popup = yes } } }
		}

        add_modifier = {
            name = modifier_ryukyu_rivalry_officer_training
            days = normal_modifier_time
        }
		custom_tooltip = ryukyu_rivalry_medium_influence_gain_self_tt
		custom_tooltip = ryukyu_rivalry_risk_of_confrontation
    }

    ai_chance = {
        value = 0
		
		if = {
			limit = {
				AND = {
					c:CHI.naval_power_projection < c:JAP.naval_power_projection
					c:JAP ?= this
				}
			}
			add = 10
		}
		if = {
			limit = {
				AND = {
					c:CHI.naval_power_projection > c:JAP.naval_power_projection
					c:CHI ?= this
				}
			}
			add = 10
		}
    }
}
