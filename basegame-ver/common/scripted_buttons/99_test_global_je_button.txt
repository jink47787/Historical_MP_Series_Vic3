je_global_test_button = {
	name = "annex_a_lippe"
	desc = "annex_a_lippe_desc"
	
	cooldown = { years = 1 }
	
	visible = {
		NOR = {
			country_definition = cd:LIP
			country_definition = cd:SCM
		}	
	}
	
	possible = {
		OR = {
			c:SCM ?= { relations:root >= relations_threshold:amicable }
			c:LIP ?= { relations:root >= relations_threshold:amicable }			
		}
	}

	effect = {
		if = {
			limit = {
				c:SCM ?= { relations:root >= relations_threshold:amicable }
				c:LIP ?= { relations:root >= relations_threshold:amicable }			
			}
			
			random_list = {
				50 = { annex = c:LIP }
				50 = { annex = c:SCM }
			}			
		}	
		else_if = {
			limit = {
				c:SCM ?= { relations:root >= relations_threshold:amicable }
			}
			
			annex = c:SCM	
		}
		else_if = {
			limit = {
				c:LIP ?= { relations:root >= relations_threshold:amicable }			
			}
			
			annex = c:LIP	
		}	
		else = {
			custom_tooltip = annex_a_lippe_tt
		}
	}

	ai_chance = {
        value = 100
    }
}

je_global_test_button_2 = {
	name = "pay_lippe_service"
	desc = "pay_lippe_service_desc"
	
	cooldown = { years = 1 }
	
	visible = {
		NOR = {
			country_definition = cd:LIP
			country_definition = cd:SCM
		}		
	}
	
	possible = {
		infamy < 50
	}

	effect = {
		change_infamy = 15
		c:LIP ?= { change_relations = { country = root value = 30 } }
		c:SCM ?= { change_relations = { country = root value = 30 } }
	}

	ai_chance = {
        value = 100
    }
}
