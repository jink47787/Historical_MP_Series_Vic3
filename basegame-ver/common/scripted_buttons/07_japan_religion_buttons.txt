je_taikyo_proclamation_button = {
	name = "je_taikyo_proclamation_button"
	desc = "je_taikyo_proclamation_button_desc"

	visible = {
		has_journal_entry = je_shinbutsu_bunri
	}

	possible = {
		religion_percent_country = {
			target = rel:shinto
			value >= 0.2
		}
		ig:ig_devout = { is_powerful = no }
		custom_tooltip = {
			text = not_already_used_this_interaction_tt
			NOT = {
				has_variable = je_taikyo_proclamation_var
			}
		}
	}

	effect = {
		set_state_religion = rel:shinto
		state_religion_switch_effect = yes
		activate_law = law_type:law_state_religion
		ig:ig_devout ?= {
			hidden_effect = { set_interest_group_name = ig_shinto_monks }
			remove_ideology = ideology_buddhist_moralist
			add_ideology = ideology_shinto_moralist
			set_ig_trait = ig_trait:ig_trait_haibutsu_kishaku
			set_ig_trait = ig_trait:ig_trait_secular_shrine_theory
			set_ig_trait = ig_trait:ig_trait_heavenly_sovereign
			leader ?= {
				change_character_religion = rel:shinto
			}
			if = {
				limit = { has_modifier = forced_monk_conversions }
				remove_modifier = forced_monk_conversions
			}
			if = {
				limit = { has_modifier = monks_stripped_of_titles_opinion }
				remove_modifier = monks_stripped_of_titles_opinion
			}
			if = {
				limit = { has_modifier = monks_left_to_duties }
				remove_modifier = monks_left_to_duties
			}
			if = {
				limit = { has_modifier = banned_syncretic_terminology }
				remove_modifier = banned_syncretic_terminology
			}
			if = {
				limit = { has_modifier = retained_syncretic_terminology }
				remove_modifier = retained_syncretic_terminology
			}
			if = {
				limit = { has_modifier = smelted_temple_bells_opinion }
				remove_modifier = smelted_temple_bells_opinion
			}
			if = {
				limit = { has_modifier = removed_buddha_statues_opinion }
				remove_modifier = removed_buddha_statues_opinion
			}
			if = {
				limit = { has_modifier = retained_buddhist_iconography }
				remove_modifier = retained_buddhist_iconography
			}
		}

		every_interest_group = {
			limit = {
				NOT = { is_interest_group_type = ig_devout }
			}
			leader ?= {
				if = {
					limit = { has_religion = mahayana }
					random = {
						chance = 75
						change_character_religion = rel:shinto
					}
				}
			}
		}
		set_variable = je_taikyo_proclamation_var
	}

	ai_chance = {
		value = 50
	}
}
