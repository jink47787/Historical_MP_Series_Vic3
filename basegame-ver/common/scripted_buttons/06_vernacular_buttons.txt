### Spanish Buttons

je_vernacular_policy_accept_catalan_button = {
	name = je_vernacular_policy_accept_catalan_button
	desc = je_vernacular_policy_accept_catalan_button_desc

	visible = {
		any_scope_state = {
			state_region = {
				is_homeland = cu:catalan
			}
		}
		NOT = {
			country_has_primary_culture = cu:catalan
		}
		has_variable = chose_integration_var
		OR = {
			OR = {
				c:SPA ?= this
				was_formed_from = SPA
			}
			OR = {
				c:FRA ?= this
				was_formed_from = FRA
			}
		}
		culture_percent_country = {
			target = cu:catalan
			value > 0
		}
	}

	possible = {
		country_average_culture_pop_acceptance = {
			culture = cu:catalan
			value >= acceptance_status_5
		}
		OR = {
			any_scope_state = {
				state_region = {
					is_homeland = cu:catalan
				}
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 10
				}
				any_scope_building = {
					OR = {
						is_building_type = building_university
						is_building_type = building_art_academy
					}
					level >= 3
				}
				has_decree = decree_encourage_manufacturing_industry
			}
			any_scope_state = {
				state_region = {
					is_homeland = cu:catalan
				}
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 5
				}
				any_scope_building = {
					OR = {
						is_building_type = building_university
						is_building_type = building_art_academy
					}
				}
				has_decree = decree_promote_social_mobility
				count >= 2
			}
		}
	}

	selected = {
	}

	effect = {
		add_primary_culture = cu:catalan
		add_to_variable_list = { name = spain_accepted_cultures target = cu:catalan }
	}

	ai_chance = {
		value = 1
	}
}
