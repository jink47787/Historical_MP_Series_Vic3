je_iwakura_mission_button = {
	name = "iwakura_mission_decision"
	desc = "iwakura_mission_decision_desc"

	visible = {
		has_dlc_feature = ep2_content
	}

	possible = {
		NOR = {
			has_law_or_variant = law_type:law_isolationism
			has_law_or_variant = law_type:law_closed_borders
		}
		any_scope_character = {
			NOR = {
				has_role_of_type = ruler
				has_role_of_type = heir
				has_role_of_type = agitator
				custom_tooltip = {
					text = house_yamato_not_tt # Please stop trying to send the Emperor to Sweden
					has_variable = house_yamato
				}
			}
			interest_group ?= {
				is_in_government = yes
			}
		}
		hidden_trigger = {
			NOT = { has_variable = on_expedition }
			NOT = { has_variable = iwakura_mission_started }
		}
	}

	effect = {
		trigger_event = { 
			id = iwakura_mission.1
			popup = yes
		}
		custom_tooltip = adds_mission_to_the_west_je_tt
		set_variable = on_expedition
		set_variable = iwakura_mission_started
	}
}

je_end_edo_system_button = {
	name = "je_end_edo_system_button_name"
	desc = "je_end_edo_system_button_desc"

	visible = {
		has_social_hierarchy = edo_system
	}

	possible = {
		has_law_or_variant = law_type:law_lax_edo_system
		legitimacy >= 50
		# No one in government would prefer a stricter enforcement of the Edo System
		NOT = {
			any_interest_group = {
				is_in_government = yes
				custom_tooltip = {
					text = edo_system_not_support_enforcement_tt
					NOR = {
		                law_enactment_stance = {
		                    law = law_type:law_intermediate_edo_system
		                    value <= neutral
		                }
		                law_enactment_stance = {
		                    law = law_type:law_strict_edo_system
		                    value <= neutral
		                }
					}
				}
			}
		}
	}

	effect = {
		custom_tooltip = {
			text = edo_system_ends_tt
			trigger_event = { id = set_hierarchy_event.3 }
		}
	}
}
