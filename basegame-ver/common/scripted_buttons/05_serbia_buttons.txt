# Slander button
je_tot_slander_button = {
	name = "je_tot_slander_button"
	desc = "je_tot_slander_button_desc"

	ai_chance = {
		value = 0
	}

	visible = {

	}

	possible = {
		has_building = building_paper_mills
		has_technology_researched = mass_communication
		scope:journal_entry = {
			not = {
				custom_tooltip = {
					text = je_tot_slander_button_cooldown_tt
					has_variable = je_tot_slander_button_cooldown
				}
			}
		}
	}

	effect = {
		scope:journal_entry = {
			set_variable = { name = je_tot_slander_button_cooldown months = 48 }
		}
		trigger_event = {
			id = serbian_throne.2
		}
		show_as_tooltip = {
			ser_save_contender_of_rival_dynasty_as_scope = yes
			scope:rival_dynasty_contender_scope ?= {
				add_modifier = {
					name = serbia_karadordevic_slandered_mod
					days = short_modifier_time
				}
			}
		}
		custom_tooltip = je_tot_slander_button_custom_tt
     	show_as_tooltip = {
     		add_modifier = {
				name = serbia_pamphlet_money_mod
				days = short_modifier_time
				multiplier = money_amount_multiplier_medium
     		}
     		add_modifier = {
				name = serbia_pamphlet_auth_mod
				days = short_modifier_time
     		}
     	}
	}
}

# Campaign button
je_tot_campaign_button = {
	name = "je_tot_campaign_button"
	desc = "je_tot_campaign_button_desc"

	ai_chance = {
		value = 0
	}

	visible = {
		always = yes
	}

	possible = {
		custom_tooltip = {
			text = each_monarch_may_only_do_this_once_tt
			NOT = {
				has_variable = campaign_button_pressed
			}
		}
		custom_tooltip = {
			text = ser_campaign_not_on_cooldown_tt
			NOT = {
				has_variable = ser_campaign_cooldown
			}
		}
		ruler.interest_group = {
			ig_approval >= 5
		}
	}

	effect = {
		custom_tooltip = {
			text = je_tot_campaign_button_tt
			trigger_event = {
				id = serbian_throne.12
			}
		}
		show_as_tooltip = {
			add_modifier = {
				name = serbia_campaign_auth_fund_mod
			}
			add_modifier = {
				name = serbia_campaign_med_fund_mod
				multiplier = money_amount_multiplier_small
			}
			add_modifier = {
				name = serbia_campaign_high_fund_mod
				multiplier = money_amount_multiplier_medium
			}
		}
		custom_tooltip = {
			text = je_tot_campaign_button_tt_2
		}
		set_variable = campaign_button_pressed
	}
}
