button_je_taming_the_north_agriculture_arable_land = {

	}

		var:hokkaido_agriculture_arable_counter_var < 3
	}

	possible = {
		custom_tooltip = {
			text = established_sapporo_college_tt
			has_variable = established_sapporo_college_var
		}
		custom_tooltip = {
			text = hokkaido_agriculture_arable_tt
			var:hokkaido_agriculture_arable_counter_var < 3
		}
		trigger_if = {
			limit = {
				s:STATE_HOKKAIDO.region_state:JAP = {
					has_state_trait = state_trait_ishikari_wetlands_1
				}
			}
			has_technology_researched = intensive_agriculture
			has_technology_researched = atmospheric_engine
		}
		trigger_if = {
			limit = {
				s:STATE_HOKKAIDO.region_state:JAP = {
					has_state_trait = state_trait_ishikari_wetlands_2
				}
			}
			has_technology_researched = improved_fertilizer
			has_technology_researched = rotary_valve_engine
		}
		trigger_if = {
			limit = {
				s:STATE_HOKKAIDO.region_state:JAP = {
					has_state_trait = state_trait_ishikari_plain_1
				}
			}
			has_technology_researched = mechanized_farming
			has_technology_researched = combustion_engine
		}
	}

	selected = {

	}

	effect = {
		change_variable = {
			name = hokkaido_agriculture_arable_counter_var
			add = 1
		}
		trigger_event = {
			id = hokkaido_events.4
			popup = yes	
		}
		show_as_tooltip = {
			s:STATE_HOKKAIDO.region_state:JAP = {
				if = { # Drain the swamp
					limit = {
						has_state_trait = state_trait_ishikari_wetlands_1
					}
					state_region = {
						remove_state_trait = state_trait_ishikari_wetlands_1
						add_state_trait =  state_trait_ishikari_wetlands_2
					}
				}
				else_if = {
					limit = {
						has_state_trait = state_trait_ishikari_wetlands_2
					}
					state_region = {
						remove_state_trait = state_trait_ishikari_wetlands_2
						add_state_trait =  state_trait_ishikari_plain_1
					}
				}
				else_if = {
					limit = {
						has_state_trait = state_trait_ishikari_plain_1
					}
					state_region = {
						remove_state_trait = state_trait_ishikari_plain_1
						add_state_trait =  state_trait_ishikari_plain_2
					}
				}
				state_region = {
					add_arable_land = 20
				}
			}
		}
	}

	ai_chance = {
		value = 10
	}
}
