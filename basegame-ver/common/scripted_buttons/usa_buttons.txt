interest_in_hawaii_button = { # je_hawaiian_interest
	name = "interest_in_hawaii"
	desc = "interest_in_hawaii_desc"

	visible = {
		always = yes
	}

	possible = {
		je:je_hawaiian_interest = {
			NOT = {
				has_modifier = influence_over_hawaii
			}
		}
	}

	selected = {
		je:je_hawaiian_interest = {
			has_modifier = influence_over_hawaii
		}
	}

	effect = {
		set_variable = hawaii_interest
		je:je_hawaiian_interest = {
			add_modifier = {
				name = influence_over_hawaii
				days = very_long_modifier_time
			}
		}
	}

	ai_chance = {
        value = 0

		if = {
			limit = {
				country_is_in_north_america = yes
			}
			add = 2

			if = {
				limit = {
					c:USA ?= this
					sr:region_pacific_coast = {
						any_scope_state = {
							owner = c:USA
						}
					}
				}
				add = 10
			}
		}

		if = { # Hawaii would be a great place for some sort of hypothetical missile related confrontation, a crisis if you will!
			limit = {
				country_is_in_east_asia = yes
			}
			add = 1

			if = {
				limit = {
					c:JAP ?= this
					has_variable = meiji_restoration_complete
				}
				add = 10
			}
		}

		if = {
			limit = {
				capital ?= {
					region = sr:region_oceania
				}
			}
			add = 1
		}

		if = {
			limit = {
				country_is_in_south_america = yes
			}
			add = 1
		}

		if = {
			limit = {
				country_is_in_europe = yes
			}
			add = 1
		}

		if = {
			limit = {
				has_strategy = ai_strategy_armed_isolationism
			}
			multiply = 0
		}

		if = {
			limit = {
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}
			multiply = 0
		}
    }
}
