# A Diplomatic Scramble
ryukyu_rivalry.1 = {
    type = country_event
    placement = ROOT

    duration = 3

	title = ryukyu_rivalry.1.t
	desc = ryukyu_rivalry.1.d
	flavor = ryukyu_rivalry.1.f

    event_image = {
		video = "unspecific_signed_contract"
	}

	dlc = dlc018

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

    trigger = {
		# Triggered by JE when rival uses je_ryukyu_rivalry_reaffirm_suzerainty_button
	}

    immediate = {
		if = {
			limit = {
				c:JAP ?= this
			}
			c:CHI = {
				save_scope_as = rival_scope
			}
		}
		else = {
			c:JAP = {
				save_scope_as = rival_scope
			}
		}
		c:RYU = {
			save_scope_as = ryukyu_scope
		}
	}

    option = { # Formal Protest
        name = ryukyu_rivalry.1.a
        default_option = yes

        change_relations = {  
			country = scope:rival_scope
			value = -50  
		}
    }

	option = { # Diplomatic Push
        name = ryukyu_rivalry.1.b
		trigger = {
			OR = {
				custom_tooltip = {
					text = ryukyu_rivalry.1.tt1
					c:RYU = {
						is_subject_of = ROOT
						liberty_desire <= 25
					}
				}
				custom_tooltip = {
					text = ryukyu_rivalry.1.tt2
					c:RYU = {
						NOR = {
							is_subject_of = ROOT
							relations:root < relations_threshold:amicable
						}
					}
				}
			}
		}
		highlighted_option = yes

		change_relations = {  
			country = scope:ryukyu_scope
			value = 50  
		}

		ig:ig_landowners = {
			add_modifier = {
				name = modifier_ryukyu_rivalry_diplomatic_appeasement
				days = short_modifier_time
			}
		}
    }

	option = { # Economic Blackmail
        name = ryukyu_rivalry.1.c
		show_as_unavailable = {
			c:RYU.capital = {
				any_scope_building = {
					#is_building_type = building_trade_center
					levels_owned_by_country = {
						target = ROOT
						value  < 3
					}
				}
			}
		}
		trigger = {
			custom_tooltip = {
				text = ryukyu_rivalry.1.tt3
				c:RYU.capital = {
					any_scope_building = {
						levels_owned_by_country = {
							target = ROOT
							value  >= 3
						}
					}
				}
			}
		}
		highlighted_option = yes

		change_relations = {  
			country = scope:ryukyu_scope
			value = -25  
		}
		custom_tooltip = {
			text = ryukyu_rivalry_small_influence_gain_self_tt
			if = {
				limit = {
					c:JAP = this
				}
				je:je_ryukyu_rivalry = {
					set_bar_progress = { value = 2 name = je_ryukyu_rivalry_score_bar }
				}
			}
			else = {
				je:je_ryukyu_rivalry = {
					set_bar_progress = { value = -2 name = je_ryukyu_rivalry_score_bar }
				}
			}
		}

    }
	option = { # Do nothing
        name = ryukyu_rivalry.1.e

		ig:ig_armed_forces = {
			add_modifier = {
				name = modifier_ryukyu_rivalry_armed_forces_discontent
				days = short_modifier_time
			}
		}
    }
}

# Final Event
ryukyu_rivalry.5 = {
	type = country_event
    placement = ROOT

    duration = 3

	title = ryukyu_rivalry.5.t

	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {	scope:loser_scope = this }
				desc = ryukyu_rivalry.5.d_loser
			}
			triggered_desc = {
				trigger = { scope:victor_scope = this }
				desc = ryukyu_rivalry.5.d_victor
			}
			triggered_desc = {
				trigger = { c:RYU = this }
				desc = ryukyu_rivalry.5.d_ryu
			}
		}
	}

	flavor = {
		first_valid = {
			# If Ryukyu was annexed
			triggered_desc = {
				trigger = {
					NOT = { exists = c:RYU }
				}
				desc = ryukyu_rivalry.5.f_annexed
			}
			# If Ryukyu became subject
			triggered_desc = {
				trigger = {
					c:RYU = {
						is_subject_of = c:JAP
					}
				}
				desc = ryukyu_rivalry.5.f_subject
			}
		}
	}

	event_image = {
		video = "ep1_transfer_of_authority"
	}

	dlc = dlc018

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	trigger = {
		# Triggered by JE when Japan or China wins
	}

	immediate = {
		if = {
			limit = {
				scope:victor_scope = this
			}
			add_modifier = {
				name = modifier_ryukyu_rivalry_succeeded_ryukyu
				days = long_modifier_time
			}
			if = { #my subject
				limit = {
					any_subject_or_below = {
						c:RYU ?= this
					}
				}
				annex = c:RYU
				if = {
					limit = {
						c:JAP = this
					}
					ig:ig_armed_forces = {
						add_modifier = {
							name = modifier_ryukyu_rivalry_southern_secured
							days = normal_modifier_time
						}
					}
				}
			}
			else_if = { #opponent subject
				limit = {
					c:RYU = {
						is_subject = yes
					}
				}
				transfer_subject = c:RYU
				if = {
					limit = {
						c:CHI = this
					}
					ig:ig_intelligentsia = {
						add_modifier = {
							name = modifier_ryukyu_rivalry_scholars_pleased
							days = normal_modifier_time
						}
					}
				}
			}
			else = { 
				create_diplomatic_pact = {
					country = c:RYU
					type = puppet
				}
				if = {
					limit = {
						c:CHI = this
					}
					ig:ig_intelligentsia = {
						add_modifier = {
							name = modifier_ryukyu_rivalry_scholars_pleased
							days = normal_modifier_time
						}
					}
				}
			}
		}
		if = {
			limit = {
				scope:loser_scope = this
			}
			scope:loser_scope = {
				add_modifier = {
					name = modifier_ryukyu_rivalry_failed_ryukyu
					days = long_modifier_time
				}
			}
		}
	}

	option = {
		name = ryukyu_rivalry.5.a

		show_as_tooltip = {
			scope:victor_scope = {
				add_modifier = {
					name = modifier_ryukyu_rivalry_succeeded_ryukyu
					days = long_modifier_time
				}
				if = {
					limit = {
						any_subject_or_below = {
							c:RYU ?= this
						}
					}
					custom_tooltip = ryukyu_rivalry_ryukyu_vassalise_tt
					if = {
						limit = {
							c:CHI = this
						}
						ig:ig_intelligentsia = {
							add_modifier = {
								name = modifier_ryukyu_rivalry_scholars_pleased
								days = normal_modifier_time
							}
						}
					}
				}
				else_if = {
					limit = {
						NOT = {
							exists = c:RYU
						}
					}
					custom_tooltip = ryukyu_rivalry_ryukyu_annex_tt
					if = {
						limit = {
							c:JAP = this
						}
						ig:ig_armed_forces = {
							add_modifier = {
								name = modifier_ryukyu_rivalry_southern_secured
								days = normal_modifier_time
							}
						}
					}
				}
				else = {
					custom_tooltip = ryukyu_rivalry_ryukyu_vassalise_tt
					if = {
						limit = {
							c:CHI = this
						}
						ig:ig_intelligentsia = {
							add_modifier = {
								name = modifier_ryukyu_rivalry_scholars_pleased
								days = normal_modifier_time
							}
						}
					}
				}
			}
		}

		show_as_tooltip = {
			scope:loser_scope = {
				add_modifier = {
					name = modifier_ryukyu_rivalry_failed_ryukyu
					days = long_modifier_time
				}
			}
		}
	}
}

ryukyu_rivalry.7 = {
    type = country_event
    placement = ROOT

    duration = 3

	title = ryukyu_rivalry.7.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { c:RYU = this }
				desc = ryukyu_rivalry.7.d_ryukyu
			}
			desc = ryukyu_rivalry.7.d
		}
	}
	flavor = ryukyu_rivalry.7.f

    event_image = {
		video = "ep2_important_japanese_men_negotiating"
	}

	dlc = dlc018

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

    trigger = {
		# Triggered by JE Progress bar reaches 100%
	}

    immediate = {
		if = {
			limit = {
				c:RYU = this
			}
			make_independent = yes
			every_scope_treaty ?= {
				limit = {
					OR = {
						binds = c:CHI
						binds = c:JAP
					}
					any_scope_article = {
						has_type = money_transfer
					}
				}
				withdraw = {
					country = root
				}
			}
		}
		c:RYU = {
			save_scope_as = ryukyu_scope
		}
		set_variable = ryukyu_won_rivalry_je
	}

    option = { # The play is at an end
        name = ryukyu_rivalry.7.a
        default_option = yes
		trigger = {
			c:RYU = this
		}

		if = {
			limit = {
				c:RYU = this
			}
			add_modifier = {
				name = modifier_ryukyu_rivalry_secured_neutrality
				days = stupidly_long_modifier_time
			}
		}
		else = {
			show_as_tooltip = {
				c:RYU = {
					add_modifier = {
						name = modifier_ryukyu_rivalry_secured_neutrality
						days = stupidly_long_modifier_time
					}
				}
			}
		}
    }
	option = { # The play is at an end
        name = ryukyu_rivalry.7.b
        default_option = yes
		trigger = {
			NOT = {
				c:RYU = this
			}
		}

		show_as_tooltip = {
			c:RYU = {
				add_modifier = {
					name = modifier_ryukyu_rivalry_secured_neutrality
					days = stupidly_long_modifier_time
				}
			}
		}
		
    }
}
