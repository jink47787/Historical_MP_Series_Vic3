namespace = japan_monarchy

# Accession of Komei
japan_monarchy.1 = {
	type = country_event
	placement = root

	duration = 3

	title = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_law = law_type:law_bakufu
				}
				desc = japan_monarchy.1.t1
			}
			triggered_desc = {
				trigger = {
					NOT = {
						has_law = law_type:law_bakufu
					}
				}
				desc = japan_monarchy.1.t2
			}
		}
	}
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_law = law_type:law_bakufu
				}
				desc = japan_monarchy.1.d1
			}
			triggered_desc = {
				trigger = {
					NOT = {
						has_law = law_type:law_bakufu
					}
				}
				desc = japan_monarchy.1.d2
			}
		}
	}
	flavor = japan_monarchy.1.f


	gui_window = event_window_1char_tabloid

	left_icon = scope:emperor_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	dlc = dlc018

	icon = "gfx/interface/icons/faction_icons/faction_japan_imperial_house.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		has_dlc_feature = ep2_content
		# Triggered by on_new_ruler
	}

	cancellation_trigger = {
		NOT = { exists = scope:emperor_scope }
	}

	immediate = {
		if = {
			limit = {
				has_law = law_type:law_bakufu
			}
			create_character = {
				template = JAP_komei_yamato
				on_created = {
					ROOT = {
						set_variable = { name = emperor_var value = PREV }
					}
					add_career_length = {
						role = character_role_emperor_of_japan
						years = 75
						random_range = { 1.0 1.5 } # 75-100 years
					}
				}
				save_scope_as = emperor_scope
			}
		}
		else = {
			random_scope_character = { 
				limit = {
					has_template = JAP_komei_yamato
				}
				c:JAP ?= {
					set_variable = { name = emperor_var value = PREV }
				}
				save_scope_as = emperor_scope
			}
		}
		random_scope_character = {
			limit = {
				NOR = {
					owner.ruler ?= this
					has_variable = house_yamato
				}
				interest_group ?= {
					OR = {
						is_interest_group_type = ig_landowners
						is_interest_group_type = ig_armed_forces
						is_interest_group_type = ig_devout
					}
				}
			}
			save_scope_as = name_scope
		}
	}

	option = {
		name = japan_monarchy.1.a
		default_option = yes
		if = {
			limit = {
				has_variable = one_emperor_one_era
			}
			scope:emperor_scope ?= {
				set_first_name = TenkyU_
			}
		}
		else = {
			scope:emperor_scope ?= {
				set_first_name = Komei
			}
		}
		show_as_tooltip = {
			scope:emperor_scope ?= {
				add_trait = special_character_japanese_emperor
			}
		}
		if = {
			limit = {
				NOT = {
					has_law = law_type:law_bakufu
				}
			}
			show_as_tooltip = {
				scope:emperor_scope ?= {
					set_character_as_ruler = yes
				}
			}
		}
	}
}

japan_monarchy.5 = {
	type = country_event
	placement = root

	duration = 3

	title = japan_monarchy.5.t
	desc = japan_monarchy.5.d
	flavor = japan_monarchy.5.f

    event_image = {
		video = "ep2_intellectual_writer"
	}

	dlc = dlc018

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/faction_icons/faction_japan_imperial_house.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		has_dlc_feature = ep2_content
		country_has_monarchy_law = yes
		ruler = {
			has_variable = house_yamato
		}
	}

	cancellation_trigger = {

	}

	immediate = {
		ruler = {
			save_scope_as = emperor_scope
		}
	}

	option = {
		name = japan_monarchy.5.a
		default_option = yes
		set_variable = one_emperor_one_era
		add_loyalists = { value = small_radicals pop_type = aristocrats }
	}
}

# Death of the Retired Emperor
japan_monarchy.6 = {
	type = country_event
	placement = root

	duration = 3

	title = japan_monarchy.6.t
	desc = japan_monarchy.6.d
	flavor = japan_monarchy.6.f

    gui_window = event_window_1char_tabloid

	left_icon = scope:retired_emperor_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	dlc = dlc018

	icon = "gfx/interface/icons/faction_icons/faction_japan_imperial_house.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		has_dlc_feature = ep2_content
		game_date >= 1840.6.1
		c:JAP ?= this
	}

	cancellation_trigger = {

	}

	immediate = {
		create_character = {
			template = JAP_kokaku_yamato
			save_scope_as = retired_emperor_scope
		}
	}

	option = {
		name = japan_monarchy.6.a
		default_option = yes
		scope:retired_emperor_scope ?= {
			kill_character = yes
		}
	}
}
