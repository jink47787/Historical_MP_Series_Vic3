# Ansei Edo Earthquake
# Kantō
# Earthquake
japan_earthquakes.2 = {
	type = country_event
	placement = s:STATE_KANTO.region_state:JAP
	title = japan_earthquakes.2.t
	desc = japan_earthquakes.2.d
	flavor = japan_earthquakes.2.f
	
	duration = 3

	cooldown = { days = never_fire_again_modifier_time }

	event_image = {
		video = "gfx/event_pictures/unspecific_devastation.bk2"
	}

	dlc = dlc018

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		# Triggered by japan_earthquake_events effect in on_yearly_pulse_country
		c:JAP ?= this
		has_dlc_feature = ep2_content
		year = 1855
		has_state_in_state_region = STATE_KANTO
	}

	immediate = {
		s:STATE_KANTO ?= {
			start_harvest_condition = earthquake
			set_variable = {
				name = natural_disaster_cooldown
				days = normal_modifier_time
			}
			region_state:JAP ?= { save_scope_as = japan_earthquake_state }
		}
	}

	option = {
		name = japan_earthquakes.2.a
		default_option = yes
		s:STATE_KANTO ?= { add_devastation = 15 }
		ig:ig_landowners ?= {
			add_modifier = {
				name = earthquake_vassal_support
				days = normal_modifier_time
			}
		}
		ig:ig_armed_forces ?= {
			add_modifier = {
				name = earthquake_vassal_support
				days = normal_modifier_time
			}
		}
	}
	option = {
		name = japan_earthquakes.2.b
		s:STATE_KANTO.region_state:JAP ?= {
			add_devastation = 15
			add_modifier = {
				name = earthquake_rebuilding_focus
				days = normal_modifier_time
			}
		}
		ig:ig_trade_unions = {
			add_modifier = {
				name = earthquake_rebuilding_redistribution
				days = normal_modifier_time
			}
		}
	}
}

# Mino–Owari earthquake
# Tōkai (Kansei/Kyōto)
# Earthquake
japan_earthquakes.3 = {
	type = country_event
	placement = s:STATE_TOKAI.region_state:JAP
	title = japan_earthquakes.3.t
	desc = japan_earthquakes.3.d
	flavor = japan_earthquakes.3.f
	
	duration = 3

	cooldown = { days = never_fire_again_modifier_time }

	event_image = {
		video = "southamerica_war_civilians"
	}

	dlc = dlc018

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		# Triggered by japan_earthquake_events effect in on_yearly_pulse_country
		c:JAP ?= this
		has_dlc_feature = ep2_content
		year = 1891
		has_state_in_state_region = STATE_TOKAI
	}

	immediate = {
		s:STATE_TOKAI ?= {
			start_harvest_condition = earthquake
			set_variable = {
				name = natural_disaster_cooldown
				days = normal_modifier_time
			}
			region_state:JAP ?= { save_scope_as = japan_earthquake_state }
		}
	}

	option = {
		name = japan_earthquakes.3.a
		s:STATE_TOKAI ?= { add_devastation = 15 }
		c:GBR ?= {
			change_relations = {
				country = root
				value = 15
			}
		}
		c:USA ?= {
			change_relations = {
				country = root
				value = 15
			}
		}
	}
	option = {
		name = japan_earthquakes.3.b
		s:STATE_TOKAI ?= { add_devastation = 15 }
		add_modifier = {
			name = meteorological_observatories
		}
		add_modifier = {
			name = meteorological_observatories_cost
			days = normal_modifier_time
			multiplier = money_amount_multiplier_small
		}
	}
	option = {
		name = japan_earthquakes.3.c
		default_option = yes
		s:STATE_TOKAI.region_state:JAP ?= {
			add_devastation = 10
			add_modifier = {
				name = earthquake_bracing_for_aftershocks
				days = short_modifier_time
			}
		}
	}
}

# Sanriku Earthquake
# Tōhoku
# Tsunami
japan_earthquakes.4 = {
	type = country_event
	placement = s:STATE_TOHOKU.region_state:JAP
	title = japan_earthquakes.4.t
	desc = japan_earthquakes.4.d
	flavor = japan_earthquakes.4.f
	
	duration = 3

	cooldown = { days = never_fire_again_modifier_time }

	event_image = {
		video = "gfx/event_pictures/unspecific_devastation.bk2"
	}

	dlc = dlc018

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		# Triggered by japan_earthquake_events effect in on_yearly_pulse_country
		c:JAP ?= this
		has_dlc_feature = ep2_content
		year = 1896
		has_state_in_state_region = STATE_TOHOKU
	}

	immediate = {
		s:STATE_TOHOKU ?= {
			start_harvest_condition = tsunami
			set_variable = {
				name = natural_disaster_cooldown
				days = normal_modifier_time
			}
			region_state:JAP ?= { save_scope_as = japan_earthquake_state }
		}
	}

	option = {
		name = japan_earthquakes.4.a
		s:STATE_TOHOKU.region_state:JAP = {
			add_devastation = 20
			add_modifier = {
				name = tsunami_stones
			}
		}
	}
	option = {
		name = japan_earthquakes.4.b
		trigger = {
			any_scope_state = {
				state_region = {
					has_harvest_condition = tsunami
				}
				count > 1
			}
		}
		s:STATE_TOHOKU = { add_devastation = 20 }
		every_scope_state = {
			limit = {
				state_region = {
					has_harvest_condition = tsunami
				}
			}
			add_modifier = {
				name = tsunami_stones
			}
		}
	}
	option = {
		name = japan_earthquakes.4.c
		default_option = yes
		s:STATE_TOHOKU = { add_devastation = 10 }
		add_modifier = {
			name = tsunami_relief_efforts
			days = short_modifier_time
		}
	}
}

	}

		s:STATE_KANTO ?= { add_devastation = 30 }
		ig:ig_devout ?= {
			add_modifier = {
				name = earthquake_divine_punishment
				days = normal_modifier_time
			}
		}
	}
	option = {
		name = japan_earthquakes.5.c
		s:STATE_KANTO.region_state:JAP ?= { add_devastation = 30 }
		add_modifier = {
			name = earthquake_modernization
			days = normal_modifier_time
		}
		ig:ig_landowners ?= {
			add_modifier = {
				name = earthquake_modernization_land_confiscation
				days = normal_modifier_time
			}
		}
	}
}
