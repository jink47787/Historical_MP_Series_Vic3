zaibatsu.1 = {
	type = country_event
	placement = root

	duration = 3

	title = zaibatsu.1.t
	desc = zaibatsu.1.d
	flavor = zaibatsu.1.f

	event_image = {
		video = "ep2_japanese_cityscape"
	}

	dlc = dlc018

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	trigger = {
		# Triggered by je_zaibatsu on_complete
	}

	cancellation_trigger = {

	}

	immediate = {
		ordered_company = {
			limit = {
				company_is_zaibatsu = yes
			}
			order_by = company_owned_levels
			position = 0
			save_scope_as = relevant_zaibatsu
			executive = {
				save_scope_as = relevant_zaibatsu_exec
			}
		}
		ig:ig_industrialists ?= {
			save_scope_as = zaibatsu_ig
		}
		ordered_interest_group = {
			limit = {
				is_in_government = yes
			}
			order_by = ig_clout
			position = 0
			save_scope_as = ruling_ig
			leader = {
				save_scope_as = pm_scope
				set_character_immortal = yes
			}
		}
	}

	option = {
		name = zaibatsu.1.a
		default_option = yes
		active_law:lawgroup_economic_system ?= { # Construction efficiency
			add_amendment = {
				type = amendment_zaibatsu_oligopoly_1
				sponsor = PREV.ig:ig_industrialists
				cooldown = 240
			}
		}
		custom_tooltip = amendment_zaibatsu_oligopoly_1_tt
		capital = {
			add_modifier = {
				name = modifier_a_sea_of_light
			}
		}
	}

	option = {
		name = zaibatsu.1.b
		active_law:lawgroup_economic_system ?= {  # Charters
			add_amendment = {
				type = amendment_zaibatsu_oligopoly_2
				sponsor = PREV.ig:ig_industrialists
				cooldown = 240
			}
		}
		custom_tooltip = amendment_zaibatsu_oligopoly_2_tt
		ig:ig_armed_forces ?= {
			add_modifier = {
				name = modifier_military_industrial_supply_chain
				days = very_long_modifier_time
			}
		}
	}

	after = {
		hidden_effect = {
			scope:pm_scope ?= {
				set_character_immortal = no
			}
		}
	}
}
