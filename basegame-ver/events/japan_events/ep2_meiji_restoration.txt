ep2_meiji.1 = { # Sonnō Jōi
	type = country_event
	placement = root

	duration = 3

	title = ep2_meiji.1.t
	desc = ep2_meiji.1.d
	flavor = ep2_meiji.1.f

    event_image = {
		video = "asia_westerners_arriving"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	cooldown = { days = long_modifier_time }

	cancellation_trigger = {

	}

	immediate = {
		ordered_interest_group = {
			limit = {
				is_in_government = yes
			}
			order_by = ig_clout
			position = 0
			check_range_bounds = no
			save_scope_as = ruling_ig
			leader = {
				save_scope_as = tairo_scope
				set_character_immortal = yes
			}
		}
		hidden_effect = {
			set_strategy = ai_strategy_industrial_expansion
			set_strategy = ai_strategy_meiji_restoration
			set_strategy = ai_strategy_maintain_power_balance
		}
	}

	option = {
		name = ep2_meiji.1.a
		default_option = yes
		if = {
			limit = {
				has_modifier = forced_market_opening
			}
			every_scope_character = {
				limit = {
					has_ideology = ideology:ideology_shojoi
				}
				add_modifier = {
					name = shogun_forced_to_open_market_character
					days = very_long_modifier_time
					is_decaying = yes
				}
			}
			random_political_movement = { 
				limit = { 
					is_political_movement_type = movement_tokugawa_loyalist 
				} 
				add_modifier = {
					name = shogun_forced_to_open_market
					days = long_modifier_time
					is_decaying = yes
				}
			}
		}
		if = {
			limit = {
				owner = {
					any_political_movement = {
						is_political_movement_type = movement_tokugawa_loyalist
						has_core_ideology = ideology:ideology_tokugawa_loyalist_movement
					}
				}
			}
			owner = {
				random_political_movement = {
					limit = {
						is_political_movement_type = movement_tokugawa_loyalist
						has_core_ideology = ideology:ideology_tokugawa_loyalist_movement
					}
					set_core_ideology = ideology_tokugawa_loyalist_movement_2
					add_character_ideology = ideology_modernizer_leader
				}
			}
		}
	}

	after = {
		hidden_effect = {
			scope:tairo_scope ?= {
				set_character_immortal = no
			}
		}
	}
}

		}
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_meiji_restorationist
			}
			save_scope_as = restorationist_movement_scope
		}
		ordered_scope_character = {
			limit = {
				has_variable = daimyo_var
			}
			order_by = loyalty
			position = 0
			set_character_immortal = yes
			save_scope_as = loyal_daimyo_scope
		}
		ordered_scope_character = {
			limit = {
				has_variable = daimyo_var
			}
			order_by = {
				value = loyalty
				multiply = -1
			}
			position = 0
			set_character_immortal = yes
			save_scope_as = disloyal_daimyo_scope
		}
	}

	option = {
		name = ep2_meiji.51.a
		default_option = yes
		show_as_tooltip = {
			scope:shogun_scope ?= {
				set_as_interest_group_leader = yes
			}
			heir ?= {
				retire_character = yes
			}
		}
		je:je_meiji_restoration ?= {
			add_modifier = modifier_meiji_interregnum
		}
		activate_law = law_type:law_oligarchy
		if = {
			limit = {
				OR = {
					AND = {
						scope:shogun_scope ?= { # If one is pursuing Unification of Court and Shogunate and somehow get to this point, Traditionalist shoguns will fail automatically
							has_ideology = ideology:ideology_traditionalist
						}
						has_variable = kobu_gattai
					}
					government_legitimacy <= 50
				}
			}
			trigger_event = {
				id = ep2_meiji.6
				days = 7
			}
		} 
		else = {
			if = { # If one is pursuing Unification of Court and Shogunate and somehow get to this point, non-Traditionalist shoguns will automatically switch to Parliamentary Government
				limit = {
					has_variable = kobu_gattai
				}
				custom_tooltip = {
					text = kobu_gattai_no_longer_viable_tt
					remove_variable = kobu_gattai
					set_variable = kogi_yoron
				}
			}
			trigger_event = {
				id = ep2_meiji.7
				days = 7
			}
		}
		hidden_effect = {
			scope:loyal_daimyo_scope ?= {
				set_character_immortal = no
			}
			scope:disloyal_daimyo_scope ?= {
				set_character_immortal = no
			}
		}
	}
}
