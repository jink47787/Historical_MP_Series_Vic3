			}

			position = 0
			save_scope_as = government_ig
			leader = {
				save_scope_as = roju_scope
			}
		}
	}

	option = {
		name = tenpo_events.4.a
		default_option = yes
		custom_tooltip = {
			text = tenpo_events.4.tt_1
			scope:roju_scope = {
				kill_character = {
					hidden = yes
				}
			}
		}
		add_modifier = {
			name = modifier_tenpo_fail
			days = normal_modifier_time
		}
	}

	option = {
		name = tenpo_events.4.b
		if = {
			limit = {
				any_active_law = {
					has_amendment = amendment_type:amendment_frugal_ordinance
				}
			}
			random_active_law = {
				limit = {
					has_amendment = amendment_type:amendment_frugal_ordinance
				}
				random_scope_amendment = {
					limit = {
						amendment_type:amendment_frugal_ordinance ?= this.type
					}
					remove_amendment = yes
				}
			}
		}
		if = {
			limit = {
				has_law_or_variant = law_type:law_strict_edo_system
			}
			activate_law = law_type:law_intermediate_edo_system
			add_radicals = {
				pop_type = aristocrats
				value = medium_radicals
			}
			add_loyalists = {
				pop_type = shopkeepers
				value = medium_radicals
			}
		}
		else_if = {
			limit = {
				has_law_or_variant = law_type:law_lax_edo_system
			}
			activate_law = law_type:law_intermediate_edo_system
			add_radicals = {
				pop_type = shopkeepers
				value = medium_radicals
			}
			add_loyalists = {
				pop_type = aristocrats
				value = medium_radicals
			}
		}
		if = {
			limit = {
				NOT = {
					has_law_or_variant = law_type:law_hereditary_bureaucrats
				}
			}
			activate_law = law_type:law_hereditary_bureaucrats
			add_radicals = {
				pop_type = bureaucrats
				value = medium_radicals
			}
		}
		add_modifier = {
			name = modifier_tenpo_fail
			days = normal_modifier_time
			multiplier = 0.5
		}
		custom_tooltip = {
			text = tenpo_events.4.tt_1
			scope:roju_scope = {
				kill_character = {
					hidden = yes
				}
			}
		}
	}
}
