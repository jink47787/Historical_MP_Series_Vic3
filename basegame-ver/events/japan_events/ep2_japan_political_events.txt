namespace = japan_politics

# Rangakusha become Intelligentsia
japan_politics.1 = {
	type = country_event
	placement = root

	duration = 3

	title = japan_politics.1.t
	desc = japan_politics.1.d
	flavor = japan_politics.1.f

    event_image = {
		video = "ep2_intellectual_writer"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/ig_icons/intelligensia.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		c:JAP ?= this
		ig:ig_intelligentsia = {
			# Intended that this IG is not required to be non-marginal
			has_ideology = ideology:ideology_rangaku
		}
		NOR = {
			has_law_or_variant = law_type:law_isolationism
			has_law_or_variant = law_type:law_closed_borders
		}
	}

	cancellation_trigger = {

	}

	immediate = {
		ig:ig_intelligentsia = {
			save_scope_as = intelligentsia_ig
		}
	}

	option = {
		name = japan_politics.1.a
		default_option = yes
		ig:ig_intelligentsia = {
			set_interest_group_name = ig_intelligentsia
			remove_ideology = ideology_rangaku
			add_ideology = ideology_constitutionalist
		}
	}
}
