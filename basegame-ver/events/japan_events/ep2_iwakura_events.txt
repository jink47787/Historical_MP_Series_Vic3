iwakura_mission.2 = {
	type = country_event
	placement = root

	duration = 3

	title = iwakura_mission.2.t
	desc = iwakura_mission.2.d
	flavor = iwakura_mission.2.f

    event_image = {
		video = "ep2_intellectual_writer"
	}

	dlc = dlc018

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		# Triggered by JE completion
	}

	cancellation_trigger = {

	}

	immediate = {
		ordered_in_list = {
			variable = iwakura_mission_law_learnings
			limit = {
				save_temporary_scope_as = iwakura_mission_law_proposal
				trigger_if = {
					limit = {
						ROOT = {
							var:iwakura_world_political_progressiveness >= 35 # If world progressiveness is above 35, and thus option 35 may be available, this can't be a land reform law
						}
					}
					NOR = {
						law_type:law_commercialized_agriculture ?= this.type
						law_type:law_peasant_proprietorship ?= this.type
						law_type:law_homesteading ?= this.type
					}
				}
				ROOT = {
					any_interest_group = {
						is_in_government = yes
						law_enactment_stance = {
							law = scope:iwakura_mission_law_proposal.type
							value > neutral
						}
					}
				}
			}
			order_by = type.progressiveness
			position = 0
			save_scope_as = iwakura_desired_law
			owner ?= {
				save_scope_as = iwakura_desired_law_country
			}
		}
		if = {
			limit = {
				var:iwakura_world_political_progressiveness >= 35
			}
			ordered_in_list = {
				variable = iwakura_mission_law_learnings
				limit = {
					save_temporary_scope_as = iwakura_mission_law_proposal
					OR = {
						law_type:law_commercialized_agriculture ?= this.type
						law_type:law_peasant_proprietorship ?= this.type
						law_type:law_homesteading ?= this.type
					}
					NOT = {
						scope:iwakura_desired_law.type ?= this.type
					}
				}
				order_by = type.progressiveness
				position = 0
				save_scope_as = iwakura_land_reform_law
				owner ?= {
					save_scope_as = iwakura_land_reform_law_country
				}
			}
		}
		ordered_in_list = {
			variable = iwakura_mission_visited_countries
			limit = {
				country_rank >= rank_value:great_power
			}
			order_by = gdp
			position = 0
			save_scope_as = iwakura_good_economy_country
		}
	}

	option = {
		name = iwakura_mission.2.a
		default_option = yes
		add_modifier = {
			name = modifier_iwakura_mission_tech_bonus
			days = long_modifier_time
			is_decaying = yes
			multiplier = {
				value = var:iwakura_mission_tech_learnings
				max = 100
				min = 10
			}
		}
		ordered_in_list = {
			variable = iwakura_mission_visited_countries
			order_by = prestige
			min = 0
			max = 2
			create_diplomatic_catalyst = {
				type = catalyst_event_positive
				target = c:JAP
			}
		}
	}

	option = {
		name = iwakura_mission.2.b
		trigger = {
			custom_tooltip = {
				text = iwakura_mission_good_economy_tt
				exists = scope:iwakura_good_economy_country
			}
		}
		highlighted_option = yes
		add_modifier = {
			name = modifier_iwakura_mission_production_tech_bonus
			days = long_modifier_time
			is_decaying = yes
			multiplier = {
				value = var:iwakura_mission_tech_learnings
				max = 100
				min = 5
			}
		}
		create_diplomatic_catalyst = {
			type = catalyst_event_positive
			target = scope:iwakura_good_economy_country
		}
	}

	option = {
		name = iwakura_mission.2.c
		trigger = {
			custom_tooltip = {
				text = iwakura_mission_misc_laws_tt
				exists = scope:iwakura_desired_law
				exists = scope:iwakura_desired_law_country
			}
		}
		highlighted_option = yes
		activate_law = scope:iwakura_desired_law.type
		add_modifier = {
			name = modifier_iwakura_mission_society_tech_bonus
			days = long_modifier_time
			is_decaying = yes
			multiplier = {
				value = var:iwakura_mission_tech_learnings
				max = 100
				min = 5
			}
		}
		create_diplomatic_catalyst = {
			type = catalyst_event_positive
			target = scope:iwakura_desired_law_country
		}
	}

	option = {
		name = iwakura_mission.2.e
		trigger = {
			custom_tooltip = {
				text = iwakura_mission_land_reform_laws_tt
				exists = scope:iwakura_land_reform_law
				exists = scope:iwakura_land_reform_law_country
			}
			custom_tooltip = {
				text = high_world_progressiveness_tt
				var:iwakura_world_political_progressiveness >= 35
			}
		}
		highlighted_option = yes
		if = {
			limit = {
				scope:iwakura_land_reform_law.type ?= law_type:law_homesteading
			}
			activate_law = law_type:law_peasant_proprietorship
		}
		else = {
			activate_law = scope:iwakura_land_reform_law.type
		}
		add_modifier = {
			name = modifier_iwakura_mission_land_justice_bonus
			days = long_modifier_time
			is_decaying = yes
			multiplier = {
				value = var:iwakura_mission_tech_learnings
				max = 100
				min = 5
			}
		}
		create_diplomatic_catalyst = {
			type = catalyst_event_positive
			target = scope:iwakura_land_reform_law_country
		}
	}

	after = {
		iwakura_mission_cleanup = yes
		set_variable = iwakura_mission_finished
		change_variable = {
			name = meiji_var
			add = 1
		}
	}
}
