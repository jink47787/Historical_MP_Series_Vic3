# Debate the status of Christianity
japan_religion.6 = {
	type = country_event
	placement = root

	duration = 3

	title = japan_religion.6.t
	desc = japan_religion.6.d
	flavor = japan_religion.6.f

    event_image = {
		video = "southamerica_christianity"
	}

	dlc = dlc018

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/religion_icons/protestant.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		has_dlc_feature = ep2_content
		# Triggered by je_shinbutsu_bunri
		OR = {
			has_law_or_variant = law_type:law_state_religion
			has_law_or_variant = law_type:law_freedom_of_conscience
		}
		NOR = {
			has_law = law_type:law_sakoku
			has_law = law_type:law_closed_borders
		}
		any_country = {
			has_interest_marker_in_region = sr:region_northeast_asia
			religion ?= { has_discrimination_trait = heritage_christian }
		}
		any_scope_pop = {
			religion ?= { has_discrimination_trait = heritage_christian }
		}
	}

	cancellation_trigger = {

	}

	immediate = {
	}

	option = {
		name = japan_religion.6.a
		every_country = {
			limit = {
				has_interest_marker_in_region = sr:region_northeast_asia
				religion ?= { has_discrimination_trait = heritage_christian }
			}
			change_relations = {
				country = root
				value = 25
			}
			create_diplomatic_catalyst = {
				type = catalyst_event_positive
				target = root
			}
		}
		add_modifier = {
			name = permitted_christianity
			days = normal_modifier_time
		}
		every_scope_state = {
			limit = {
				any_scope_pop = {
					religion ?= { has_discrimination_trait = heritage_christian }
				}
			}
			hidden_effect = {
				random_scope_pop = {
					limit = {
						religion ?= { has_discrimination_trait = heritage_christian }
					}
					religion = { save_temporary_scope_as = christian_pop_religion }
				}
			}
			convert_population = {
				target = scope:christian_pop_religion
				value = 0.01
			}
		}
	}
	option = {
		name = japan_religion.6.b
		add_modifier = {
			name = permitted_christianity
			days = normal_modifier_time
		}
	}
	option = {
		name = japan_religion.6.c
		default_option = yes
		every_country = {
			limit = {
				has_interest_marker_in_region = sr:region_northeast_asia
				religion ?= { has_discrimination_trait = heritage_christian }
			}
			change_relations = {
				country = root
				value = -20
			}
			create_diplomatic_catalyst = {
				type = catalyst_event_negative
				target = root
			}
		}
		ig:ig_devout ?= {
			add_modifier = {
				name = christianity_disallowed
				days = normal_modifier_time
			}
		}
	}
}
