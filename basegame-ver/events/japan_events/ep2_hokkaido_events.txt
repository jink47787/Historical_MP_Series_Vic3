	}

	cancellation_trigger = {
	}

	immediate = {
		s:STATE_HOKKAIDO.region_state:JAP = {
			save_scope_as = hokkaido_state_scope
			set_variable = {
				name = optimization_counter_var
				value = 0
			}
		}
	}

	option = {
		name = hokkaido_events.3.a
		default_option = yes
		add_modifier = {
			name = modifier_introducing_new_crops
			days = normal_modifier_time
			multiplier = money_amount_multiplier_large
		}
		s:STATE_HOKKAIDO.region_state:JAP = {
			if = {
				limit = {
					var:optimization_counter_var = 0
				}
				add_modifier = {
					name = modifier_opimizing_existing_variants
					days = normal_modifier_time
					multiplier = 0.5
				}
			}
			else_if = {
				limit = {
					var:optimization_counter_var = 1
				}
				remove_modifier = modifier_opimizing_existing_variants
				add_modifier = {
					name = modifier_opimizing_existing_variants
					days = normal_modifier_time
				}
			}
			else_if = {
				limit = {
					var:optimization_counter_var = 2
				}
				remove_modifier = modifier_opimizing_existing_variants
				add_modifier = {
					name = modifier_opimizing_existing_variants
					days = normal_modifier_time
					multiplier = 1.5
				}
			}
		}
		ai_chance = {
			base = 10
		}
	}

	option = {
		name = hokkaido_events.3.b
		trigger = {
			NOT = {
				has_variable = hokkaido_rice_var
			}
		}
		set_variable = hokkaido_rice_var
		s:STATE_HOKKAIDO.region_state:JAP = {
			state_region = {
				add_resource_potential = building_rice_farm
				remove_resource_potential = building_wheat_farm				
			}
			if = {
				limit = {
					has_building = building_wheat_farm
				}
				remove_building = building_wheat_farm
				create_building = {
	 				building = building_rice_farm
	 				level = 3
	 				reserves = 1
	 			}
			}
		}
		add_modifier = {
			name = modifier_introducing_new_crops
			days = normal_modifier_time
			multiplier = money_amount_multiplier_medium
		}
		ai_chance = {
			base = 70
		}
	}

	option = {
		name = hokkaido_events.3.c
		trigger = {
			NOT = {
				has_variable = hokkaido_dye_var
			}
		}
		set_variable = hokkaido_dye_var
		s:STATE_HOKKAIDO = {
			add_resource_potential = building_dye_plantation
		}
		add_modifier = {
