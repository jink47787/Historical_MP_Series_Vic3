namespace = ottoman_monarchs

ottoman_monarchs.1 = {
	type = country_event
	placement = root

	duration = 3

	dlc = dlc014

	title = ottoman_monarchs.1.t
	desc = ottoman_monarchs.1.d
	flavor = ottoman_monarchs.1.f

	event_image = {
		video = "votp_prison"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	cooldown = { days = very_long_modifier_time }

	trigger = {

	}

	immediate = {

	}

	cancellation_trigger = {
		ruler = {
			NOT = {
				has_variable = is_abdulaziz_osmanoglu
			}
		}
	}

	option = {
		name = ottoman_monarchs.1.a
		default_option = yes
		custom_tooltip = {
			create_character = {
				template = tur_murad_v_osmanoglu_template
			}
			if = {
				limit = {
					any_scope_character = {
						has_template = tur_abdulaziz_osmanoglu_template
					}
				}
				random_scope_character = {
					limit = {
						has_template = tur_abdulaziz_osmanoglu_template
					}
					kill_character = {
						hidden = yes
					}
				}
			}
			text = ottoman_monarchs.1_tt
		}
		hidden_effect = {
			set_variable = sultan_aziz_coup_happened
		}
	}
}
