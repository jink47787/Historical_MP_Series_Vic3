		}
	}

	option = {
		default_option = yes
		name = serbian_throne.3.a

			75 = {
				scope:serb_busted_ig_leader = {
					kill_character = yes
				}
				add_modifier = {
					name = serbia_exposing_hypocrite_mod
					days = short_modifier_time
				}
				custom_tooltip = serbian_dethrone_loyalty_progress_increase_tt

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	trigger = {
		# Triggered by JE
	}

	immediate = {
		hidden_effect = {
			serbian_throne_modifier_cleanser = yes
			serbian_throne_variable_cleanser = yes
		}
		if = {
			limit = {
				ser_exists_contender_of_other_dynasty = yes
			}
			ser_save_contender_of_rival_dynasty_as_scope = yes
		}
		je:je_serbian_throne ?= {
			set_bar_progress = { value = 250 name = serbian_dethrone_bar }
		}

		name = serbian_throne.10.a

			limit = {
				ser_character_is_of_current_ruling_dynasty = yes
			}

		}

			limit = {
				exists = scope:rival_dynasty_contender_scope
