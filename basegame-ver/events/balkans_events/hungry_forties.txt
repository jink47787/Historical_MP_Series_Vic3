namespace = hungry_forties

hungry_forties.1 = {
	type = country_event
	event_image = {
		video = "asia_dead_cattle_poor_harvest"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	title = hungry_forties.1.t
	desc = hungry_forties.1.d
	flavor = hungry_forties.1.f

	trigger = {
		# Triggered by on_action
	}

	immediate = {
		random_state = {
			limit = {
				state_is_in_europe = yes
				state_region = {
					has_harvest_condition = disease_outbreak
				}
			}
			save_scope_as = relevant_state
		}
	}

	duration = 3

	option = {
		name = hungry_forties.1.a
        default_option = yes
		je:je_the_hungry_forties = {
			add_modifier = {
				name = modifier_european_potato_famine
			}
		}
	}

    option = {
		name = hungry_forties.1.b
		je:je_the_hungry_forties = {
			add_modifier = {
				name = modifier_european_potato_famine_relief
			}
		}
		add_modifier = {
			name = modifier_european_potato_famine_relief_costs
			multiplier = money_amount_multiplier_small
			days = short_modifier_time
		}
	}
}
