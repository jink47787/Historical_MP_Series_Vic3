hawaii_events.2 = { #JE success
	type = country_event
	placement = ROOT

	title = hawaii_events.2.t
	desc = hawaii_events.2.d
	flavor = hawaii_events.2.f

	event_image = {
		video = "southamerica_factory_opening"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		remove_global_variable = gp_hawaii_interest
	}

	option = { #get an extra defense modifier
		name = hawaii_events.2.a
		ai_chance = {
			base = 75
		}
		default_option = yes
		every_scope_state = {
			limit = { is_incorporated = yes }
			add_modifier = {
				name = national_defence_modifier
				days = very_long_modifier_time
			}
		}
	}
	option = { #deport businessmen thinking of coup
		name = hawaii_events.2.b
		ai_chance = {
			base = 50
			modifier = {
				trigger = {
					ig:ig_industrialists = {
						ig_approval < -5
					}
				}
				add = -25
			}
			modifier = {
				trigger = {
					ig:ig_industrialists = {
						is_powerful = yes
					}
				}
				add = -25
			}
		}
		add_modifier = {
			name = national_defence_modifier
			days = long_modifier_time
			multiplier = 0.5
		}
		set_variable = deported_americans
		ig:ig_industrialists = {
			add_modifier = {
				name = deporting_disloyal_igs
				days = long_modifier_time
				is_decaying = yes
			}
			if = {
				limit = {
					leader = {
						culture = {
							NOT = { this = cu:hawaiian }
						}
					}
				}
				leader = {
					kill_character = yes
				}
			}
		}
		ig:ig_devout = {
			set_interest_group_name = ig_evangelicals
			remove_ideology = ideology_laissez_faire
			if = {
				limit = {
					leader = {
						culture = {
							NOT = { this = cu:hawaiian }
						}
					}
				}
				leader = {
					kill_character = yes
				}
			}
		}
		if = {
			limit = {
				has_law = law_type:law_monarchy
			}
			ig:ig_intelligentsia = {
				set_interest_group_name = ig_intelligentsia
				remove_ideology = ideology_republican
				add_ideology = ideology_constitutionalist
			}
		}
		if = {
			limit = {
				any_country = {
					has_variable = hawaii_interest
				}
			}
			every_country = {
				limit = {
					has_variable = hawaii_interest
				}
				trigger_event = {
					id = hawaii_events.5
				}
			}
		}
	}
}
hawaii_events.3 = { #Failed JE
	type = country_event
	placement = ROOT

	title = hawaii_events.3.t
	desc = hawaii_events.3.d
	flavor = hawaii_events.3.f

	event_image = {
		video = "europenorthamerica_springtime_of_nations"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		remove_global_variable = gp_hawaii_interest
		if = {
			limit = {
				is_subject = yes
			}
			overlord = {
				save_scope_as = hawaiian_coup
			}
		}
		else_if = {
			limit = {
				exists = power_bloc
				is_power_bloc_leader = no
			}
			power_bloc ?= {
				power_bloc_leader ?= {
					save_scope_as = hawaiian_coup
				}
			}
		}
		else_if = {
			limit = {
				any_country = {
					has_variable = hawaii_interest
				}
			}
			random_country = {
				limit = {
					has_variable = hawaii_interest
				}
				weight = {
					base = 1
					modifier = {
						je:je_hawaiian_interest ?= {
							has_modifier = influence_over_hawaii
						}
						add = 300
					}
					modifier = {
						c:USA ?= this
						add = 300
					}
					modifier = {
						c:USA ?= this
						is_player = yes
						add = 300
					}
					modifier = {
						has_diplomatic_pact = {
							who = root
							type = fund_lobbies
						}
						add = 150
					}
					modifier = {
						has_treaty_foreign_investment_rights_with = { TARGET = root }
						add = 150
					}
					modifier = {
						has_treaty_trade_privilege_with = { TARGET = root }
						add = 150
					}
					modifier = {
						"relations(root)" >= relations_threshold:cordial
						"relations(root)" < relations_threshold:amicable
						add = 100
					}
					modifier = {
						"relations(root)" >= relations_threshold:amicable
						"relations(root)" < relations_threshold:friendly
						add = 200
					}
					modifier = {
						"relations(root)" >= relations_threshold:friendly
						add = 300
					}
				}
				save_scope_as = hawaiian_coup
			}
		}
	}

	option = { #Government overthrown policy change
		name = hawaii_events.3.a
		default_option = yes
		activate_law = law_type:law_presidential_republic
		activate_law = law_type:law_oligarchy
		ruler = {
			kill_character = yes
		}
		if = {
			limit = {
				exists = scope:hawaiian_coup
			}
			scope:hawaiian_coup = {
				trigger_event = {
					id = hawaii_events.4
				}
			}
		}
	}
}

hawaii_events.4 = { #Great Power makes hawaii puppet or annex's them
	type = country_event
	placement = ROOT

	title = hawaii_events.4.t
	desc = hawaii_events.4.d
	flavor = hawaii_events.4.f

	event_image = {
		video = "southamerica_aristocrats"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 4

	trigger = {
	}

	immediate = {
	}

	option = { #makes hawaii puppet
		name = hawaii_events.4.a
		default_option = yes
		create_diplomatic_pact = {
			country = c:HAW
			type = puppet
		}
		create_incident = {
			country = c:HAW
			value = 5
		}
		ai_chance = {
			base = 50
			modifier = {
				add = 50
				c:USA = ROOT
			}
		}
	}
	option = { #annex hawaii
		name = hawaii_events.4.b
		trigger = {
			exists = c:HAW
			c:HAW = {
				is_player = no
			}
		}
		annex = c:HAW
		create_incident = {
			country = c:HAW
			value = 10
		}
		ai_chance = {
			base = 25
			modifier = {
				add = 50
				c:USA = ROOT
			}
		}
	}
	option = { #no hostile action improve relations
		name = hawaii_events.4.c
		change_relations = {
			country = c:HAW
			value = 20
		}
		ai_chance = {
			base = 25
		}
	}
}
