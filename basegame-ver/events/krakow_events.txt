namespace = krakow

#Manifesto for the Polish Nation
krakow.1 = {
	type = country_event
	placement = scope:krakow_state

	title = krakow.1.t
	desc = krakow.1.d
	flavor = krakow.1.f

	event_image = {
		video = "europenorthamerica_springtime_of_nations"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	dlc = dlc004

	trigger = {
		#Fired by on_action
	}

	immediate = {
		c:KRA = {
			capital = {
				save_scope_as = krakow_state
			}
			save_scope_as = krakow_scope
		}
	}

	option = {
		name = krakow.1.a
		default_option = yes
		ai_chance = {
			base = 75
			modifier = {
				trigger = {
					OR = {
						any_country = {
							country_definition = cd:KRA
							is_player = yes
						}
					}
				}
				add = -100
			}
		}
		set_variable = { name = krakow_uprising_happened months = 120 }
		add_journal_entry = {
			type = je_the_krakow_uprising
		}
		every_scope_state = {
			limit = {
				state_region = {
					is_homeland = cu:polish
				}
			}
			add_radicals_in_state = {
				value = small_radicals
				culture = cu:polish
			}
		}
	}

	option = {
		name = krakow.1.b
		ai_chance = {
			base = 5
		}
		every_scope_state = {
			limit = {
				state_region = {
					is_homeland = cu:polish
				}
			}
			add_radicals_in_state = {
				value = large_radicals
				culture = cu:polish
			}
		}
		set_variable = dismissed_krakow
	}
}

namespace = galicia_formation

# Formation of Galicia
galicia_formation.1 = {
	type = country_event
	placement = scope:west_galicia_state

	title = galicia_formation.1.t
	desc = galicia_formation.1.d
	flavor = galicia_formation.1.f

	event_image = {
		video = "europenorthamerica_london_center"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	dlc = dlc004

	trigger = {
		#Fired by on_action
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_WEST_GALICIA
			}
			save_scope_as = west_galicia_state
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_EAST_GALICIA
			}
			save_scope_as = east_galicia_state
		}
	}

	option = {
		name = galicia_formation.1.a
		default_option = yes
		remove_primary_culture = cu:ukrainian
		add_loyalists = {
			value = large_radicals
			culture = cu:polish
			pop_type = aristocrats
		}
		add_radicals = {
			value = medium_radicals
			culture = cu:ukrainian
		}
	}

	option = {
		name = galicia_formation.1.b
		custom_tooltip = ukrainian_culture_remains_krakow_tt
		add_radicals = {
			value = medium_radicals
			culture = cu:polish
		}
		custom_tooltip = ukrainian_culture_loss_warning_krakow_tt
	}

	option = {
		name = galicia_formation.1.c
		trigger = {
			custom_tooltip = {
				text = ruler_is_romance_tt
				ruler = {
					culture = {
						has_discrimination_trait = language_hispanophone
					}
				}
			}
		}
		highlighted_option = yes
		custom_tooltip = ukrainian_culture_remains_krakow_tt
		add_primary_culture = cu:galician
		add_radicals = {
			value = medium_radicals
			culture = cu:polish
		}
		add_radicals = {
			value = medium_radicals
			culture = cu:ukrainian
		}
		s:STATE_GALICIA = {
			add_claim = ROOT
		}
		custom_tooltip = ukrainian_culture_loss_warning_krakow_tt
	}
}
