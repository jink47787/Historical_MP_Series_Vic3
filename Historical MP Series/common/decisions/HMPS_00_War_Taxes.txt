levy_war_taxes = {
	is_shown = {
		is_player = yes
		is_at_war = yes
	}
	
	possible = {
		trigger_if = {
			limit = {
				has_variable = trimonthly
			}
			custom_tooltip = {
				text = levy_war_taxes.t
				NOT = { has_variable = trimonthly }
			}
		}
		has_technology_researched = centralization
		any_scope_war = {
			OR = {
				num_country_casualties = { 
					target = root
					value > 150000
				}
				has_war_support = {
					target = ROOT
					value < 0
				}
				num_dead >= 1000000
			}
		}
		government_legitimacy >= 25
	}
	
	when_taken = {
		add_modifier = {
			name = modifier_war_taxes
			years = 1
		}
		add_radicals = {
			value = small_radicals
		}
		set_variable = {
			name = trimonthly
			months = 3
		}
	}
}
