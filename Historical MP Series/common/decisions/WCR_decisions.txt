WCR_de_simple_westernization = {
    is_shown = {
        has_game_rule = WCR_simple_westernization
        WCR_st_is_country_westernized = no # WCR_is_westernized_trigger doesn't work right here for some reason?
        NOT = {
            has_variable = WCR_decision_taken
        }
        OR = {
            is_ai = no
            has_game_rule = WCR_allow_AI_westernization
        }
    }
    
    possible = {
        trigger_if = {
            limit = {
                has_game_rule = WCR_WCF_requirements
            }
            WCR_can_start_westernizing_trigger = yes
        }
        trigger_else = {
            WCR_snori_can_westernize_trigger = yes
        }
    }
    
    when_taken = {
        set_variable = {
            name = WCR_decision_taken
            days = 120
        }
        if = {
            limit = {
                has_game_rule = WCR_enable_influences
            }
            trigger_event = {
                id    = WCR_ev.2
                popup = yes
            }
        }
        else = {
            set_variable = {
                name  = WCR_ruler_influence
                value = 0
            }
            set_variable = {
                name  = WCR_army_influence
                value = 0
            }
            set_variable = {
                name  = WCR_navy_influence
                value = 0
            }
            trigger_event = {
                id = WCR_ev.1
            }
            add_modifier = {
                name  = WCR_ig_westernized_modifier
                years = 10
                is_decaying = yes
            }
        }
    }
    
    ai_chance = {
        base = 0
        modifier = {
            trigger = {
                calc_true_if = {
                    amount >= 2
                    ig:ig_intelligentsia = {
                        is_powerful = yes
                    }
                    ig:ig_industrialists = {
                        is_powerful = yes
                    }
                    ig:ig_armed_forces = {
                        is_powerful = yes
                    }
                    ig:ig_trade_unions = {
                        is_powerful = yes
                    }
                }
            }
            add = 100
        }
        modifier = {
            trigger = {
                game_date >= 1900.1.1
            }
            add = 100
        }
        modifier = {
            trigger = {
                game_date >= 1915.1.1
            }
            add = 200
        }
        modifier = {
            trigger = {
                game_date >= 1930.1.1
            }
            add = 300
        }
        modifier = {
            trigger = {
                has_variable = meiji_var
            }
            add = 100
        }
        modifier = {
            trigger = {
                calc_true_if = {
                    amount >= 3
                    ig:ig_landowners = {
                        is_marginal = no
                        ig_approval < -5
                    }
                    ig:ig_devout = {
                        is_marginal = no
                        ig_approval < -5
                    }
                    ig:ig_rural_folk = {
                        is_marginal = no
                        ig_approval < -5
                    }
                    ig:ig_petty_bourgeoisie = {
                        is_marginal = no
                        ig_approval < -5
                    }
                }
            }
            add = -500
        }
    }
}

WCR_abandon_westernization = {
    is_shown = {
        has_game_rule = WCR_complex_westernization
        WCR_st_is_country_westernized = no
        has_journal_entry = WCR_je_westernization
		is_country_type = unrecognized
		NOT = { has_variable = abandoned_westernisation }
    }
    
    possible = {
        is_ai = no
    }
    
    when_taken = {
		custom_tooltip = {
			text = abandoned_westernisation_tt
			set_variable = abandoned_westernisation
		}
    }
    
    ai_chance = {
        base = 0
    }
}
