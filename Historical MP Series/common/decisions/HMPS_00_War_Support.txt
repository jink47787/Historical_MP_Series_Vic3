boost_war_support = {
	is_shown = {
		is_player = yes
		is_at_war = yes
		has_technology_researched = war_propaganda
	}
	
	possible = {
		in_default = no
		any_scope_war = {
		    has_war_support = {
		        target = ROOT
		        value < 0
		    }
		}
		has_technology_researched = war_propaganda
		government_legitimacy >= 25
		hidden_trigger = {
			NOT = { has_variable = bad_code }
		}
	}
	
	when_taken = {
		set_variable = {
			name = bad_code
			days = 93
		}
		trigger_event = {
			id = war_support.1
			days = 0
			popup = yes
		}
	}
}
