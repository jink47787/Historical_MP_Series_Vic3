indonesia_home_rule_decision = {
	is_shown = {
		exists = c:DEI
		this = c:DEI
		is_subject = yes
	}

	possible = {
		country_has_voting_franchise = yes
		has_law = law_type:law_elected_bureaucrats
		has_law = law_type:law_multicultural
		root.average_sol > 15
	}

	when_taken = {
		remove_modifier = modifier_has_shown_colonial_favouritism
		if = { 
			limit = {
				NOT = {
					has_modifier = secondary_culture_javan
				}
			}
			add_modifier = secondary_culture_javan
		}
		add_modifier = secondary_culture_malay
		add_modifier = secondary_culture_bornean
		add_modifier = secondary_culture_sumatran
		add_modifier = secondary_culture_balinese
		if = {
			limit = {
				has_law = law_type:law_extraction_economy
			}
			activate_law = law_type:law_agrarianism
		}
		if = {
			limit = {
				has_modifier = modifier_indonesia_crown_rule
			}
			remove_modifier = modifier_indonesia_crown_rule
		}
		if = {
			limit = {
				ig:ig_industrialists = {
					has_ideology = ideology_colonialist
				}
			}
			ig:ig_industrialists = {
				remove_ideology = ideology_colonialist
				add_ideology = ideology_laissez_faire
			}
		}
	}

	ai_chance = {
		value = 100
	}
}

south_africa_home_rule_decision = {
	is_shown = {
		AND = {
			exists = c:SAF
			this = c:SAF
			cu:boer = {
				NOT = { has_discrimination_trait = tradition_mzansian }
			}
			#NOT = { country_has_primary_culture = cu:zulu }
		}
	}

	possible = {
		country_has_primary_culture = cu:boer
		#country_has_voting_franchise = yes
		has_law_or_variant = law_type:law_universal_suffrage
		has_law_or_variant = law_type:law_elected_bureaucrats
		has_law_or_variant = law_type:law_multicultural
		root.average_sol > 15
	}

	when_taken = {
		cu:boer = { add_tradition_trait = tradition_mzansian }
		cu:sotho = { add_tradition_trait = tradition_mzansian }
		cu:zulu = { add_tradition_trait = tradition_mzansian }
		cu:khoisan = { add_tradition_trait = tradition_mzansian }
		cu:xhosa = { add_tradition_trait = tradition_mzansian }
		cu:tswana = { add_tradition_trait = tradition_mzansian }
		cu:nguni = { add_tradition_trait = tradition_mzansian }
		cu:shona = { add_tradition_trait = tradition_mzansian }
		cu:griqua = { add_tradition_trait = tradition_mzansian }
		every_scope_culture = {
			limit = {
				has_discrimination_trait_group = heritage_group_european
			}
			prev = {
				add_radicals = {
					culture = prev
					value = 0.2
				}
			}
		}
		if = {
			limit = {
				has_modifier = secondary_culture_sotho
			}
			remove_modifier = secondary_culture_sotho
		}
		remove_modifier = modifier_has_shown_colonial_favouritism
	}

	ai_chance = {
		value = 100
	}
}
