population_replacement_and_seeding = {### this fires yearly
	save_scope_as = new_state
	
	### Seeding Conversion
	convert_population = { ### Seeding Conversion
		target = owner.religion
		value = 0.0001
	}
	
	
	
	###### Forced Pop replacement and Seeding Assimilation
	owner = {
		random_primary_culture = {
			save_scope_as = spawning_culture
		}
	}
	if = {#### Forced population replacement
		limit = {
			modifier:state_yearly_population_replacement_add > 0
		}
		ordered_scope_pop = {
			limit = {
				culture = {
					shares_heritage_trait_with_any_primary_culture = root
				}
			}
			order_by = total_size
			min=0
			#max=10
			check_range_bounds = no
			change_pop_culture = { 
				target = scope:spawning_culture 
				value = state.modifier:state_yearly_population_replacement_add
			}
		}
	}
	else_if = {#### Seeding Assimilation in states where a primary Culture does not have a cultural community
		limit = {
			NOT = {
				has_cultural_community = scope:spawning_culture
			}
		}
		ordered_scope_pop = {
			limit = {
				culture = {
					shares_heritage_trait_with_any_primary_culture = root
				}
			}
			order_by = total_size
			min=0
			#max=10
			check_range_bounds = no
			change_pop_culture = { 
				target = scope:spawning_culture 
				value = 0.001
			}
		}
	}
	
}
