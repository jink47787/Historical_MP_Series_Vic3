INJECT:conquer_state = { #HMPS: Applies revanchism target modifier on state annexed
	on_enforced = {
		if = {
			limit = {
				scope:target_state = {
					has_modifier = modifier_hmps_revanchism_target
				}
			}
			scope:target_state = {
				remove_modifier = modifier_hmps_revanchism_target
			}
		}
		scope:target_state = {
			add_modifier = {
				name = modifier_hmps_revanchism_target
				years = 50
			}
		}
	}
}

INJECT:annex_country = { #HMPS: Applies revanchism target modifier on all state annexed
	on_enforced = {
		scope:target_country = {
			every_state = {
				limit = {
					owner = scope:target_country
				}
				if = {
					limit = {	
						has_modifier = modifier_hmps_revanchism_target
					}
					remove_modifier = modifier_hmps_revanchism_target
				}
				add_modifier = {
					name = modifier_hmps_revanchism_target
					years = 50
				}
			}
		}
	}
}

INJECT:make_protectorate = { #HMPS: Applies revanchism target modifier on all states of a german minor when forced to become a subject
	on_enforced = {
		scope:target_country = {
			set_variable = { #HMPS: Prevents new subjects from immediatly removing the revanchism target from land fed to them
				name = hmps_cant_remove_revanchism_targets
				years = 50
			}
		}
		if = {
			limit = {
				scope:target_country = {
					capital = {
						is_in_geographic_region = geographic_region_lesser_germany
					}
				}
			}			
			scope:target_country = {
				every_state = {
					limit = {
						owner = scope:target_country
					}
					if = {
						limit = {	
							has_modifier = modifier_hmps_revanchism_target
						}
						remove_modifier = modifier_hmps_revanchism_target
					}
					add_modifier = {
						name = modifier_hmps_revanchism_target
						years = 50
					}
				}
			}
		}
	}
}
