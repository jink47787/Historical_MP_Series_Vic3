	# scope:amenability = IG amenability to negotiation divided by MAX_AMENABILITY (ie, if MAX_AMENABILITY is 100 and IG amenability is 50, this value will be 0.5)
	# scope:advance_chance = Chance to advance law enactment
	# scope:stall_chance = Chance for law enactment to stall

			}

				value = scope:advance_chance

			}

	# How likely should the AI be to have a particular strategic desire towards another country
	# Trigger is used for filtering, so if it fails score will always be 0 from this strategy
	# Using this value in other strategies will function additively
	# scope:target_country = target country
	secret_goal_scores = {
		none = {
			trigger = {
				NOT = {
					has_subject_relation_with = scope:target_country
				}
			}
			score = {
				value = 150

				if = { # Monarchist Austria really does not want revolutions or secessions
					limit = {
						c:AUS ?= this
						has_law = law_type:law_monarchy
						scope:target_country = {
							OR = {
								is_revolutionary = yes
								is_secessionist = yes
							}
						}
					}
					add = 500
				}

				# Minor german states shouldn't have far-reaching diplomatic ambitions
				if = {
					limit = {
						country_rank < rank_value:major_power
						OR = {
							country_has_primary_culture = cu:north_german
							country_has_primary_culture = cu:south_german
						}
						scope:target_country = {
							NOR = {
								country_has_primary_culture = cu:north_german
								country_has_primary_culture = cu:south_german
							}
						}
						NOT = { is_adjacent_to_country = scope:target_country }
					}
					add = 1000
				}

				# Large isolationist countries should ignore the world
				if = {
					limit = {
						has_law_or_variant = law_type:law_isolationism
						gdp >= 5000000
					}
					add = 10000
				}

				if = {
					limit = {
						country_rank = scope:target_country.country_rank
					}
					multiply = 0.5
				}
			}
		}
		befriend = {
			trigger = {
				# Don't befriend large isolationist powers
				NOT = {
					scope:target_country = {
						has_law_or_variant = law_type:law_isolationism
						gdp >= 5000000
					}
				}

				# Monarchist Austria really does not want to help revolutions or secessions
				NAND = {
					c:AUS ?= this
					has_law = law_type:law_monarchy
					scope:target_country = {
						OR = {
							is_revolutionary = yes
							is_secessionist = yes
						}
					}
				}
			}

			score = {
				value = 200
				if = {
					limit = {
						has_journal_entry = je_the_grand_collapse
						any_subject_or_below = {
							count >= 1
						}
						scope:target_country = {
							relations:root <= relations_threshold:cordial
							is_subject_of = root
						}
					}
					add = 500
				}

				if = {
					limit = {
						country_rank <= scope:target_country.country_rank
					}
					add = 100
				}

				if = {
					limit = {
						has_journal_entry = je_earn_recognition
						scope:target_country = { country_rank >= rank_value:great_power }
					}
					add = 200
				}

				if = { # To make Britain want to befriend an Asian major power
					limit = {
						country_has_primary_culture = cu:british
						exists = c:RUS
						c:RUS ?= {
							any_scope_state = {
								region = sr:region_manchuria
								OR = {
									any_scope_building = {
										is_building_type = building_port
										level >= 5
									}
									any_scope_building = {
										is_building_type = building_naval_base
										level >= 15
									}
								}
							}
						}
						scope:target_country = {
							country_rank >= rank_value:unrecognized_major_power
							OR = {
								capital.region = sr:region_japan
								capital.region = sr:region_manchuria
								capital.region = sr:region_north_china
								capital.region = sr:region_south_china
								capital.region = sr:region_indonesia
							}
						}
					}
					add = 200
				}

				if = { # Try to be on good terms with GBR/USA to avoid intervention
					limit = {
						has_journal_entry = je_reconquista
						scope:target_country = {
							country_rank = rank_value:great_power
							OR = {
								c:GBR ?= this
								AND = {
									is_in_geographic_region = geographic_region_americas
									is_historical_spanish_american_colony = no
								}
							}
						}
					}
					add = 200
				}
				else_if = { # Reconcile with former Spanish Colonies
					limit = {
						has_journal_entry = je_hispanoamerica
						scope:target_country = {
							has_modifier = modifier_not_recognized_by_spain

		force_nationalization = {

			add = {
				value = "nationalization_cost(scope:target_country)"
				divide = {

				}

			}

			# Score by fraction of GDP that is owned by the target country
			add = {
				value = "gdp_ownership_ratio(scope:target_country)"
				divide = {
					value = root.gdp
					min = 100000 # Set a floor for GDP to avoid dividing by zero or other very small numbers
				}
				multiply = 10 # Max score at 10%+ of GDP
				max = 50
			}

			if = {
				limit = {
					NOT = {
						has_attitude = {
							who = scope:target_country
							attitude = antagonistic
						}
					}
				}

				multiply = 0.5
			}
		}

		open_market = {
			value = 0

				limit = {
					scope:target_country = {
						has_diplomatic_pact = {
							who = root
							type = embargo
						}
					}
					can_reach_target_country = yes

						has_variable = opium_wars_target
					}
				}
				add = 75
			}

					AND = {
						has_radical_leftist_government = no

							has_radical_leftist_government = yes

					AND = {
						has_radical_leftist_economy = no
						scope:target_country = {
							has_radical_leftist_economy = yes
						}
					}

				}
				add = 50
			}

			if = {
				limit = {
					scope:target_country = {
						gdp < 2000000
					}
				}
				multiply = 0
			}

			# Don't want to open our own market...
			if = {
				limit = {
					market_capital.market = scope:target_country.market_capital.market
				}
				multiply = 0
			}
		}

		revoke_claim = {

			max = 50
		}

					}

				}

				limit = {
					has_journal_entry = je_reunify_china
					scope:target_country = {
						has_journal_entry = je_reunify_china
						has_strategic_adjacency = scope:target_country
					}
				}
				add = 100
			}

			if = {
				limit = {
					has_journal_entry = je_unify_afghanistan
					scope:target_country = {
						OR = {
							country_has_primary_culture = cu:pashtun
							country_has_primary_culture = cu:tajik
						}
					}
				}
				add = 100
			}

			if = {
				limit = {
					has_journal_entry = je_eastern_frontier
					capital = {
						has_claim_by = ROOT
					}
				}
				add = 50
			}

			if = {
				limit = {
					has_strategic_adjacency = scope:target_country
				}
				multiply = 1.5
			}

						is_princely_state = yes
					}

			}

			if = {
				limit = {
					lenient_ai_behavior_trigger = yes
				}
				multiply = 0.0
			}

			if = {
				limit = {
					OR = {
						has_journal_entry = je_eastern_frontier
						has_journal_entry = je_unify_afghanistan
						has_journal_entry = je_reunify_china
					}
				}
				min = 0
				max = 500
			}
			else = {
				min = 0
				max = 200
			}
		}

						country_has_primary_culture = cu:argentine
