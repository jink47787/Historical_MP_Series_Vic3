# Checks if any diplomatic play is possible against the target country
# Root = Country
# scope:target_country = target of a diplomatic play
can_start_diplomatic_plays_against = {
	# Not allowed to attack subjects of play participants
	NOT = {
		scope:target_country.top_overlord ?= {
			is_diplomatic_play_participant_with = root
		}
	}
	# Might be disallowed by power bloc
	NOT = {
		custom_tooltip = {
			text = "power_bloc_disallow_war_bool"
			is_in_same_power_bloc = scope:target_country
			power_bloc ?= {
				modifier:power_bloc_disallow_war_bool = yes
			}
			# Conflicts between Overlord and (their own) Subject are always permitted despite Power Bloc special conditions
			NOR = {
				is_direct_subject_of = scope:target_country
				scope:target_country = {
					is_direct_subject_of = root
				}
			}
		}
	}
	NOT = {
		# Disallow plays targeting your indirect subjects
		scope:target_country = {
			is_indirect_subject_of = root
		}
	}

	OR = {
		highest_overlapping_interest_tier = { target = scope:target_country value >= interest_tier_effect_can_start_diplomatic_plays_min_tier }
		is_direct_subject_of = scope:target_country
		scope:target_country = { is_direct_subject_of = root }
	}	

	NOT = {
		scope:target_country = {
			has_journal_entry = je_the_first_carlist_war
		}
	}
}

# Checks if Country can join the target country's side in a diplomatic play
# Root = Country
# scope:target_country = country of the side to join to
# scope:enemy_country = enemy country of target_country to which we could fight against
can_join_side_in_diplomatic_play = {
	# Might be disallowed by power bloc
	NOT = {
		custom_tooltip = {
			text = "POWER_BLOC_CANNOT_SIDE_AGAINST_OVERLORD"
			is_in_same_power_bloc = scope:target_country
			is_in_same_power_bloc = scope:enemy_country
			power_bloc ?= {
				modifier:power_bloc_disallow_war_bool = yes
			}
			# Conflicts between Overlord and (their own) Subject are always permitted despite Power Bloc special conditions
			# We can join the overlord side but not the subject's one
			scope:target_country = {
				is_direct_subject_of = scope:enemy_country
			}
		}
	}
	NOT = {
		# Disallow plays targeting your indirect subjects
		scope:enemy_country = {
			is_indirect_subject_of = root
		}
	}

	OR = {
		highest_overlapping_interest_tier = { target = scope:target_country value >= interest_tier_effect_can_join_diplomatic_plays_min_tier }
		is_direct_subject_of = scope:enemy_country
		scope:enemy_country = { is_direct_subject_of = root }
		is_direct_subject_of = scope:target_country
		scope:target_country = { is_direct_subject_of = root }
	}
}

# Evaluated when checking if a country can sign a treaty with another. Either through the player doing it in the UI or the AI looking for treaties to propose
# root = initiating country
# scope:other_country = other country in the potential treaty
can_sign_treaty_with = {
	scope:other_country = {
		NOT = { is_country_type = decentralized }
					   
	}
	root = {
		NOT = { is_diplomatic_play_enemy_of = scope:other_country }
	}

	# TODO PRCAL-41248: Add a new and properly localized trigger
	#has_diplomatic_relevance = scope:other_country
	OR = {
		has_strategic_adjacency = scope:other_country
		AND = {
			has_port_country = yes
			scope:other_country = { has_port_country = yes }
		}
		is_in_same_power_bloc = scope:other_country
	}
}

# Evaluated when checking if a country can threaten naval hostilities in a treaty
# root = threatening country
# scope:other_country = threatened country
can_threaten_naval_hostilities = {
	navy_size >= 10

	custom_tooltip = {
		text = "THREATEN_NAVAL_HOSTILITIES_NAVY_MUST_BE_STRONGER"
		naval_combat_power > scope:other_country.naval_combat_power
	}

	custom_tooltip = {
		text = "THREATEN_NAVAL_HOSTILITIES_NOT_LANDLOCKED"
		scope:other_country = {
			any_scope_state = {
				is_coastal = yes
			}
		}
	}
	
	"relations(scope:other_country)" < relations_threshold:cordial
	NOR = { 
		is_diplomatic_play_ally_of = scope:other_country
		any_scope_treaty = {
			binds = scope:other_country
			any_scope_article = {
				OR = {
					has_type = alliance
					has_type = defensive_pact
					has_type = guarantee_independence
				}
			}
		}
	}
}
