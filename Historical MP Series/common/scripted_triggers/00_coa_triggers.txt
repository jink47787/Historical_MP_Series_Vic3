coa_fascist_trigger = {
	# A fascist can lead a monarchy or a theocracy without the government changing form. The Papal State should retain the same flag even if the Pope's a fascist, for example.
	# A fascist can also lead a democracy without the democracy immediately becoming fascist. If the fascist successfully implements his wildest dreams and becomes a dictatorship, then sure the government will become Fascist, but while there's still voting the government shouldn't count as (fully) fascist yet.
	OR = {
		coa_oligarchy_trigger = yes
		coa_dictatorship_trigger = yes
	}
	OR = {
		has_law_or_variant = law_type:law_ethnostate
		ruler ?= {
			OR = {
				has_ideology = ideology:ideology_fascist
				has_ideology = ideology:ideology_integralist
			}
		}
	}
}

	}
	ruler ?= {
		has_ideology = ideology:ideology_nihilist
    }
}

### UFB LARP ###

coa_UFB_trigger = {
	has_law_or_variant = law_type:law_republic
	has_law_or_variant = law_type:law_elected_bureaucrats
	has_law_or_variant = law_type:law_universal_suffrage
	#has_law_or_variant = law_type:law_multicultural
	has_law_or_variant = law_type:law_total_separation
	has_law_or_variant = law_type:law_protected_speech
	has_law_or_variant = law_type:law_guaranteed_liberties
	has_law_or_variant = law_type:law_no_colonial_affairs
	NOT = { has_law_or_variant = law_type:law_plurality }
	
	scope:actor ?= {
		NOT = {
			any_scope_state = {
				NOT = {
					state_region = {
						OR = {
							this = s:STATE_HOME_COUNTIES
							this = s:STATE_LANCASHIRE
							this = s:STATE_WALES
							this = s:STATE_YORKSHIRE
							this = s:STATE_HIGHLANDS
							this = s:STATE_MIDLANDS
							this = s:STATE_EAST_ANGLIA
							this = s:STATE_WEST_COUNTRY
							this = s:STATE_LOWLANDS
							this = s:STATE_ULSTER
							this = s:STATE_MUNSTER
							this = s:STATE_LEINSTER
							this = s:STATE_CONNAUGHT
						}
					}
				}
			}
		}
	}
}

coa_def_independent_trigger = {
	exists = scope:actor
	scope:actor ?= { is_subject = no }
}

coa_def_subject_trigger = {
	NOT = { has_game_rule = no_subject_flags }
	exists = scope:actor
	scope:actor ?= { is_subject = yes }
}

coa_def_lesser_subject_trigger = {
	NOT = { has_game_rule = no_subject_flags }
	OR = {
		NOT = { exists = scope:actor }
		AND = {
			exists = scope:actor
			scope:actor ?= {
				OR = {
					is_subject_type = subject_type_colony
					is_subject_type = subject_type_puppet
					is_subject_type = subject_type_vassal
				}
			}
		}
	}
}

coa_def_independence_war_trigger = { #checks if subject is in Independence diplomatic play/war
	exists = scope:actor
	scope:actor ?= {
		is_subject = yes
		OR = {
			play_participant_has_war_goal_of_type_against = {
				type = independence
				target = overlord
			}
			war_participant_has_war_goal_of_type_against = {
				type = independence
				target = overlord
			}
		}
	}
}

coa_def_ensign_trigger = {
	NOT = { has_game_rule = no_subject_flags }
	exists = scope:overlord
	scope:overlord ?= {
		OR = {
			this = scope:custom_overlord
			any_overlord_or_above ?= {
				this = scope:custom_overlord
			}
		}
	}
}
