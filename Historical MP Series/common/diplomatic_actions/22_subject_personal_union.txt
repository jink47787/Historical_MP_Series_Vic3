personal_union = {
	groups= {
		general # for breaking only
	}
	show_in_lens = no

	potential = { # Only created through diplomatic plays
		always = no
	}
	
	pact = {
		cost = 25
		has_junior_participant = yes
		maintenance_paid_by = first_country
		income_transfer = 0.05
		second_country_gets_income_transfer = no
		income_transfer_based_on_second_country = yes
		
		relations_progress_per_day = 1
		relations_improvement_max = 50

		max_target_involvement = 5000
		target_involvement_applies_to = source_country

		subject_type = subject_type_personal_union

		second_modifier = {
		}

		actor_can_break = {
			is_diplomatic_play_committed_participant = no
		}

		target_can_break = { # Has to use diplomatic play to break
			always = no
		}

		requirement_to_maintain = {
			trigger = {
				country_has_monarchy_law = yes
				NOT = { has_law = law_type:law_bakufu } # I have seen things
			}
		}

		requirement_to_maintain = {
			trigger = {
				scope:target_country = {
					country_has_monarchy_law = yes
					NOT = { has_law = law_type:law_bakufu }
				}
			}
		}

		manual_break_effect = {
			if = {
				limit = {
					NOT = {
						c:POR ?= THIS
						scope:target_country ?= {
							has_variable = pedro_republican_coup
						}
					}
				}
				create_bidirectional_truce = { country = scope:target_country months = 60 }
			}
		}

		auto_break_effect = {
			if = {
				limit = {
					NOT = {
						c:POR ?= THIS
						scope:target_country ?= {
							has_variable = pedro_republican_coup
						}
					}
				}
				create_bidirectional_truce = { country = scope:target_country months = 60 }
			}
		}
	}

	ai = {
		will_break = {
			always = no
		}
	}
}
