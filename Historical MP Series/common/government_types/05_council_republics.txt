# Council
REPLACE_OR_CREATE:gov_council_democracy_unitary = {
	transfer_of_power = parliamentary_elective
	new_leader_on_reform_government = yes

	male_ruler = "RULER_TITLE_COUNCILLOR"
	female_ruler = "RULER_TITLE_COUNCILLOR"
	
	possible = {
		has_law_or_variant = law_type:law_republic
		has_law_or_variant = law_type:law_council_republic
		has_law_or_variant = law_type:law_appointed_bureaucrats
		country_has_voting_franchise = yes
		NOT = { has_law_or_variant = law_type:law_vanguardism }
		NOT = {
			OR = {
				is_subject_type = subject_type_puppet
				is_subject_type = subject_type_dominion
			}
		}
	}
	
	on_government_type_change = {
		change_to_parliamentary_elective = yes
	}
	
	on_post_government_type_change = {
		post_change_to_parliamentary_elective = yes
	}
}
