REPLACE_OR_CREATE:gov_oligarchy = {
	transfer_of_power = dictatorial
	new_leader_on_reform_government = yes

	male_ruler = "RULER_TITLE_CHANCELLOR"
	female_ruler = "RULER_TITLE_CHANCELLOR"
	
	possible = {
		has_law_or_variant = law_type:law_republic
		has_law_or_variant = law_type:law_oligarchy 
		has_law_or_variant = law_type:law_parliamentary_system
		country_has_voting_franchise = no
		NOT = {
			OR = {
				is_subject_type = subject_type_puppet
				is_subject_type = subject_type_dominion
			}
		}
	}
	
	on_government_type_change = {
		change_to_dictatorial = yes
	}
	
	on_post_government_type_change = {
		post_change_to_dictatorial = yes
	}
}
