
### AFRICA
## SOUTHERN AFRICA
REPLACE_OR_CREATE:je_divide_and_rule_sotho_button = {
	name = "je_divide_and_rule_sotho_button"
	desc = "je_divide_and_rule_reason"
	visible = { OR = {
		capital.region = sr:region_southern_africa
		capital.state_region = { is_homeland = cu:sotho } }
	}
	selected = { has_modifier = secondary_culture_sotho }
	possible = {
		NOT = { has_modifier = modifier_has_shown_colonial_favouritism }
		has_technology_researched = civilizing_mission
		average_sol_for_culture = { target = cu:sotho	value >= 10 }
		any_scope_state = { state_region = { is_homeland = cu:sotho } }
	}
	effect = {
		add_modifier = secondary_culture_sotho
		add_modifier = modifier_has_shown_colonial_favouritism
		every_scope_culture = {
			limit = { shares_heritage_trait_group_with_any_primary_culture = prev }
			prev = { add_radicals = { culture = prev		value = 0.2 } }
		}
	}
	ai_chance = { value = 100 }
}
