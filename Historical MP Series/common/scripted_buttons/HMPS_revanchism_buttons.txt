REPLACE:je_danubian_federation_polish_button = {
	name = "je_danubian_federation_polish_button"
	desc = "je_danubian_federation_polish_button_desc"

	visible = {
		always = yes
	}

	selected = {
		cu:polish = {
			has_discrimination_trait = tradition_danubian
		}
	}

	ai_chance = {
		value = 10
	}

	possible = {
		custom_tooltip = { text = danubian_federation_started_tt has_variable = danubian_federation_started }
		custom_tooltip = { text = danubian_federation_button_cooldown_tt NOT = { has_variable = danubian_federation_button_cooldown } }
		hidden_trigger = { cu:polish = { custom_tooltip = { text = already_integrated_tt   ROOT = { NOT = { any_in_list = { variable = aus_integrated_cultures cu:polish ?= this } } } } } }
		cu:ukrainian = { custom_tooltip = { text = already_integrated_tt   ROOT = { NOT = { any_in_list = { variable = aus_integrated_cultures cu:ukrainian ?= this } } } } }
		OR = {
			average_sol_for_culture = { target = cu:polish value >= var:integrated_cultures_sol_requirement }
			any_subject_or_below = {
				average_sol_for_culture = { target = cu:polish value >= prev.var:integrated_cultures_sol_requirement }
			}
		}
	}

	effect = {
		hidden_effect = { #Revanchism addition: ->
			every_state = {
				limit = {
					OR = {
						state_region = s:STATE_EAST_GALICIA
						state_region = s:STATE_WEST_GALICIA
					}
					has_variable = austria_state_locked
				}
				remove_variable = austria_state_locked
			}
		} # <-
		cu:polish = {
			save_scope_as = temp_culture
			add_tradition_trait = tradition_danubian
		}
		set_variable = { name = danubian_federation_button_cooldown days = 180 } # Setting Cool Down, if you change the duration of this make sure to change the loc (danubian_federation_button_cooldown_tt) for it as well
		add_to_variable_list = { name = aus_integrated_cultures target = scope:temp_culture }
		change_variable = { name = count_integrated_cultures add = 1 }
		change_variable = { name = integrated_cultures_sol_requirement add = var:integrated_cultures_sol_requirement_increase_per_culture }
		scope:temp_culture = { custom_tooltip = incorporate_any_this_culture_states_including_in_ai_subjects_tt }
		hidden_effect = {
			every_subject_or_below = {
				every_scope_state = {
					limit = { state_region = { is_homeland = scope:temp_culture } }
					if = { limit = { owner = { is_player = no } }
						set_state_owner = root
					}
					set_state_type = incorporated
				}
			}
			every_scope_state = {
				limit = { state_region = { is_homeland = scope:temp_culture } is_incorporated = no }
				set_state_type = incorporated
			}
		}
        add_radicals = { culture = cu:south_german value = 0.1 }
        add_loyalists = { culture = scope:temp_culture value = 0.1 }
		every_subject_or_below = { ### giving subjects the radicals and loyalists as well
			add_radicals = { culture = cu:south_german value = 0.1 }
			add_loyalists = { culture = scope:temp_culture value = 0.1 }
		}
	}

	ai_chance = {
		value = 100
	}
}
