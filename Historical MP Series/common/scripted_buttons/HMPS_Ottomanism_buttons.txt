je_ottomanism_greeks = {
	name = {
		first_valid = {
			triggered_desc = {
				desc = "je_ottomanism_intergrate_greeks"
				trigger = {
					NOT = {
						has_variable = greeks_integrated
					}
				}
			}
			triggered_desc = {
				desc = "je_ottomanism_intergrate_greeks_already"
				trigger = {
					has_variable = greeks_integrated
				}
			}
		}
	}
	desc = "je_ottomanism_greeks_desc"

	visible = { always = yes }

	possible = {

		# requirements here
		calc_true_if = {
			amount >= 14
			any_state_region = {
				is_homeland = cu:greek
			}
		}
		has_law = law_type:law_total_separation
		NOT = { has_law = law_type:law_hereditary_bureaucrats }

		hidden_trigger = {
			NOT = {
				has_variable = greeks_integrated
			}
		}
	}

	effect = {
		trigger_event = { id = ottomanism_events.1 }
		hidden_effect = {
			set_variable = {
				name = greeks_integrated
			}
		}
	}
}
