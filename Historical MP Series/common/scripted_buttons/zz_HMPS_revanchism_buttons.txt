	name = "hmps_revanchism_gain_claims_button"
	desc = "hmps_revanchism_gain_claims_button_desc"

	visible = {

			filter = {
				is_homeland_of_country_cultures = ROOT

							is_indirect_subject_of = ROOT
						}
					}
				}
			}
			count >= 1
			has_modifier = modifier_hmps_revanchism_target
			NOT = {
				has_claim_by = ROOT
			}
		}
	}

	possible = {
		always = yes
	}

	effect = {
		every_state = {
			limit = {

				#	}
