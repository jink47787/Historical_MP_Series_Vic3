	name = austrian_neo_absolutism_papal_concordat_button
	desc = austrian_neo_absolutism_papal_concordat_button_desc

		NOT = {
			has_variable = austrian_neo_absolutism_papal_concordat_var

	}

			limit = {
				exists = c:PAP
			}
			change_relations = {
				country = c:PAP
				value = 40
			}
			if = {
				limit = {
					is_power_bloc_leader = yes
				}

					add_leverage = { target = c:PAP value = 100 }
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_law = law_type:law_state_religion
				}
			}
			activate_law = law_type:law_state_religion
		}
		#if = {
		#	limit = {
		#		NOT = {
		#			has_law = law_type:law_religious_schools
		#		}
		#	}
		#	activate_law = law_type:law_religious_schools
		#}
		add_loyalists = {
			religion = rel:catholic
			value = medium_radicals
		}
		add_radicals = {
			religion = rel:protestant
			value = small_radicals
		}

	ai_chance = {
		value = 1
	}
}


REPLACE:je_danubian_federation_hungarian_button = {
	name = "je_danubian_federation_hungarian_button"
	desc = "je_danubian_federation_hungarian_button_desc"

	visible = {
		always = yes
	}

	selected = {
		cu:szekely = {
			has_discrimination_trait = tradition_danubian
		}
	}

	ai_chance = {
		value = 10
	}

	possible = {
		custom_tooltip = { text = danubian_federation_started_tt has_variable = danubian_federation_started }
		custom_tooltip = { text = danubian_federation_button_cooldown_tt NOT = { has_variable = danubian_federation_button_cooldown } }
		hidden_trigger = { cu:hungarian = { custom_tooltip = { text = already_integrated_tt   ROOT = { NOT = { any_in_list = { variable = aus_integrated_cultures cu:hungarian ?= this } } } } } }
		cu:romanian = { custom_tooltip = { text = already_integrated_tt   ROOT = { NOT = { any_in_list = { variable = aus_integrated_cultures cu:romanian ?= this } } } } }
		#cu:slovak = { custom_tooltip = { text = already_integrated_tt   ROOT = { NOT = { any_in_list = { variable = aus_integrated_cultures cu:slovak ?= this } } } } }
		c:HUN ?= { is_player = no }
		OR = {
			average_sol_for_culture = { target = cu:hungarian value >= var:integrated_cultures_sol_requirement }
			any_subject_or_below = {
				average_sol_for_culture = { target = cu:hungarian value >= prev.var:integrated_cultures_sol_requirement }
			}
		}
	}

	effect = {
		if = { limit = { c:HUN ?= { is_subject_of = root } }
			annex_with_incorporation = c:HUN
		}
		je:je_danubian_federation = {
			remove_modifier = modifier_HMPS_danubian_federation_hungarian_negative_acceptance
			add_modifier = modifier_HMPS_danubian_federation_hungarian_option_buff
		}
		cu:szekely = { add_tradition_trait = tradition_danubian }
		cu:hungarian = {
			save_scope_as = temp_culture
			#add_tradition_trait = tradition_danubian
		}
		set_variable = { name = danubian_federation_button_cooldown days = 180 } # Setting Cool Down, if you change the duration of this make sure to change the loc (danubian_federation_button_cooldown_tt) for it as well
		add_to_variable_list = { name = aus_integrated_cultures target = scope:temp_culture }
		change_variable = { name = count_integrated_cultures add = 1 }
		change_variable = { name = integrated_cultures_sol_requirement add = var:integrated_cultures_sol_requirement_increase_per_culture }
		scope:temp_culture = { custom_tooltip = incorporate_any_this_culture_states_including_in_ai_subjects_tt }
		hidden_effect = {
			every_subject_or_below = {
				every_scope_state = {
					limit = { state_region = { is_homeland = scope:temp_culture } }
					if = { limit = { owner = { is_player = no } }
						set_state_owner = root
					}
					set_state_type = incorporated
				}
			}
			every_scope_state = {
				limit = { state_region = { is_homeland = scope:temp_culture } is_incorporated = no }
				set_state_type = incorporated
			}
		}
        add_radicals = { culture = cu:south_german value = 0.05 }
        add_loyalists = { culture = scope:temp_culture value = 0.05 }
		every_subject_or_below = { ### giving subjects the loyalists as well
			add_loyalists = { culture = scope:temp_culture value = 0.05 }
		}
	}

	ai_chance = {
		value = 100
	}
}
