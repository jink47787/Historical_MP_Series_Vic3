# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law

law_no_legislature = {
	group = lawgroup_cameralism
	
	icon = "gfx/interface/icons/law_icons/no_legislature.dds"
	
	progressiveness = -25

	disallowing_laws = {
		law_anarchy
		law_landed_voting
		law_wealth_voting
		law_census_voting
		law_universal_suffrage
	}

	modifier = {
		country_authority_add = 100
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

law_unicameralism = {
	group = lawgroup_cameralism
	
	icon = "gfx/interface/icons/law_icons/unicameral.dds"
	
	progressiveness = 0

	disallowing_laws = {
		law_vanguardism
		law_autocracy
		law_oligarchy
		law_technocracy
	}

	modifier = {
		country_law_enactment_time_mult = 0.05
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
	
	on_activate = {
		if = {
			limit = {
				OR = {
					has_journal_entry = je_napoleons_legacy
					overlord = {
						has_journal_entry = je_napoleons_legacy
					}
				}
				has_global_variable = global_1848_progress
			}
			change_global_variable = {
				name = global_1848_progress
				add = 3
			}
		}
	}
}
