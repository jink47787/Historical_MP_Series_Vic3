law_hindu_caste_enforced = {
	group = lawgroup_caste_hegemony
	
	icon = "gfx/interface/icons/law_icons/social_hierarchy_enforced.dds"
	
	progressiveness = -75
	
	is_visible = {
		OR = {
			has_law_or_variant = law_type:law_hindu_caste_enforced
			AND = {
				is_in_geographic_region = geographic_region_india
				religion_percent_country = {
					target = rel:hindu
					value >= 0.50
				}
			}
		}
	}

	can_enact = {
		OR = {
			has_law_or_variant = law_type:law_hindu_caste_enforced
			AND = {
				is_in_geographic_region = geographic_region_india
				religion_percent_country = {
					target = rel:hindu
					value >= 0.50
				}
			}
		}
	}
	
	on_enact = {
		recalculate_pop_ig_support = yes
	}

	modifier = {
		state_bureaucracy_population_base_cost_factor_mult = -0.5
		interest_group_ig_landowners_pol_str_mult = 0.5

		country_brahmins_qualification_growth_same_class_mult = -0.80
		country_kshatriyas_qualification_growth_same_class_mult = -0.80
		country_vaishyas_qualification_growth_same_class_mult = -0.80
		country_shudras_qualification_growth_same_class_mult = -0.80
		country_dalit_qualification_growth_same_class_mult = -0.80

		country_kshatriyas_qualification_growth_other_class_mult = -0.95
		country_vaishyas_qualification_growth_other_class_mult = -0.95
		country_shudras_qualification_growth_other_class_mult = -0.95
		country_dalit_qualification_growth_other_class_mult = -0.95
		
		country_brahmins_acceptance_min_add = 80
		country_kshatriyas_acceptance_min_add = 60
		country_vaishyas_acceptance_min_add = 40
		country_shudras_acceptance_min_add = 20
		
		
		country_kshatriyas_acceptance_max_add = 60
		country_vaishyas_acceptance_max_add = 40
		country_shudras_acceptance_max_add = 20
		country_dalit_acceptance_max_add = 0.001
		
		country_kshatriyas_education_access_mult = -0.10 #THIS IS MULTIPLICITIVE, ie, -0.25 would take a pop from 40% education access to 30% education access
		country_vaishyas_education_access_mult = -0.20
		country_shudras_education_access_mult = -0.30
		country_dalit_education_access_mult = -0.40
		
		country_brahmins_qualification_growth_other_class_mult = -0.99
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}
