law_no_police = {
	group = lawgroup_policing
	
	icon = "gfx/interface/icons/law_icons/no_police.dds"
	
	progressiveness = 0
	
	can_enact = {
		custom_tooltip = {
			modifier:country_disallow_law_no_police_bool = no
			text = disallow_law_no_police_tt
		}
	}

	on_activate = {
		if = {
			limit = {
				has_journal_entry = je_army_reform_china
			}
			add_modifier = modifier_green_standard_army
		}
	}
	
	# AI shouldn't try to abolish institutions
	ai_will_do = {
		always = no
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

law_local_police = {
	group = lawgroup_policing
	
	icon = "gfx/interface/icons/law_icons/local_police.dds"
	
	unlocking_technologies = {
		tech_bureaucracy
	}
	
	modifier = {
		country_authority_mult = 0.05
		country_institution_cost_institution_police_mult = -0.50
	}
	
	progressiveness = -50
	
	disallowing_laws = {
		law_anarchy
	}

	institution = institution_police
	institution_modifier = {
		interest_group_ig_landowners_pol_str_mult = 0.025
		state_turmoil_effects_mult = -0.05
	}

	on_activate = {
		if = {
			limit = {
				has_modifier = modifier_green_standard_army
			}
			remove_modifier = modifier_green_standard_army
		}
	}
	
	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_policing_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}
	
	# Don't return to local police if we have something more advanced
	ai_will_do = {
		NOR = {
			has_law_or_variant = law_type:law_dedicated_police
			has_law_or_variant = law_type:law_militarized_police
		}
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = {
				has_law_or_variant = law_type:law_no_police
			}
			add = 500
		}
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		
		if = {
			limit = {
				has_journal_entry = je_hawaii
			}
			add = 10000
		}
	}
}
