law_serfdom = { #HMPS
	group = lawgroup_land_reform
	
	icon = "gfx/interface/icons/law_icons/serfdom.dds"
	
	progressiveness = -50

	is_visible = {
		NOR = {
			c:AUS ?= this # Austria and the Danubian Principalities have manorialism instead
			AND = {
				exists = c:AUS
				is_subject_of = c:AUS
				OR = {
					is_subject_type = subject_type_crown_land
					is_subject_type = subject_type_personal_union
				}
				c:AUS ?= { has_law = law_type:law_manorialism }
			}
			country_has_primary_culture = cu:romanian
		}
	}
	
	disallowing_laws = {
		law_interventionism
		law_laissez_faire
		law_command_economy
		law_cooperative_ownership
	}
	
	on_enact = {
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_commercialized_agriculture
					has_law_or_variant = law_type:law_homesteading
				}
			}
			custom_tooltip = farmers_pb_ig_shift_effect_negative_tt
		}
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_commercialized_agriculture
					has_law_or_variant = law_type:law_collectivized_agriculture
				}
			}
			custom_tooltip = rural_laborers_tu_ig_shift_effect_negative_tt
		}
	}
	
	modifier = {
		interest_group_ig_landowners_pol_str_mult = 0.5
		state_birth_rate_mult = 0.05
		
		state_peasants_internal_migration_disallowed_bool = yes
		state_peasants_mass_migration_disallowed_bool = yes
		state_minimum_incorporated_subsistence_arable_land_add = 0.75
		building_minimum_incorporated_subsistence_employment_add = 0.5
		state_market_access_price_impact = -0.05
	}
	
	# AI should never try to bring back serfdom
	ai_will_do = {
		always = no
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

law_tenant_farmers = {
	group = lawgroup_land_reform
	
	icon = "gfx/interface/icons/law_icons/tenant_farmers.dds"
	
	progressiveness = 0

	is_visible = {
		NOR = {
			has_law = law_type:law_latifundias
			has_law = law_type:law_expanded_latifundias
		}
	}

	disallowing_laws = {
		law_command_economy
		law_cooperative_ownership
	}

	modifier = {
		interest_group_ig_landowners_pol_str_mult = 0.25
		state_peasants_internal_migration_disallowed_bool = yes
	}
	
	on_enact = {
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_commercialized_agriculture
					has_law_or_variant = law_type:law_homesteading
				}
			}
			custom_tooltip = farmers_pb_ig_shift_effect_negative_tt
		}
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_commercialized_agriculture
					has_law_or_variant = law_type:law_collectivized_agriculture
				}
			}
			custom_tooltip = rural_laborers_tu_ig_shift_effect_negative_tt
		}
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		
		# Once the AI gets to a certain society tech level they should try to ditch Serfdom for Tenant Farmers
		if = {
			limit = {
				OR = {
					has_technology_researched = civilizing_mission
					has_technology_researched = human_rights
					has_technology_researched = pan-nationalism
				}
				has_law_or_variant = law_type:law_serfdom
			}

			add = 1000
		}
		
		if = {
			limit = {
				 OR = {
					has_journal_entry = je_meiji_main
					has_journal_entry = je_great_reforms_serfdom
				}
				has_law_or_variant = law_type:law_serfdom
			}
			add = 10000 # Top priority for Meiji AI is to get out of Serfdom, likewise for Great Reforms
		}

		# AI should really try to get out of Serfdom under the right circumstances
		if = {
			limit = {
				has_law_or_variant = law_type:law_serfdom
				NOT = { has_strategy = ai_strategy_reactionary_agenda }
				OR = {
					NOT = { has_strategy = ai_strategy_conservative_agenda }
					any_political_movement = {
						law_enactment_stance = {
							law = law_type:law_tenant_farmers
							value > neutral
						}
						political_movement_support >= 0.10
					}
				}
			}
			add = 1000
		}		
	}
}
