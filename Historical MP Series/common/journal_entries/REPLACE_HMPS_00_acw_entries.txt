REPLACE:je_acw_reincorporate = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"
	
	group = je_group_usa_american_civil_war

	immediate = {
		every_scope_state = {
			limit = {
				state_region = {
					is_homeland = cu:dixie
				}
				culture_percent_state = {
					target = cu:dixie
					value > 0
				}
			}
			root = {
				add_to_variable_list = {
					name = acw_dixie_states
					target = prev
				}
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = "ACW_REINCORPORATE_TT"

			any_scope_state = {
				filter = {
					state_region = {
						is_homeland = cu:dixie
					}
					culture_percent_state = {
						target = cu:dixie
						value > 0
					}
				}
				count = all
				is_incorporated = yes
				loyalist_fraction = {
					value >= acw_reincorporate_dixie_loyalist_min # defined in event_values
					culture = cu:dixie
				}
			}
		}
	}

	on_complete = {
		clear_variable_list = acw_dixie_states
		custom_tooltip = {
			text = gain_dixie_culture
			set_variable = acw_gain_dixie
		}
		custom_tooltip = {
			text = reconstruction_progress
			change_variable = {
				name = reconstruction_var
				add = 1
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			180 = 0
			20 = acw_followup_events.6
			20 = acw_followup_events.7
			20 = acw_followup_events.8
		}
	}

	on_weekly_pulse = {
		effect = {
			clear_variable_list = acw_dixie_states
			if = {
				limit = {
					any_scope_state = {
						state_region = {
							is_homeland = cu:dixie
						}
						culture_percent_state = {
							target = cu:dixie
							value > 0
						}
					}
				}
				every_scope_state = {
					limit = {
						state_region = {
							is_homeland = cu:dixie
						}
						culture_percent_state = {
							target = cu:dixie
							value > 0
						}
					}
					root = {
						add_to_variable_list = {
							name = acw_dixie_states
							target = prev
						}
					}
				}
			}
		}
	}

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = {
					is_homeland = cu:dixie
				}
			}
		}
	}

	on_invalid = {
		clear_variable_list = acw_dixie_states
	}

	timeout = 6570 #HMPS: Increased time by 50% to get Dixie

	on_timeout = {
		clear_variable_list = acw_dixie_states
	}
	
	should_be_pinned_by_default_uninvolved_or_context = yes
}
