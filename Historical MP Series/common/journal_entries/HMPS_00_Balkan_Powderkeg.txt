je_balkan_powderkeg = {

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		exists = c:TUR
		c:TUR = this
    }

	possible = {
		exists = c:TUR
		#c:TUR = this
		has_technology_researched = nationalism
		any_scope_state = {
			region = sr:region_balkans
    	}
	}

	modifiers_while_active = {
		#modifier_balkan_seperatism
	}

	immediate = {
		
	}

	complete = {
		any_scope_state = {
			OR = {
				region = sr:region_balkans
				region = sr:region_danubia
			}
			is_split_state = no
			custom_tooltip = {
				text = turkish_40_percent
				culture_percent_state = {
					target = cu:turkish
					value >= 0.4
				}
			}
			count >= 10
		}
	}

	on_complete = {

	}

	fail = {
		any_scope_state = {
			OR = {
				region = sr:region_balkans
				region = sr:region_danubia
			}
			count < 3
		}
	}

	on_fail = {

	}

	on_monthly_pulse = {	

		effect = {
			every_scope_state = {
				limit = {
					NOT = { has_modifier = modifier_balkan_seperatism }
				}
				add_modifier = {
					name = modifier_balkan_seperatism
				}
			}
		}

		random_events = {
			11 = 0
			1 = mansur_slaving.1
		}

	}
	

	weight = 100
	should_be_pinned_by_default = yes
}
