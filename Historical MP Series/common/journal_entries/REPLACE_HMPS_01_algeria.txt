#Probably needs a long term rework once assilimation/cultural acceptance is fixed. Goal should be to settle French pops in Algeria
REPLACE:je_the_algerian_departements = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_france_expansion

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_ALGIERS
			}
			save_scope_as = je_algiers_state_scope
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_ORAN
			}
			save_scope_as = je_oran_state_scope
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_CONSTANTINE
			}
			save_scope_as = je_constantine_state_scope
		}
		cu:berber = {
			save_scope_as = berber_culture_scope
		}
		cu:maghrebi = {
			save_scope_as = maghrebi_culture_scope
		}
		cu:french = {
			save_scope_as = french_culture_scope
		}
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_ORAN
					state_region = s:STATE_ALGIERS
					state_region = s:STATE_CONSTANTINE
				}
			}
			add_modifier = {
				name = crushed_algerian_resistance
				days = very_long_modifier_time
			}
		}
	}
	complete = {
		custom_tooltip = { #HMPS: Removes incorporation requrements and increases turmoil limit from 0.1
			text = je_the_algerian_departements_tt_1
			any_scope_state = {
				state_region = s:STATE_ALGIERS
				turmoil < 0.26
				is_split_state = no
			}
			any_scope_state = {
				state_region = s:STATE_ORAN
				turmoil < 0.26
				is_split_state = no
			}
			any_scope_state = {
				state_region = s:STATE_CONSTANTINE
				turmoil < 0.26
				is_split_state = no
			}
		}
		#HMPS: Removed other completion triggers as they are basically impossible to do in HMPS
	}

	on_complete = { #HMPS: Gives Algeria french homelands on completion
		s:STATE_ALGIERS = { #HMPS: ->
			add_homeland = cu:french
		}
		s:STATE_ORAN = {
			add_homeland = cu:french
		}
		s:STATE_CONSTANTINE = {
			add_homeland = cu:french
		} #HMPS: <-
		trigger_event = algeria_events.4
	}

	fail = {
		NOT = { has_state_in_state_region = STATE_ORAN }
		NOT = { has_state_in_state_region = STATE_ALGIERS }
		NOT = { has_state_in_state_region = STATE_CONSTANTINE }
	}

	on_fail = {
		trigger_event = algeria_events.3
	}

	on_monthly_pulse = {
		random_events = {
			90 = 0
			10 = algeria_events.6
		}
		effect = {
			every_scope_state = { # In case the states are lost as a result of a secession or the like
				limit = {
					OR = {
						state_region = s:STATE_ORAN
						state_region = s:STATE_ALGIERS
						state_region = s:STATE_CONSTANTINE
					}
					is_incorporated = no
					NOT = {
						has_modifier = crushed_algerian_resistance
					}
				}
				add_modifier = {
					name = crushed_algerian_resistance
					days = very_long_modifier_time
				}
			}
		}
	}

	transferable = no

	should_be_pinned_by_default_uninvolved_or_context = yes

	weight = 5000
}
