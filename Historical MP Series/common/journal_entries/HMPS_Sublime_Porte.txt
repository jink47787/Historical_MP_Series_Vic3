	}
	
	possible = {
		c:TUR = {
			custom_tooltip = {
				text = ottoman_fed_tt.a
				has_variable = sick_man_complete_success
			}
		}
	}
	
	complete = {
		has_law_or_variant = law_type:law_elected_bureaucrats
		OR = {
			has_law_or_variant = law_type:law_total_separation
			#has_law_or_variant = law_type:law_state_atheism
		}
		OR = { has_law_or_variant = law_type:law_census_voting has_law_or_variant = law_type:law_universal_suffrage has_law_or_variant = law_type:law_anarchy }
		NOT = { has_law_or_variant = law_type:law_independent_executive } # "Parliamentary-in-nature"
		custom_tooltip = {
            text = ottoman_fed_tt.b
			owns_entire_state_region = STATE_BAGHDAD
			owns_entire_state_region = STATE_GREATER_SYRIA
			owns_entire_state_region = STATE_TRANSJORDAN

		}
		has_technology_researched = pan-nationalism

		cu:mashriqi = {
			add_tradition_trait = tradition_rumelian
		}
		if = {
			limit = {
				owns_entire_state_region = STATE_NORTHERN_THRACE
				owns_entire_state_region = STATE_BULGARIA
			}
			cu:bulgarian = {
				add_tradition_trait = tradition_rumelian
			}
		}
		if = {
			limit = {
				owns_entire_state_region = STATE_DIYARBAKIR

			}
			cu:kurdish = {
				add_tradition_trait = tradition_rumelian
			}
			cu:armenian = {
				add_tradition_trait = tradition_rumelian
			}
		}
		if = {
			limit = {
				has_variable = completed_tanzimat_greece
			}
			cu:greek = {
				add_tradition_trait = tradition_rumelian
			}
		}
		

		
		#add_primary_culture = cu:mashriqi
		custom_tooltip = {
			text = ottoman_fed_tt.c
		}
	}

	can_revolution_inherit = yes
	should_be_pinned_by_default = yes
	weight = 5000
}
