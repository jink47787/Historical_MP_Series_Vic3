
je_lower_canada = {

	icon = "gfx/interface/icons/event_icons/event_military.dds"
	
	group = je_group_historical_content

	possible = {
		this ?= c:QUE
	}

	modifiers_while_active = {
		modifier_chateau_clique
	}

	complete = {
		OR = {
			culture_percent_country = {
				target = cu:franco_canadian
				value <= 0.5
			}
			religion_percent_country = {
				target = rel:catholic
				value <= 0.5
			}
			game_date >= 1850.1.1
		}
	}

	fail = {
		OR = {
			c:QUE = {
				liberty_desire >= 95
			}
			s:STATE_QUEBEC = {
				region_state:QUE = {
					turmoil >= 0.6
				}
			}
			c:QUE = {
				any_political_movement = {
					is_political_movement_type = movement_cultural_minority
					political_movement_support >= 0.5 
					political_movement_support >= 0.5
				}
			}
		}
	}

	on_monthly_pulse = {
		
	}


	on_complete = {
		add_primary_culture = cu:franco_canadian
	}

	on_fail = {
		c:QUE = {
			capital = {
				save_scope_as = que_capital_scope
			}
		}
		c:QUE = {
			remove_primary_culture = cu:anglo_canadian
			add_primary_culture = cu:franco_canadian
			set_state_religion = rel:catholic
			add_liberty_desire = 100

			create_diplomatic_play = {
				target_country = c:GBR
				escalation = 80
				type = dp_independence
				
				target_state = scope:que_capital_scope
		
				add_war_goal = {
					holder = c:GBR
					type = reduce_autonomy
					target_country = c:QUE
				}

				#add_war_goal = {
				#	holder = c:QUE
				#	type = humiliation
				#	target_country = c:GBR
				#}
			}
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes

}
