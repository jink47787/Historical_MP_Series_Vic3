#global JE for europeans to track 1848 revolutions

REPLACE_OR_CREATE:je_napoleons_legacy = { #The Political Question
	icon = "gfx/interface/icons/event_icons/phrygian.dds"

	group = je_group_global_international_situations
	
	context = none
	progressbar = yes
	should_be_pinned_by_default = yes

	#things that cause this to increase (non laws are once per country) (this is outdated but kept for reference)
		#laws
			#any european major passing voting laws (scaling)
			#any european major passing parliamentary or dual executiy (+3)
			#any european major passing total_separation (+2)
			#any european major passing state atheism (+5)
			#any european major passing republic (+10)
		#Italy or Germany forming (+5)
		#a revolution in any european major+ country or subject of one (+8)
		#having turmoil in a European capital (+3)
		#the Hungry Forties ending (+5)
		#a European losing a war (+5)

	should_be_involved = { #Initial Involved Nations
				OR = {
					capital.region = sr:region_england
					capital.region = sr:region_north_sea_coast
					capital.region = sr:region_france
					capital.region = sr:region_occitania
					capital.region = sr:region_rhine
					capital.region = sr:region_italy
					capital.region = sr:region_balkans
					capital.region = sr:region_north_germany
					capital.region = sr:region_south_germany
					capital.region = sr:region_baltic
					capital.region = sr:region_danubia
					capital.region = sr:region_poland
				}
			country_rank >= rank_value:major_power
			is_subject = no
			OR = {
				OR = {
					has_law_or_variant = law_type:law_autocracy
					has_law_or_variant = law_type:law_oligarchy
					has_law_or_variant = law_type:law_technocracy
					has_law_or_variant = law_type:law_landed_voting
				}
				NOT = { has_law_or_variant = law_type:law_right_of_assembly }
			}
	}
	
	should_show_when_not_involved = { #Lets all of Europe see the Political Question Progress
		is_in_geographic_region = geographic_region_europe
	}

	is_shown_when_inactive = {
		always = yes
	}
	
	possible = {
	}
	
	immediate = {
		if = {
			limit = {
				NOT = { has_global_variable = global_1848_progress }
			}
			set_global_variable = {
				name = global_1848_progress
				value = 0
			}
		}
		if = {
			limit = {
				NOT = { has_global_variable = global_1848_progress }
			}
			set_global_variable = {
				name = 1848_radical_leader_mult_var
				value = 1
			}
		}
	}

	immediate_all_involved = {
		set_variable = 1848_country_var
	}

	on_become_involved_after_activation = {
		set_variable = 1848_country_var
	}
	
	on_no_longer_involved = { #Avoiding Springtime Effects
		#if = {
		#	limit = {
		#		country_rank >= rank_value:major_power
		#	}
			set_variable = passed_1848_var
			add_modifier = {
				name = modifier_1848_victory
				months = 60
			}
		#}
	}

	current_value = {
		value = global_var:global_1848_progress
	}

	goal_add_value = {
		value = 40
	}

	invalid = { #Removed it looking for global variables to invalidate to limit chances of it accidentally invalidating
		game_date >= 1856.1.1
	}
	
	on_invalid = {
	}

	timeout = 7300
	
	on_timeout = {
		set_global_variable = peoples_springtime_happened
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		if = {
			limit = {
				NOT = { has_global_variable = peoples_springtime_happening }
			}
			set_global_variable = peoples_springtime_happening
		}
	}

	on_complete_all_involved = { #Springtime hitting
		set_variable = failed_1848_var
		add_journal_entry = {
			type = je_springtime_of_the_peoples
		}
		ordered_interest_group = {
			limit = {
				ig_counts_as_marginal = no
				can_have_radical_leader = yes
			}
			order_by = ig_clout
			position = 0
			check_range_bounds = no
			leader = {
				set_ideology = ideology:ideology_radical
				add_modifier = {
					name = 1848_popular_radical
					days = normal_modifier_time
					is_decaying = yes
				}
			}
		}
		custom_tooltip = je_napoleons_legacy_t1
		if = {
			limit = {
				c:FRA ?= this
			}
			show_as_tooltip = {
				add_radicals = {
					value = medium_radicals
				}
			}
			hidden_effect = {
				every_country = {
					limit = {
						has_modifier = modifier_napoleon_announce
					}
					remove_modifier = modifier_napoleon_announce
				}
				every_scope_state = {
					limit = {
						is_incorporated = yes
					}
					add_radicals_in_state = {
						value = medium_radicals
					}
				}
			}
		}
		else = {
			show_as_tooltip = {
				add_radicals = {
					value = large_radicals
				}
			}
			hidden_effect = {
				every_country = {
					limit = {
						has_modifier = modifier_napoleon_announce
					}
					remove_modifier = modifier_napoleon_announce
				}
				every_scope_state = {
					limit = {
						is_incorporated = yes
					}
					add_radicals_in_state = {
						value = very_large_radicals
					}
				}
			}
		}
	}

	on_weekly_pulse = {
		effect = {
			scope:journal_entry = { #Check for removing involved countries who shouldn't get hit with Springtime
				every_scope_je_involved = {
					limit = {
						NOR = {
							has_law_or_variant = law_type:law_autocracy
							has_law_or_variant = law_type:law_oligarchy
							has_law_or_variant = law_type:law_technocracy
							has_law_or_variant = law_type:law_landed_voting
							has_law_or_variant = law_type:law_traditionalism
							has_law_or_variant = law_type:law_serfdom
							has_law_or_variant = law_type:law_secret_police
							has_law_or_variant = law_type:law_militarized_police
						}
						has_law_or_variant = law_type:law_right_of_assembly
						NOT = { has_journal_entry = je_divided_monarchists }
					}
					scope:journal_entry = { remove_involved_country	= prev }
				}
			}
			every_country = { #Check for adding countries who meet the requirements of the Political Question after start
				limit = {
					OR = {
						capital.region = sr:region_england
						capital.region = sr:region_north_sea_coast
						capital.region = sr:region_france
						capital.region = sr:region_occitania
						capital.region = sr:region_rhine
						capital.region = sr:region_italy
						capital.region = sr:region_balkans
						capital.region = sr:region_north_germany
						capital.region = sr:region_south_germany
						capital.region = sr:region_baltic
						capital.region = sr:region_danubia
						capital.region = sr:region_poland
					}
					OR = {
						OR = {
							has_law_or_variant = law_type:law_autocracy
							has_law_or_variant = law_type:law_oligarchy
							has_law_or_variant = law_type:law_technocracy
							has_law_or_variant = law_type:law_landed_voting
						}
						NOT = { has_law_or_variant = law_type:law_right_of_assembly }
					}
					country_rank >= rank_value:major_power
					is_subject = no
					NOT = {
						has_variable = passed_1848_var
					}
				}
				scope:journal_entry = { add_involved_country = prev }		
			}	
		}
	}				
}
