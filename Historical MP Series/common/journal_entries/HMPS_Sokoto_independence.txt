je_sokoto_independence = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"
	
	group = je_group_historical_content
	
    possible = {

    }
	
	immediate = {
		this = {
			save_scope_as = new_overlord_sok
		}
	}
	
	complete = {
		this = {
			is_country_type = recognized
		}
	}
	
	fail = {
		OR = {
			is_subject = yes
			custom_tooltip = {
				text = gdp_ownership_sok_loc
				"gdp_ownership_ratio(this)" <= 0.8
			}
		}
	}
	
	on_monthly_pulse = {
		
	}
	
	on_complete = {

	}

	on_fail = {
		every_country = {
			limit = {
				NOT = {
					THIS = ROOT
				}
				OR = {
					ROOT = {
						"gdp_ownership_ratio(prev)" > 0.1
					}	
					overlord = ROOT
				}
			}
			save_scope_as = new_overlord_sok
		}



		this = {
			remove_primary_culture = cu:hausa
			remove_primary_culture = cu:fulbe
			activate_law = law_type:law_extraction_economy
			activate_law = law_type:law_cultural_exclusion
			
			if = {
				limit = {
					is_subject = no
				}
				create_diplomatic_pact = {
					country = scope:new_overlord_sok.country
					type = colony
				}
			}

			if = {
				limit = {
					overlord = {
						country_has_primary_culture = cu:british
					}
				}
				add_primary_culture = cu:british
				add_primary_culture = cu:scottish
			}
			else_if = {
				limit = {
					overlord = {
						OR = {
							country_has_primary_culture = cu:south_german
							country_has_primary_culture = cu:north_german
						}
					}
				}
				add_primary_culture = cu:south_german
				add_primary_culture = cu:north_german
			}
			else_if = {
				limit = {
					overlord = {
						country_has_primary_culture = cu:french
					}
				}
				add_primary_culture = cu:french
			}
			else_if = {
				limit = {
					overlord = {
						OR = {
							country_has_primary_culture = cu:south_italian
							country_has_primary_culture = cu:north_italian
						}
					}
				}
				add_primary_culture = cu:north_italian
				add_primary_culture = cu:south_italian
			}
			else_if = {
				limit = {
					overlord = {
						country_has_primary_culture = cu:wallonian
					}
				}
				add_primary_culture = cu:flemish
				add_primary_culture = cu:wallonian
			}
			else_if = {
				limit = {
					overlord = {
						country_has_primary_culture = cu:dutch
					}
				}
				add_primary_culture = cu:dutch
			}
			else_if = {
				limit = {
					overlord = {
						country_has_primary_culture = cu:spanish
					}
				}
				add_primary_culture = cu:spanish
			}
			else_if = {
				limit = {
					overlord = {
						country_has_primary_culture = cu:yankee
					}
				}
				add_primary_culture = cu:yankee
			}
			else_if = {
				limit = {
					overlord = {
						country_has_primary_culture = cu:russian
					}
				}
				add_primary_culture = cu:russian
			}
			else_if = {
				limit = {
					overlord = {
						country_has_primary_culture = cu:swedish
					}
				}
				add_primary_culture = cu:swedish
			}
			else_if = {
				limit = {
					overlord = {
						country_has_primary_culture = cu:japanese
					}
				}
				add_primary_culture = cu:japanese
			}
			else_if = {
				limit = {
					overlord = {
						country_has_primary_culture = cu:turkish
					}
				}
				add_primary_culture = cu:turkish
				add_primary_culture = cu:fulbe
			}
			else = {
				add_primary_culture = cu:han
			}


			
			
		}
	}
	
	progressbar = no
	weight = 10000
	should_be_pinned_by_default = yes

}
