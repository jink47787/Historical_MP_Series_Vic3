je_sokoto_independence = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"
	
	group = je_group_historical_content
	
	#
	
	is_shown_when_inactive = {
		exists = c:SOK
		c:SOK = this
    }
	
    possible = {
		exists = c:SOK
		c:SOK = this
    }
	
	immediate = {
		s:STATE_EQUATORIA = {
			save_scope_as = equatoria_scope
		}
	}
	
	complete = {
		this = {
			is_country_type = recognized
		}
	}
	
	fail = {
		OR = {
			is_subject = yes
			gdp_ownership_ratio = 0.8
		}
	}
	
	on_monthly_pulse = {
		
	}
	
	on_complete = {

	}

	on_fail = {
		every_country = {
			limit = {
				NOT = {
					THIS = ROOT
				}
				OR = {
					ROOT = {
						"gdp_ownership_ratio(prev)" > 0.1
					}	
					overlord = ROOT
				}
			}
			save_scope_as = new_overlord_sok
		}
		scope:new_overlord_sok = {
			random_primary_culture = {
				save_scope_as = new_overlord_sok_culture
			}
		}
		this = {
			add_primary_culture = new_overlord_sok_culture
		}
	}
	
	progressbar = no
	weight = 10000
	should_be_pinned_by_default = yes

}
