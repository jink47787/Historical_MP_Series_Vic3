je_hmps_china_government_based_nerf = { #from the Yuan to the Ming to the Jin to the Qing, wheres my crown,
#the middle kingdom has long been
#This JE is essentially Is what will be handing china balance, a pick your poison
	icon = "gfx/interface/icons/event_icons/eu5_middle_kingdom.dds"
	
	group = je_group_historical_content
	
	modifiers_while_active = {
		china_country_nerf
	}
	is_shown_when_inactive = {
		country_has_primary_culture = cu:han
	}

	possible = {
		always = yes
	}
	
	immediate = {
	}

	complete = {
	}

	on_complete = {
	}
	
	invalid = {
	}
	
	on_invalid = {
	}

	#current_value = {
	#}

	#goal_add_value = {
	#}
	
	#progressbar = yes

	on_weekly_pulse = {
		effect = {
			scope:journal_entry = {
				if = { 
					limit = {
						has_modifier = china_country_nerf_monarchy
					}
					remove_modifier = china_country_nerf_monarchy
				}
				if = { 
					limit = {
						has_modifier = china_country_nerf_theocracy
					}
					remove_modifier = china_country_nerf_theocracy
				}
				if = { 
					limit = {
						has_modifier = china_country_nerf_communist
					}
					remove_modifier = china_country_nerf_communist
				}
				if = { 
					limit = {
						has_modifier = china_country_nerf_republic
					}
					remove_modifier = china_country_nerf_republic
				}
			}
			if = {
				limit = {
					country_has_monarchy_law = yes
				}
				scope:journal_entry = {
					add_modifier = china_country_nerf_monarchy
				}
			}
			else_if = {
				limit = {
					has_law_or_variant = law_type:law_theocracy
				}
				scope:journal_entry = {
					add_modifier = china_country_nerf_theocracy
				}
			}
			else_if = {
				limit = {
					has_law_or_variant = law_type:law_council_republic
				}
				scope:journal_entry = {
					add_modifier = china_country_nerf_communist
				}
			}
			else = {
				scope:journal_entry = {
					add_modifier = china_country_nerf_republic
				}
			}
		}
	}

	weight = 10000
	should_be_pinned_by_default_uninvolved_or_context = yes
	
	can_revolution_inherit = yes
}
