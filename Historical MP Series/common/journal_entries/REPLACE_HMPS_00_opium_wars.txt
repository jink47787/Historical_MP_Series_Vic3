REPLACE:je_opium_wars = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_foreign_affairs

	complete = {
		OR = {
			scope:target ?= {
				has_law_or_variant = law_type:law_free_trade
			}
			has_treaty_foreign_investment_rights_with = { TARGET = scope:target }
			any_scope_state = {
				is_treaty_port = yes
				state_region = {
					any_scope_state = {
						owner = scope:target
					}
				}
			}
		}
		hidden_trigger = {
			scope:target = {
				has_modifier = opium_wars_lost
			}
		}
	}

	on_complete = {
		add_loyalists = {
			strata = upper
			value = 0.2
		}
		add_loyalists = {
			strata = middle
			value = 0.1
		}
		add_loyalists = {
			strata = lower
			value = 0.05
		}
		trigger_event = {
			id = opium_wars.5
			days = 5
		}
	}

	fail = {
		custom_tooltip = {
			text = opium_wars_fail_tt
			scope:target = {
				NOT = {
					any_primary_culture = {
						has_cultural_obsession = opium
					}
				}
			}
		}
	}

	invalid = {
		NOT = {
			exists = scope:target
		}
	}

	on_invalid = {
		remove_variable = opium_wars_aggressor
	}

	on_fail = {
		add_radicals = {
			strata = upper
			value = 0.2
		}
		add_radicals = {
			strata = middle
			value = 0.1
		}
		add_radicals = {
			strata = lower
			value = 0.05
		}
		custom_tooltip = {
			text = opium_wars_fail_invalid_tt
			remove_variable = opium_wars_aggressor
		}
	}

	on_timeout = {
		custom_tooltip = {
			text = opium_wars_fail_invalid_tt
			remove_variable = opium_wars_aggressor
		}
	}

	weight = 10000
		
	should_be_pinned_by_default_uninvolved_or_context = yes

	timeout = 3650
}
