		c:CHI = THIS
	}
	
	scripted_button = ban_opium_button

		random_events = {
			200 = 0
			10 = opium_wars.6
			5 = opium_wars.7
			5 = opium_wars.8
		}
		effect = {
			if = {
				limit = {
					is_banning_goods = opium
					is_at_war = no
				}
				change_variable = {
					name = opium_wars_completion_tracker
					add = 1
				}
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		custom_tooltip = {
			text = opium_ban_tt
			is_banning_goods = opium
		}
		is_at_war = no
	}

	on_complete = {
		remove_variable = opium_wars_target
		trigger_event = {
			id = opium_wars.3
		}
	}

			has_law_or_variant = law_type:law_free_trade

				text = cust_chi_tp_tt
				#has_journal_entry = je_opium_wars
				any_country = {
					country_rank = rank_value:great_power #HMPS <-
					OR = {
						has_treaty_foreign_investment_rights_with = { TARGET = root }
						#has_treaty_port_in_market = root.market
						any_scope_state = {
							is_treaty_port = yes
							state_region = {
								any_scope_state = {
									owner = root
								}
							}
						}
					}
				}
			} #HMPS: <- (only for the closing bracket for the custom tooltip)
		}
	}

	on_fail = {
		remove_variable = opium_wars_target
		trigger_event = {
			id = opium_wars.4
		}
	}

	current_value = {
		value = root.var:opium_wars_completion_tracker
	}

	goal_add_value = {
		add = 60
	}

	progressbar = yes
	display_progressbar_as_months = yes

	weight = 10000
	should_be_pinned_by_default_uninvolved_or_context = yes
}

REPLACE:je_opium_wars = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	
	group = je_group_foreign_affairs

	complete = {
		OR = {
			scope:target ?= {
				has_law_or_variant = law_type:law_free_trade
			}
			has_treaty_foreign_investment_rights_with = { TARGET = scope:target }
			any_scope_state = {
				is_treaty_port = yes
				state_region = {
					any_scope_state = {
						owner = scope:target
					}
				}
			}
		}
		hidden_trigger = {
			scope:target = {
				has_modifier = opium_wars_lost
			}
		}
	}

	on_complete = {
		remove_variable = opium_wars_aggressor
		trigger_event = {
			id = opium_wars.5
		}
	}

	fail = { #HMPS: Only removed custom tooltip for this trigger
		scope:target = {
			NOT = {
				cu:han = {
					has_cultural_obsession = opium
				}
			}
		}
	}
	
	invalid = {
		NOT = {
			exists = scope:target
		}
	}

	on_invalid = {
		remove_variable = opium_wars_aggressor
	}

	on_fail = { #HMPS: Added modifiers for britian not winning the opium wars
		custom_tooltip = {
			text = opium_wars_fail_invalid_tt
			remove_variable = opium_wars_aggressor
		}
		add_modifier = { #HMPS: ->
			name = british_lost_opium_wars
			days = long_modifier_time
			is_decaying = yes
		} #HMPS: <-
	}

	on_timeout = { #HMPS: Added modifiers for britian not winning the opium wars
		custom_tooltip = {
			text = opium_wars_fail_invalid_tt
			remove_variable = opium_wars_aggressor
		}
		add_modifier = { #HMPS: ->
			name = british_lost_opium_wars
			days = long_modifier_time
			is_decaying = yes
		} #HMPS: <-
	}

	weight = 10000
	should_be_pinned_by_default_uninvolved_or_context = yes

	timeout = 3650
}
