je_hmps_revanchism_generic = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_internal_affairs

	should_be_pinned_by_default_uninvolved_or_context = yes

	#scripted_button = hmps_revanchism_gain_claims

	is_shown_when_inactive = {
		OR = {
			is_player = yes
			AND = {
				is_player = no
				OR = {
					country_rank >= rank_value:minor_power
					country_rank = rank_value:unrecognized_regional_power
				}
			}
		}
	}
	
	possible = {
		OR = {
			is_player = yes
			AND = {
				is_player = no
				country_rank >= rank_value:minor_power
			}
			AND = {
				is_country_type = unrecognized
				country_rank > insignificant_power
				is_subject = yes
			}
		}
		NOR = {
			c:DEI ?= THIS
			c:BIC ?= THIS
			c:PRU ?= THIS
			c:NGF ?= THIS
			c:GER ?= THIS
			c:AUS ?= THIS
			c:KUK ?= THIS
			c:TUR ?= THIS
			c:CHI ?= THIS
			c:TPG ?= THIS
		}
		NOR = {
			is_subject_type = subject_type_chartered_company
			is_country_type = company
		}
	}

	immediate = {
		ROOT = {
			set_variable = { #HMPS: Creates max state value
				name = max_revanchism_states
				value = 0
			}
			set_variable = { #HMPS: Creates current state value
				name = current_revanchism_states
				value = 0
			}
			set_variable = { #HMPS: Creates percent of homeland states occupied value
				name = revanchism_states_percent
				value = 0
			}
		}
		#HMPS: we have to account for the part that each 'split' of a state adds up, so if a state was split 4 ways, it would add to the value 4 times.
		every_state = { #HMPS: Counts/adds up non-split states with homelands of countries primary culture(s)
			limit = {
				is_homeland_of_country_cultures = ROOT
				is_split_state = no
			}
			ROOT = {
				change_variable = {
					name = max_revanchism_states
					add = 1
				}
			}
		}
		every_state = { #HMPS: Counts/adds up split states with homelands of countries primary culture(s)
			limit = {
				is_homeland_of_country_cultures = ROOT
				is_split_state = yes
			}
			ROOT = {
				change_variable = {
					name = max_revanchism_states
					add = 0.5
				}
			}
		}
		every_state = {
			limit = {
				state_region = s:STATE_ALSACE_LORRAINE
				NOT = {
					has_variable = hmps_state_cant_remove_revanchism
				}
			}
			set_variable = hmps_state_cant_remove_revanchism
		}
	}

	invalid = {
		OR = { #HMPS: add any new unique revanchism JEs that replace generic one here
			has_journal_entry = je_hmps_revanchism_ottoman
			has_journal_entry = je_hmps_revanchism_germany
			has_journal_entry = je_hmps_revanchism_austria
		}
	}
	
	on_invalid = {
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOT = {
						ROOT = {
							has_variable = hmps_cant_remove_revanchism_targets
						}
					}
				}
				every_state = {
					limit = { #HMPS: Removes revanchism target modifier from states you own that are your homeland, or overlords homeland if a subject
						is_homeland_of_country_cultures = ROOT
						has_modifier = modifier_hmps_revanchism_target
						NOT = { #HMPS: This must be applied to all JE variants for revanchism
							has_variable = hmps_state_cant_remove_revanchism
						}
						OR = {
							owner = ROOT
							owner = {
								OR = {
									is_subject_of = ROOT
									is_indirect_subject_of = ROOT
								}
							}
						}
					}
					remove_modifier = modifier_hmps_revanchism_target
				}
			}
			ROOT = { #HMPS: Resets the max homeland state value before recalculating it
				change_variable = {
					name = max_revanchism_states
					multiply = 0
				}
			}
			every_state = { #HMPS: Recalculated current total homeland states as this can change for some nations
				limit = {
					is_homeland_of_country_cultures = ROOT
					is_split_state = no
				}
				ROOT = {
					change_variable = {
						name = max_revanchism_states
						add = 1
					}
				}
			}
			every_state = {
				limit = {
					is_homeland_of_country_cultures = ROOT
					is_split_state = yes
				}
				ROOT = {
					change_variable = {
						name = max_revanchism_states
						add = 0.5
					}
				}
			}
			ROOT = { #HMPS: Resets current revanchism states value so it isnt increasing forever
				change_variable = {
					name = current_revanchism_states
					multiply = 0
				}
			}
			every_state = { #HMPS: Counts up states that are current revanchism targets for the country
				limit = {
					is_homeland_of_country_cultures = ROOT
					is_treaty_port = no
					NOR = {
						owner = ROOT
						owner = {
							OR = {
								is_subject_of = ROOT
								is_indirect_subject_of = ROOT
							}
						}
					}
					has_modifier = modifier_hmps_revanchism_target
				}
				ROOT = {
					change_variable = {
						name = current_revanchism_states
						add = 1
					}
				}
			}
			ROOT = { #HMPS: Resets the percent of revanchism target states to total homeland states
				change_variable = {
					name = revanchism_states_percent
					multiply = 0
				}					
				change_variable = { #HMPS: Calculated the percentage of revanchism target states to total homeland states to scale the modifier(s) gained from revanchism
					name = revanchism_states_percent
					add = ROOT.var:current_revanchism_states
				}
				change_variable = {
					name = revanchism_states_percent
					divide = ROOT.var:max_revanchism_states
				}
			}
			ROOT = { #HMPS: Removes revanchism modifiers before re-applying them so it doesn't stack up multiple times
				remove_modifier = modifier_hmps_revanchism_0
				remove_modifier = modifier_hmps_revanchism_10
				remove_modifier = modifier_hmps_revanchism_25
				remove_modifier = modifier_hmps_revanchism_50
				remove_modifier = modifier_hmps_revanchism_subject_0
				remove_modifier = modifier_hmps_revanchism_subject_10
				remove_modifier = modifier_hmps_revanchism_subject_25
				remove_modifier = modifier_hmps_revanchism_subject_50
			}
			#Threashold levels/percents to be heavily looked at to make sure they are balanced
			if = { #HMPS: Checks if you have reached a certain revanchism threashold and applies a modifiers for that level.
				limit = {
					ROOT.var:revanchism_states_percent > 0
					ROOT.var:revanchism_states_percent < 0.10
				}
				if = {
					limit = {
						ROOT = {
							is_subject = no
						}
					}
					ROOT = {
						add_modifier = {
							name = modifier_hmps_revanchism_0
							multiplier = ROOT.var:revanchism_states_percent
						}
					}
				}
				else = {
					ROOT = {
						add_modifier = {
							name = modifier_hmps_revanchism_subject_0
							multiplier = ROOT.var:revanchism_states_percent
						}
					}
				}
			}				
			if = {
				limit = {
					ROOT.var:revanchism_states_percent >= 0.10
					ROOT.var:revanchism_states_percent < 0.25
				}
				if = {
					limit = {
						ROOT = {
							is_subject = no
						}
					}
					ROOT = {
						add_modifier = {
							name = modifier_hmps_revanchism_10
							multiplier = ROOT.var:revanchism_states_percent
						}
					}
				}
				else = {
					ROOT = {
						add_modifier = {
							name = modifier_hmps_revanchism_subject_10
							multiplier = ROOT.var:revanchism_states_percent
						}
					}
				}
			}
			if = {
				limit = {
					ROOT.var:revanchism_states_percent >= 0.25
					ROOT.var:revanchism_states_percent < 0.50
				}
				if = {
					limit = {
						ROOT = {
							is_subject = no
						}
					}
					ROOT = {
						add_modifier = {
							name = modifier_hmps_revanchism_25
							multiplier = ROOT.var:revanchism_states_percent
						}
					}
				}
				else = {
					ROOT = {
						add_modifier = {
							name = modifier_hmps_revanchism_subject_25
							multiplier = ROOT.var:revanchism_states_percent
						}
					}
				}
			}
			if = {
				limit = {
					ROOT.var:revanchism_states_percent >= 0.50
				}
				if = {
					limit = {
						ROOT = {
							is_subject = no
						}
					}
					ROOT = {
						add_modifier = {
							name = modifier_hmps_revanchism_50
							multiplier = ROOT.var:revanchism_states_percent
						}
					}
				}
				else = {
					ROOT = {
						add_modifier = {
							name = modifier_hmps_revanchism_subject_50
							multiplier = ROOT.var:revanchism_states_percent
						}
					}
				}
			}
		}
	}			
}
