je_india_neoimperialism = {
	should_be_pinned_by_default = yes
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_british_india

	modifiers_while_active = {
		modifier_india_neoimperialism_cost_reduction
	}

	is_shown_when_inactive = {
		OR = {
			c:BIC ?= this
			c:BHT ?= this
			c:BGL ?= this
		}
		#has_technology_researched = pan-nationalism
	}

	possible = {
		OR = {
			c:BIC ?= this
			c:BHT ?= this
			c:BGL ?= this
		}
		#has_technology_researched = pan-nationalism
		is_subject = no
	}

	immediate = {
		set_global_variable = {
			name = neoimperialism_local_ownership
			value = 0
		}
	}
	
	timeout = 36500 # 100 years
	on_timeout = {
		remove_modifier = modifier_india_neoimperialism
	}
	
	on_weekly_pulse = {
		effect = {
			### Scaling by Railroads
			
			set_global_variable = {
				name = neoimperialism_local_ownership
				value = {
					add = "gdp_ownership_ratio(ROOT)"
					multiply = 100
				}
			}
			
			if = {
				limit = {
					has_modifier = modifier_india_neoimperialism
				}
				remove_modifier = modifier_india_neoimperialism
			}
			add_modifier = {
				name = modifier_india_neoimperialism
                multiplier = {
					add = 1
                    subtract = "gdp_ownership_ratio(ROOT)"
                }
			}
		}
	}

	current_value = {
		value = global_var:neoimperialism_local_ownership
	}

	goal_add_value = {
		value = 100
	}

	progressbar = yes

	can_revolution_inherit = yes

	#weight = 5000
	should_be_pinned_by_default = yes
}
