je_africa_funny = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"
	
	group = je_group_internal_affairs
	
	is_shown_when_inactive = {
		any_primary_culture = {
			has_discrimination_trait_group = heritage_group_african
		}
		is_in_geographic_region = geographic_region_africa
		NOT = { has_technology_researched = quinine }
		is_player = yes
	}

	possible = {
		any_primary_culture = {
			has_discrimination_trait_group = heritage_group_african
		}
		is_in_geographic_region = geographic_region_africa
		NOT = { has_technology_researched = quinine }
		is_player = yes
	}
	
	immediate = {
		set_variable = {
			name = africa_funny_var
			value = 0
		}
	}
	
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_technology_researched = colonization
					is_player = yes
				}
				change_variable = {
					name = africa_funny_var
					add = 1
				}
			}
			if = {
				limit = {
					has_technology_researched = shaft_mining
					is_player = yes
				}
				change_variable = {
					name = africa_funny_var
					add = 1
				}
			}
			if = {
				limit = {
					has_technology_researched = line_infantry
					is_player = yes
				}
				change_variable = {
					name = africa_funny_var
					add = 1
				}
			}
			if = {
				limit = {
					has_technology_researched = academia
					is_player = yes
				}
				change_variable = {
					name = africa_funny_var
					add = 1
				}
			}
			if = {
				limit = {
					NOT = { has_law_or_variant = law_type:law_no_colonial_affairs }
					is_player = yes
				}
				change_variable = {
					name = africa_funny_var
					add = 1
				}
			}
		}
	}
	
	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		has_technology_researched = colonization
		has_technology_researched = shaft_mining
		has_technology_researched = line_infantry
		has_technology_researched = academia
		NOT = { has_law_or_variant = law_type:law_no_colonial_affairs }
	}
	
	on_complete = {
		if = {
			limit = {
				NOT = { has_technology_researched = medical_degrees }
			}
			add_technology_researched = medical_degrees
		}
		if = {
			limit = {
				NOT = { has_technology_researched = pharmaceuticals }
			}
			add_technology_researched = pharmaceuticals
		}
		add_technology_researched = quinine
		add_modifier = {
			name = viable_african_country
			years = 25
			is_decaying = yes
		}
	}
	
	invalid = {
		has_technology_researched = quinine
	}

	
	current_value = {
		value = root.var:africa_funny_var
	}

	goal_add_value = {
		value = 300
	}

	progressbar = yes
	can_revolution_inherit = yes

	weight = 5000
	should_be_pinned_by_default = yes
}
