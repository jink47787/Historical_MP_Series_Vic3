je_egypt_european_influences = { #Sets Egypt to No Schools if Non-Ottoman Empire Nation Puppets Them Before They Get Public Schools, Discourages Egypts from Wanting to Get Puppeted, Also Represents Their Education System Getting Destroyed
	icon = "gfx/interface/icons/event_icons/event_skull.dds"
	
	group = je_group_historical_content
	
	modifiers_while_active = {
		modifier_egyptian_foreign_influences
	}
		
	fail = {
		AND = {
			is_subject = yes
			NOT = {
				is_subject_of = c:TUR
			}
			is_at_war = no
		}
	}
	
	complete = {
		AND = {
			is_country_type = recognized
			has_law = law_type:law_public_schools
		}
	}
	
	on_fail = {
		activate_law = law_type:law_no_schools
	}
}
