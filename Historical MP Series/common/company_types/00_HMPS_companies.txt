company_bhp = { #established NSW 1885, Large mining company, still one of the largest today.
	icon = "gfx/interface/icons/company_icons/historical_company_icons/company_bhp.dds"
	background = "gfx/interface/icons/company_icons/company_backgrounds/comp_illu_mining.dds"

	flavored_company = yes

	preferred_headquarters = { STATE_NEW_SOUTH_WALES }

	building_types = {
		building_iron_mine
		building_lead_mine
	}

	extension_building_types = { # There are some arguments to make some of these buildings default part of company however for now I am keeping these as only extension options
		building_sulfur_mine
		building_gold_mine
		building_coal_mine
	}

	possible_prestige_goods = {
	}

	potential = {
		country_has_primary_culture = cu:australian # Locking behind formation would be ahistorical
	}

	attainable = {
		hidden_trigger = { # Hide attainable if it's always fulfilled by possible
			any_scope_state = {
				state_region = s:STATE_NEW_SOUTH_WALES
			}
		}
	}

	possible = {
		any_scope_state = {
			state_region = s:STATE_NEW_SOUTH_WALES
			is_incorporated = yes
			any_scope_building = {
				is_building_type = building_iron_mine
				level >= 5
			}
		}
	}

	prosperity_modifier = {
		building_group_bg_mining_throughput_add = 0.1
	}

	ai_will_do = {
		any_scope_state = {
			state_region = s:STATE_NEW_SOUTH_WALES
			is_incorporated = yes
		}
	}

	ai_construction_targets = {
		building_iron_mine = {
			level = 5
			state_trigger = {
				state_region = s:STATE_NEW_SOUTH_WALES
			}
		}
	}

	ai_weight = {
		value = 3 # Higher base value for flavored companies
	}
}

company_cpr = {

	flavored_company = yes

	preferred_headquarters = { STATE_QUEBEC }

	building_types = {
		building_railway
		building_port
	}

	extension_building_types = {
		building_coal_mine
		building_art_academy
	}
	
	potential = {
		        OR = {
            country_has_primary_culture = cu:anglo_canadian
            country_has_primary_culture = cu:franco_canadian
        }
    }	
	
	possible = {
        any_scope_state = {
            state_region = s:STATE_QUEBEC
            any_scope_building = {
                is_building_type = building_railway
            }
        }
        any_scope_state = {
            state_region = s:STATE_ONTARIO
            any_scope_building = {
                is_building_type = building_railway
            }
        }
        any_scope_state = {
            state_region = s:STATE_MANITOBA
            any_scope_building = {
                is_building_type = building_railway
            }
        }
        any_scope_state = {
            state_region = s:STATE_SASKATCHEWAN
            any_scope_building = {
                is_building_type = building_railway
            }
        }
		any_scope_state = {
            state_region = s:STATE_ALBERTA
            any_scope_building = {
                is_building_type = building_railway
            }
        }
		any_scope_state = {
            state_region = s:STATE_BRITISH_COLUMBIA
            any_scope_building = {
                is_building_type = building_railway
            }
        }
    }
	
	prosperity_modifier = {
		state_migration_pull_mult = 0.1
		building_urban_center_throughput_add = 0.1
	}
	
	ai_will_do = {
		any_scope_state = {
			state_region = s:STATE_QUEBEC
			is_incorporated = yes
		}
	}

	ai_weight = {
		value = 3 # Higher base value for flavored companies
	}
}
