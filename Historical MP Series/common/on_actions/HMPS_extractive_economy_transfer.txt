#Code Based on Treasury to Investment Pool by Lord R

on_monthly_pulse_country = {
	on_actions = {
		extractive_economy_investment_transfer_monthly_effect
	}
}

extractive_economy_investment_transfer_monthly_effect = {
	effect = {
		if = {
			limit = {
				is_subject = yes
				OR = {
					has_law = law_type:law_extraction_economy
					has_law = law_type:law_the_company
				}
			}
			set_global_variable = {
				name = investment_pool_var
				value = investment_pool
			}
			overlord = {
				add_investment_pool = global_var:investment_pool_var
			}
			change_global_variable = {
				name = investment_pool_var
				multiply = -1
			}
			add_investment_pool = global_var:investment_pool_var
		}
		remove_global_variable = investment_pool_var
	}
}
