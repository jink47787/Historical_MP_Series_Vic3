on_state_created = {
	on_actions = {
		HMPS_add_resource_indicators
	}
}
on_monthly_pulse_state = {
	on_actions = {
		HMPS_add_resource_indicators
	}
}
HMPS_add_resource_indicators = { #State Scope
	effect = {
		if = {
			limit = {
				has_modifier = modifier_max_oil_indicator
			}
			remove_modifier = modifier_max_oil_indicator
		}
		if = {
			limit = {
				has_modifier = modifier_max_rubber_indicator
			}
			remove_modifier = modifier_max_rubber_indicator
		}
		if = {
			limit = {
				has_modifier = modifier_max_gold_indicator
			}
			remove_modifier = modifier_max_gold_indicator
		}
		
		if = {
			limit = {
				"num_potential_resources(bt:building_oil_rig)" > 0
			}
			add_modifier = { 
				name = modifier_max_oil_indicator
				multiplier = "num_potential_resources(bt:building_oil_rig)"
				days = -1
			}
		}
		if = {
			limit = {
				"num_potential_resources(bt:building_rubber_plantation)" > 0
			}
			add_modifier = { 
				name = modifier_max_rubber_indicator
				multiplier = "num_potential_resources(bt:building_rubber_plantation)"
				days = -1
			}
		}
		if = {
			limit = {
				OR = {
					"num_potential_resources(bt:building_gold_field)" > 0
					"num_potential_resources(bt:building_gold_mine)" > 0
				}
			}
			add_modifier = { 
				name = modifier_max_gold_indicator
				multiplier = {
					value = "num_potential_resources(bt:building_gold_field)"
					add = "num_potential_resources(bt:building_gold_mine)"
				}
				days = -1
			}
		}
	}
}
