on_game_started_after_lobby = {
	on_actions = {
		HMPS_coastal_states_get_port
	}
}
HMPS_coastal_states_get_port = {
	effect = {
		every_state = {
			limit = {
				NOT = {
					has_building = building_port
				}
			}
			create_building = {
				building = building_port
				activate_production_methods = { pm_anchorage }
				subsidized = yes
				reserves = 1
				level = 1
			}
		}
	}
}
