on_game_started_after_lobby = {
	on_actions = {
		HMPS_coastal_states_get_port
		HMPS_day_one_bloc_leverage
	}
}
HMPS_day_one_bloc_leverage = { # i genuinely dont know if this does anything but when i test things the problems seems to go away when i look in game
	effect = {
		c:PRU ?= {
			power_bloc = {
				add_leverage = {
				  target = c:BAV
				  value = 200
				}
			}
		}
		every_country = {
			limit = {
				is_in_power_bloc = yes
				is_power_bloc_leader = no
			}
			power_bloc = {
				add_leverage = { target = prev value = 200 }
			}
		}
	}
}
HMPS_coastal_states_get_port = {
	effect = {
		every_state = {
			limit = {
				NOT = {
					has_building = building_port
				}
				is_coastal = yes
			}
			create_building = {
				building = building_port
				activate_production_methods = { pm_anchorage }
				subsidized = yes
				reserves = 1
				level = 1
			}
		}
	}
}
