on_game_started_after_lobby = {
	on_actions = {
		HMPS_coastal_states_get_port
		HMPS_day_one_bloc_leverage
		#HMPS_annex_athabaska
	}
}

HMPS_annex_athabaska = {
	effect = {
		s:STATE_YUKON.region_state:ALK ?= {
			set_state_owner = c:HBC
		}
		s:STATE_BRITISH_COLUMBIA.region_state:ALK ?= {
			set_state_owner = c:ORG
		}
		c:ALK ?= {
			annex ?= c:ATB
		}
	}
}
HMPS_day_one_bloc_leverage = { # i genuinely dont know if this does anything but when i test things the problems seems to go away when i look in game
	effect = {
		c:PRU ?= {
			power_bloc = {
				add_leverage = {
				  target = c:BAV
				  value = 200
				}
			}
		}
		every_country = {
			limit = {
				is_in_power_bloc = yes
				is_power_bloc_leader = no
			}
			power_bloc = {
				add_leverage = { target = prev value = 200 }
			}
		}
	}
}
HMPS_coastal_states_get_port = {
	effect = {
		every_state = {
			limit = {
				NOT = {
					has_building = building_port
				}
				is_coastal = yes
			}
			owner = {
				if = {
					limit = {
						not = { has_technology_researched = navigation }
					}
					add_technology_researched = navigation
				}
			}
			create_building = {
				building = building_port
				activate_production_methods = { pm_anchorage }
				subsidized = yes
				reserves = 1
				level = 1
			}
		}
	}
}
