#This modifier is to essentially to ensure that population doenst implode in cases like this. It is only applied when a puppet is essentially entirely occupied during a war.

	state_mortality_mult = -1.1

	unit_defense_mult = -10

	state_tax_capacity_mult = -10

	building_throughput_add = -10

	country_government_wages_mult = -10 
	country_loan_interest_rate_add = -10
	state_welfare_payments_mult = -10
