tanzimat_education_core_progress_bar = {
 	name = "tanzimat_education_core_progress_bar_name"
	desc = "tanzimat_education_core_progress_bar_text_1" #This will be printed on the left side of the progress bar


 	default_green = yes
	
	monthly_progress = {
		if = {
			limit = {
 				c:TUR = {
					literacy_rate <= 0.20 
				}
 			}
			add = {
				desc = "tanzimat_education_NEGATIVE_PASSIVE_1"
				value = -0.1
			}
 		}
		if = {
			limit = {
 				c:TUR = {
					AND = {
						literacy_rate >= 0.20 
						literacy_rate <= 0.30 
					}
				}
 			}
			add = {
				desc = "tanzimat_education_POSITIVE_PASSIVE_1"
				value = 0.1
			}
 		}
		if = {
			limit = {
 				c:TUR = {
					AND = {
						literacy_rate >= 0.30 
						literacy_rate <= 0.40 
					}
				}
 			}
			add = {
				desc = "tanzimat_education_POSITIVE_PASSIVE_2"
				value = 0.2
			}
 		}
		if = {
			limit = {
 				c:TUR = {
					AND = {
						literacy_rate >= 0.40 
						literacy_rate <= 0.50 
					}
				}
 			}
			add = {
				desc = "tanzimat_education_POSITIVE_PASSIVE_3"
				value = 0.3
			}
 		}
		if = {
			limit = {
 				c:TUR = { literacy_rate >= 0.50 }
 			}
			add = {
				desc = "tanzimat_education_POSITIVE_PASSIVE_4"
				value = 0.4
			}
 		}
		if = {
			limit = {
 				NOT = { has_law_or_variant = law_type:law_no_schools }
 			}
			add = {
				desc = "tanzimat_education_POSITIVE_PASSIVE_5"
				value = 0.5
			}
 		}
 	}

 	start_value = 0
 	min_value = 0
 	max_value = 100
}

tanzimat_bureaucracy_core_progress_bar = {
 	name = "tanzimat_bureaucracy_core_progress_bar_name"
	desc = "tanzimat_bureaucracy_core_progress_bar_text_1" #This will be printed on the left side of the progress bar


 	default_green = yes
	
	monthly_progress = {
		if = {
			limit = {
 				bureaucracy >= 0 
 			}
			add = {
				desc = "tanzimat_bureaucracy_POSITIVE_PASSIVE_1"
				value = 0.5
			}
 		}
				if = {
			limit = {
 				bureaucracy <= 0 
 			}
			add = {
				desc = "tanzimat_bureaucracy_NEGATIVE_PASSIVE_1"
				value = -0.5
			}
 		}
		if = {
			limit = {
 				approaching_bureaucracy_shortage = yes
 			}
			add = {
				desc = "tanzimat_bureaucracy_NEGATIVE_PASSIVE_2"
				value = -0.5
			}
 		}
		if = {
			limit = {
 				any_scope_building = {
					filter = {
						is_building_type = building_government_administration
					}			
					OR = {
						building_has_goods_shortage = yes
						has_active_production_method = pm_simple_organization
					}
					percent = 1
				}
 			}
			add = {
				desc = "tanzimat_bureaucracy_NEGATIVE_PASSIVE_3"
				value = -0.5
			}
 		}
		if = {
			limit = {
 				NOT = { has_law_or_variant = law_type:law_hereditary_bureaucrats }
 			}
			add = {
				desc = "tanzimat_bureaucracy_POSITIVE_PASSIVE_2"
				value = 0.5
			}
 		}
		if = {
			limit = {
 				NOT = { has_law_or_variant = law_type:law_traditionalism }
 			}
			add = {
				desc = "tanzimat_bureaucracy_POSITIVE_PASSIVE_3"
				value = 0.5
			}
 		}
 	}

 	start_value = 0
 	min_value = 0
 	max_value = 100
}
