	placement = ROOT
	
	title = resignation.1.t

					ruler.interest_group = {
						OR = {
							ig_approval <= angry
							is_marginal = yes
						}
					}
				}
			}
			triggered_desc = {
				desc = resignation.1.d2
				trigger = {
					OR = {
						ruler = {
							popularity <= -50
						}
						has_revolution = yes
					}
				}
			}
			triggered_desc = {
				desc = resignation.1.d3
				trigger = {
					ruler = {
						OR = {
							has_trait = psychological_affliction
							has_trait = cancer
							has_trait = senile
							has_trait = wounded
						}
					}
				}
			}
			triggered_desc = {
				desc = resignation.1.d4
				trigger = {
					always = yes
				}
			}

				save_scope_as = resignation_ig
			}
		}
	}

	option = { # Support successor
		name = resignation.1.a
		default_option = yes
		trigger = {
			NOR = {
				scope:resignation_ig = {
					OR = {
						ig_approval <= angry
						is_marginal = yes
					}
				}
				scope:resigning_politician = {
					popularity <= -50
				}
			}
		}
		add_modifier = {
			name = modifier_best_of_luck
			months = normal_modifier_time
		}
		custom_tooltip = {
			text = leader_resigns_from_office_tt
			scope:resigning_politician = {
				kill_character = {
					hidden = yes
				}
			}
		}
	}

	option = { # Disastrous regime
		name = resignation.1.b
		default_option = yes
		trigger = {
			OR = {
				scope:resignation_ig = {
					OR = {
						ig_approval <= angry
						is_marginal = yes
					}
				}
				scope:resigning_politician = {
					popularity <= -50

			}

		add_modifier = {
			name = modifier_disastrous_administration

			value = small_radicals

		custom_tooltip = {
			text = leader_resigns_from_office_tt

			}

			text = leader_resigns_from_office_tt

					hidden = yes
				}
			}
		}

		}
