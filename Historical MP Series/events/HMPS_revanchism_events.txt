namespace = hmps_revanchism

hmps_revanchism.1 = { #HMPS: Removes revanchism targets on states and inability to remove targets on owned homelands if a player released a country willingly as independent for some reason (exists to prevent exploiting the revanchism system)
	type = country_event
	hidden = yes

	immediate = {
		if = {
			limit = {
				any_country = {
					has_truce_with = ROOT
				}
			}
			add_treasury = 1
		}
		else = {
			ROOT = {
				remove_variable = hmps_cant_remove_revanchism_targets
			}
			every_state = {
				limit = {
					owner = ROOT
				}
				remove_modifier = modifier_hmps_revanchism_target
			}
		}
	}
}
