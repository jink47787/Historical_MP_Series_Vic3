# Nian Rebellion
heavenly_concurrent.1 = {
	type = country_event
	placement = scope:nian_spawn

	title = heavenly_concurrent.1.t
	desc = heavenly_concurrent.1.d
	flavor = heavenly_concurrent.1.f

	event_image = {
		video = "asia_union_leader"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
		# Fired by event
	}
	
	immediate = {
		create_country = {
			tag = NIA
			origin = this
			state = scope:nian_spawn
		}
		c:NIA = {
			create_character = {
				first_name = "Zhang"
				last_name = "Lexing"
				birth_date = 1810.1.1
				culture = cu:han
				religion = rel:mahayana
				interest_group = ig_landowners
				ruler = yes
				historical = yes
				ideology = ideology_moderate
				traits = {
					inspirational_orator
					pillager
				}
			}
			taiping_concurrent_rebellion_setup = yes
		}
	}

	option = {
		name = heavenly_concurrent.1.a
		default_option = yes
		
		c:NIA = {
			create_diplomatic_play = {
				target_country = ROOT
				escalation = 80
				type = dp_independence
				
				target_state = s:STATE_HENAN.region_state:CHI
		
				add_war_goal = {
					holder = ROOT
					type = annex_country
					target_country = c:NIA
				}
			}
		}
	}
}

heavenly_concurrent.5 = {
	type = country_event
	placement = scope:dungan_spawn

	title = heavenly_concurrent.5.t
	desc = heavenly_concurrent.5.d
	flavor = heavenly_concurrent.5.f

	event_image = {
		video = "asia_union_leader"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
		# Fired by event
	}
	
	immediate = {
		create_country = {
			tag = DGN
			origin = this
			state = scope:dungan_spawn
			on_created = {
				set_state_religion = rel:sunni
			}
		}
		c:DGN = {
			activate_law = law_type:law_theocracy
			create_character = {
				first_name = "Yakub"
				last_name = "Beg"
				birth_date = 1820.1.1
				culture = cu:uighur
				religion = rel:sunni
				interest_group = ig_landowners
				ruler = yes
				historical = yes
				ideology = ideology_moderate
				traits = {
					inspirational_orator
				}
			}
			taiping_concurrent_rebellion_setup = yes
		}
		every_state = {
			limit = {
				state_region = s:STATE_DZUNGARIA
			}
			set_state_owner = c:DGN
			set_state_type = incorporated
		}
	}

	option = {
		name = heavenly_concurrent.1.a
		default_option = yes
		
		c:DGN = {
			create_diplomatic_play = {
				target_country = ROOT
				escalation = 80
				type = dp_independence
				
				target_state = s:STATE_TIANSHAN.region_state:CHI
		
				add_war_goal = {
					holder = ROOT
					type = annex_country
					target_country = c:DGN
				}
			}
		}
	}
}
