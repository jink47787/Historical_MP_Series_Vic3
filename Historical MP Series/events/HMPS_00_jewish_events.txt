namespace = zionism

zionism.1 = {
	type = country_event
	placement = ROOT
	title = zionism.1.t
	desc = zionism.1.d
	flavor = zionism.1.f
	
	duration = 3
	
	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}
	
	trigger = {
		any_scope_pop = {
			culture = cu:ashkenazi
			size >= 1000
		}
	}
	
	immediate = {
		random_scope_state = {
            limit = {
                #region = sr:region_caucasus
                any_scope_pop = {
                    culture = cu:ashkenazi
                }
            }
        	save_scope_as = ashkenazi_origin_state
        }
        random_country = {
            limit = {
                any_scope_state = {
                    OR = {
                        state_region = s:STATE_TRANSJORDAN
                    }
                }
            }
            save_scope_as = ashkenazi_destination_country
            random_scope_state = {
                limit = {
                    OR = {
                        state_region = s:STATE_TRANSJORDAN
                    }
                }
                save_scope_as = ashkenazi_destination_state
            }
        }
        cu:circassian = {
            save_scope_as = ashkenazi_culture
        }
        set_variable = {
            name = ashkenazi_expulsion_stall
            months = 3
        }
	}
	
	option = {
		name = zionism.1.a
		scope:ashkenazi_origin_state = {
            #text = three_pops_deported_tt
            random_scope_pop = {
                limit = {
                    culture = cu:ashkenazi
                    strata = lower
                }
                move_partial_pop = {
					state = scope:ashkenazi_destination_state 
					population_ratio =  0.5
				}
            }
            random_scope_pop = {
                limit = {
                    culture = cu:ashkenazi
                    strata = middle
                }
                move_partial_pop = {
					state = scope:ashkenazi_destination_state 
					population_ratio =  0.5
				}
            }
            random_scope_pop = {
                limit = {
                    culture = cu:ashkenazi
                    strata = upper
                }
                move_partial_pop = {
					state = scope:ashkenazi_destination_state 
					population_ratio =  0.5
				}
            }
        }
     }
}
