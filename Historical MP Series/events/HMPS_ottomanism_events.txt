namespace = ottomanism_events


# intergrate Greeks into Ottomanism
ottomanism_events.1 = {
	type = country_event
	placement = ROOT

	title = ottomanism_events.1.t
	desc = ottomanism_events.1.d
	flavor = ottomanism_events.1.f

	event_image = {
		video = "ottomanism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/ottoman_fed.dds"
	duration = 3
	trigger = {
		# Fired by button
	}

	option = {
		name = ottomanism_events.1.a
		default_option = yes

		ROOT = {
			# Integrate Greeks into Ottomanism
			every_scope_pop = {
					limit = { culture = cu:greek }
					change_pop_culture = { target = cu:ottoman value = 0.2 }
			}
		}
	}
}
