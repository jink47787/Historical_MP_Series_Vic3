			NOT = {
				any_diplomatic_play = {
					any_scope_play_involved = {
						THIS = ROOT
					}
				}
			}
		}
		add_modifier = {
			name = liberal_germany
			months = 120
			is_decaying = yes
		}
		custom_tooltip = {
			text = gutter_crown.1.b.t
			if = {
				limit = {
					c:AUS = {
						NOT = {has_technology_researched = nationalism}
					}
				}
				c:AUS = {
					add_technology_researched = nationalism
				}
			}
			
			every_country = {
				limit = {
					country_rank >= rank_value:great_power
				}
				change_relations = {
					country = ROOT
					value = -50
				}
			}
			c:AUS = {
				trigger_event = {
					id = gutter_crown.2
					days = 7
					popup = yes
				}
			}
		}

		
		custom_tooltip = gutter_crown.1.b.tt
		ai_chance = {
			base = 25
		}
		default_option = yes
	}

				#	type = unification_leadership
				#	target_country = c:AUS
				#}
			}
		}

		custom_tooltip = gutter_crown.2.a.t
		ai_chance = {
			base = 75
		}
		default_option = yes
	}


	
	#Give up on Germany
	option = {
		name = gutter_crown.2.b	
		
		remove_modifier = german_confederation_leader
		#Give every AI German subject an event that annexes them into Prussia
		custom_tooltip = {
			text = gutter_crown.2.b.t
			c:PRU = {
				trigger_event = {
					id = gutter_crown.4
					days = 14
					popup = yes
				}
			}
		
			every_country = {
				limit = {
					AND = {
						OR = {
							country_has_primary_culture = cu:north_german
							country_has_primary_culture = cu:south_german
						}
						OR = {
							has_diplomatic_pact = {
								who = c:PRU
								type = customs_union
							}
							is_subject_of = c:PRU
						}
					}
				}
				trigger_event = {
					id = gutter_crown.3
					days = 0
					popup = yes
				}
			}
		}
		ai_chance = {
			base = 25
		}
