					is_treaty_port = yes

				}

					save_scope_as = treaty_port_owner
				}
			}
		}
	}

	option = {
		name = heavenly.1.a
		default_option = yes
		ai_chance = {
			base = 20

			modifier = {
				trigger = {
					has_modifier = opium_wars_lost
				}

		}

			add_modifier = {
				name = missionary_schools_hospitals

			}

				target = rel:protestant

	}
	option = {
		name = heavenly.1.b

			base = 20
		}
