 namespace = russia_monarchy #files brought over to make sure that monarchs take over relevant interest groups

russia_monarchy.1 = { #Accession of Alexander II
	type = country_event
	placement = root

	title = russia_monarchy.1.t
	desc = russia_monarchy.1.d
	flavor = russia_monarchy.1.f

	gui_window = event_window_1char_tabloid
	
	left_icon = scope:tsar_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/law_icons/monarchy.dds"

	duration = 3

	cooldown = { days = stupidly_long_modifier_time }

	trigger = {
		#Fired by on_action
	}

	immediate = {
		ruler = {
			save_scope_as = tsar_scope
		}
		root = {
			save_scope_as = russia_scope
		}
	}

	option = {
		name = russia_monarchy.1.a
		default_option = yes
		show_as_tooltip = {
			scope:tsar_scope = {
				set_character_as_ruler = yes
			}
		}
		custom_tooltip = {
			text = alexander_ii_enthroned_tt
		}
		hidden_effect = {
			create_character = {
				template = RUS_tsarevich_nicholas_alexandrovich
			}
		}
		ruler = {
			set_as_interest_group_leader = yes #HMPS Change
			add_modifier = {
				name = modifier_alexander_ii
				days = very_long_modifier_time
			}
		}
		if = {
			limit = {
				NOT = {
					any_political_movement = {
						is_political_movement_type = movement_modernizer
					}
				}
			}
			create_political_movement = { type = movement_modernizer }
		}
		add_journal_entry = { type = je_great_reformer }
		add_journal_entry = { type = je_great_reforms_serfdom }
		add_journal_entry = { type = je_great_reforms_military }
		add_journal_entry = { type = je_great_reforms_bureaucratic }
	}
}

	}

	option = {
		name = russia_monarchy.5.a
		default_option = yes
		show_as_tooltip = {
			scope:tsar_scope = {
				set_character_as_ruler = yes
			}
		}
		ruler = {
			set_as_interest_group_leader = yes #HMPS Change
			add_modifier = {
				name = modifier_nicholas_ii
				days = very_long_modifier_time
			}
		}
		hidden_effect = {
			if = {
				limit = {
					game_date >= 1904.8.12
				}
				create_character = {
					template = RUS_alexei_ii
				}
				set_variable = alexei_born
			}
			else_if = {
				limit = {
					game_date < 1904.8.12
					game_date >= 1895.1.1
				}
				create_character = {
					template = RUS_michael_ii
				}
			}
			else_if = {
				limit = {
					game_date < 1895.1.1
					game_date >= 1871.1.1
				}
				create_character = {
					template = RUS_george_i
				}
			}
		}
	}
}
