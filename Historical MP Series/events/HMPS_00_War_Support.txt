namespace = war_support

#Event for succeeding countries

war_support.1 = {
	type = country_event
	placement = root
	title = war_support.1.t
	desc = war_support.1.d
	flavor = war_support.1.f
	
	duration = 3
	
	event_image = {
		video = "southamerica_aristocrats"
	}
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	
	trigger = {
		#Triggered by decision
	}
	
	#Affect the initial nation
	
	immediate = {
		random_scope_war = {
			limit = {
				has_war_support = {
					target = ROOT
					value < 0
				}
			}
			save_scope_as = player_war
		}
	}
	
	option = {
		name = war_support.1.a
		add_modifier = {
			name = prolonged_war
			months = 121
			is_decaying = yes
		}
		add_radicals = {
			value = large_radicals
		}
		scope:player_war = {
			add_war_war_support = {
				target = ROOT
				value = 100
			}
		}
		remove_variable = bad_code
	}
	
	option = {
		name = gm_actions.14.b
		trigger_event = {
			id = war_support.1
			days = 0
			popup = yes
		}
	}
	
	option = {
		name = gm_actions.14.c
		default_option = yes
		remove_variable = bad_code
	}
	
}
