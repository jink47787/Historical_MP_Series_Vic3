namespace = war_support

#Event for succeeding countries

war_support.1 = {
	type = country_event
	placement = root
	title = war_support.1.t
	desc = war_support.1.d
	flavor = war_support.1.f
	
	duration = 3
	
	event_image = {
		video = "southamerica_aristocrats"
	}
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	
	trigger = {
		#Triggered by decision
	}
	
	#Affect the initial nation
	
	immediate = {
		random_scope_war = {
			limit = {
				has_war_support = {
					target = ROOT
					value < 0
				}
			}
			save_scope_as = player_war
		}
		ROOT = {
			save_scope_as = boost_war_support_scope
		}
	}
	
	option = {
		name = war_support.1.a
		scope:player_war = {
			add_war_war_support = {
				target = ROOT
				value = 100
			}
		}
		if = {
			limit = {
				has_technology_researched = war_propaganda
			}
			add_modifier = {
				name = prolonged_war
				multiplier = 0.5
				years = 5
				is_decaying = yes
			}
			add_radicals = {
				value = medium_radicals
			}
		}
		else = {
			add_modifier = {
				name = prolonged_war
				years = 5
				is_decaying = yes
			}
			add_radicals = {
				value = large_radicals
			}
		}
		remove_variable = bad_code
		every_country = {
			limit = {
				has_war_with = ROOT
			}
			post_notification = took_war_support
		}
	}
	
	option = {
		name = gm_actions.14.b
		trigger_event = {
			id = war_support.1
			days = 0
			popup = yes
		}
	}
	
	option = {
		name = gm_actions.14.c
		default_option = yes
		remove_variable = bad_code
	}
}
