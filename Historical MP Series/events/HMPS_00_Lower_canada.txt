lower_canada_evt.2 = { #tensions raised leading to rebellion
	type = country_event
	placement = ROOT
	title = lower_canada_evt.2.t
	desc = lower_canada_evt.2.d
	flavor = lower_canada_evt.2.f

	duration = 3

	event_image = {
		video = "unspecific_military_parade"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	option = {
		name = lower_canada_evt.2.a
		default_option = yes


		set_global_variable = {
			name = lower_canada_rebelled
			value = 1
		}
		c:QUE = {
			capital = {
				save_scope_as = que_capital_scope
			}
		}
		c:QUE = {
			remove_primary_culture = cu:anglo_canadian
			add_primary_culture = cu:franco_canadian
			set_state_religion = rel:catholic
			add_liberty_desire = 100

			create_diplomatic_play = {
				target_country = c:GBR
				#escalation = 80
				type = dp_independence
				
				target_state = scope:que_capital_scope
		
				add_war_goal = {
					holder = c:GBR
					type = reduce_autonomy
					target_country = c:QUE
				}

				#add_war_goal = {
				#	holder = c:QUE
				#	type = humiliation
				#	target_country = c:GBR
				#}
			}
		}
		s:STATE_QUEBEC = {
			add_devastation = 50
		}
	}
}
