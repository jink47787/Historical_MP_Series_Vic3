ideology_moderate = {
	icon = "gfx/interface/icons/ideology_icons/ideology_leader/ideology_leader_no_ideology.dds"

	show_in_list = no

	character_ideology = yes

	non_interest_group_leader_trigger = {
		NOT = { has_role_of_type = agitator }
	}

	interest_group_leader_weight = {
		value = {
			add = 50 # moderates are boring
			desc = "base_value"
		}

		# The End of History
		add = {
			value = owner.loyalist_fraction
			multiply = 100
			desc = "COUNTRY_LOYALISTS"
		}

		# angry IG's more likely to have ideological firebrands
		if = {
			limit = {
				ig_approval <= define:NPolitics|APPROVAL_THRESHOLD_UNHAPPY
			}
			add = {
				value = -25
				desc = "IG_APPROVAL_BELOW_UNHAPPY"
			}
		}

		# what makes a man turn neutral? lust for gold? power? or were you just born with a heart full of neutrality?
		if = {
			limit = {
				ig_approval >= define:NPolitics|APPROVAL_THRESHOLD_HAPPY
			}
			add = {
				value = 25
				desc = "IG_APPROVAL_ABOVE_HAPPY"
			}
		}
		if = {
			limit = {
				OR = {
					has_variable = communist_ig_var
					has_variable = chose_vanguardism
					has_variable = chose_anarchism
					has_variable = chose_communism
					has_variable = chose_ethno_nationalism
					has_variable = chose_conservatism
					has_variable = chose_fascism
				}
			}
			multiply = {
				value = 0.1
				desc = "CHOSE_A_DIFFERENT_PATH"
			}
		}
	}

	non_interest_group_leader_weight = {
		value = 50

		# The End of History
		add = {
			value = owner.loyalist_fraction
			multiply = 100
		}

		# angry IG's more likely to have ideological firebrands
		if = {
			limit = {
				interest_group.ig_approval <= define:NPolitics|APPROVAL_THRESHOLD_UNHAPPY
			}
			add = -25
		}

		# what makes a man turn neutral? lust for gold? power? or were you just born with a heart full of neutrality?
		if = {
			limit = {
				interest_group.ig_approval >= define:NPolitics|APPROVAL_THRESHOLD_HAPPY
			}
			add = 25
		}
		# Military personnel are more inclined to be neutral
		if = {
			limit = {
				OR = {
					has_role_of_type = general
					has_role_of_type = admiral
				}
			}
			add = 50
		}
		if = {
			limit = {
				interest_group ?= {
					OR = {
						has_variable = communist_ig_var
						has_variable = chose_vanguardism
						has_variable = chose_anarchism
						has_variable = chose_communism
						has_variable = chose_ethno_nationalism
						has_variable = chose_conservatism
						has_variable = chose_fascism
					}
				}
			}
			multiply = 0.1
		}
	}
}
