amendment_purchase_system = {
	parent = law_landed_voting

	allowed_laws = {
		law_professional_army
		law_national_militia
		law_mass_conscription
	}

	modifier = {
		unit_army_experience_gain_mult = -0.25
		country_military_goods_cost_mult = -0.1
		interest_group_ig_landowners_pol_str_mult = 0.1
	}

	possible = {
		c:GBR ?= this
	}

	would_sponsor = {
		AND = {
			exists = scope:approval
			scope:approval > 0
		}
		is_interest_group_type = ig_landowners
	}

	ai_will_revoke = {
		date >= 1871.1.1
	}

	can_repeal = {
		legitimacy >= legitimacy_to_repeal_amendment
	}

	amendment_activism_multiplier = 1
}

amendment_cc_irish_and_welsh_migratants = {
	parent = law_racial_segregation

	allowed_laws = {
		law_language_discrimination
	}

	modifier = {
		country_irish_cultural_acceptance_add = 40
		country_welsh_cultural_acceptance_add = 40
		country_scottish_gaelic_cultural_acceptance_add = 40
	}

	possible = {
		AND = {
			has_law = law_type:law_language_discrimination
			OR = {
				country_has_primary_culture = cu:australian
				country_has_primary_culture = cu:anglo_canadian
				country_has_primary_culture = cu:franco_canadian
				c:SAF ?= this
			}
		}
	}

	would_sponsor = {
		AND = {
			exists = scope:approval
			scope:approval > 0
		}
		is_interest_group_type = ig_intelligentsia
	}

	ai_will_revoke = {
		always = no
	}

	can_repeal = {
		legitimacy >= legitimacy_to_repeal_amendment
	}

	amendment_activism_multiplier = 1
}
