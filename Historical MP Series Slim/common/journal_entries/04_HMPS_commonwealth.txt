	scripted_progress_bar = saf_commonwealth_progress_bar
	
	should_be_involved = {
		OR = {
			c:GBR ?= this
			AND = {
				OR = {
					is_in_same_power_bloc = c:GBR
					is_subject_of = c:GBR
				}
				OR = {
					c:SAF ?= this
					country_has_primary_culture = cu:anglo_canadian
					country_has_primary_culture = cu:franco_canadian
					country_has_primary_culture = cu:boer
					country_has_primary_culture = cu:australian
				}
			}
		}
	}

	should_show_when_not_involved = {
		OR = {
			is_in_same_power_bloc = c:GBR
			is_subject_of = c:GBR
		}

		set_global_variable = { name = commonwealth_progress_saf value = "scope:journal_entry.scripted_bar_progress(saf_commonwealth_progress_bar)" }

		#root.average_sol >= 10

		#	value <= 0.10
		#}
	}

	#custom_failure_header = je_uneasy_raj_fail_header

	fail = {
		custom_tooltip = {
			text = je_commonwealth_fail_tt
			NOT = {
				any_country = {
					OR = {
						top_overlord = c:GBR
						is_in_same_power_bloc = c:GBR
					}
					OR = {
						this ?= c:SAF
						country_has_primary_culture = cu:boer

		custom_tooltip = {
			text = je_commonwealth_complete_reward_loyalists_for_involved_tt

		remove_global_variable = commonwealth_progress_saf
	}

	on_weekly_pulse = {
		events = {
		}
		effect = {
			set_global_variable = { name = commonwealth_progress_ast value = "scope:journal_entry.scripted_bar_progress(ast_commonwealth_progress_bar)" }
			set_global_variable = { name = commonwealth_progress_can value = "scope:journal_entry.scripted_bar_progress(can_commonwealth_progress_bar)" }
			set_global_variable = { name = commonwealth_progress_saf value = "scope:journal_entry.scripted_bar_progress(saf_commonwealth_progress_bar)" }

	on_monthly_pulse = {
		effect = {
			scope:journal_entry ?= {
				### AUSTRALIA
				if = {
					limit = {
						global_var:commonwealth_progress_ast >= 50
					}
					c:AST ?= {
						if = {
							limit = {
								this = {
									has_diplomatic_pact = {
										who = c:GBR
										type = colony
									}
								}
							}
							increase_autonomy = yes
						}
					}
				}
				if = {
					limit = {
						global_var:commonwealth_progress_ast >= 100
					}
					c:AST ?= {
						if = {
							limit = {
								this = {
									has_diplomatic_pact = {
										who = c:GBR
										type = dominion
									}
								}
							}
							this = {
								remove_diplomatic_pact = {
									country = c:GBR
									type = dominion
								}
							}
							join_power_bloc = c:GBR
						}
					}
				}
				### CANADA
				if = {
					limit = {
						global_var:commonwealth_progress_can >= 50
					}
					c:CAN ?= {
						if = {
							limit = {
								this = {
									has_diplomatic_pact = {
										who = c:GBR
										type = colony
									}
								}
							}

					}

					limit = {

					}
					c:CAN ?= {
						if = {
							limit = {

									has_diplomatic_pact = {

						}
					}
				}
				### SAFRICA

						global_var:commonwealth_progress_saf >= 50
					}

							limit = {
								this = {
									has_diplomatic_pact = {
										who = c:GBR
										type = colony
									}
								}
							}
							increase_autonomy = yes
						}
					}
				}
				if = {
					limit = {
						global_var:commonwealth_progress_saf >= 100
					}
					c:SAF ?= {
						if = {
							limit = {
								this = {
									has_diplomatic_pact = {
										who = c:GBR
										type = dominion
									}
								}
							}
							this = {
								remove_diplomatic_pact = {

								}
