INJECT:ig_armed_forces = {

	ideologies = {
		ideology_cultural_chauvinism
	}

	on_enable = {
		# HMPS Eight Banners

		if = {
			limit = {
				c:CHI ?= this
			}
			ig:ig_armed_forces = {
				set_interest_group_name = ig_eight_banners
			}
		}
		
		# HMPS Military Monarchism
		if = {
			limit = {
				has_law = law_type:law_monarchy
				ig:ig_armed_forces = {
					NOT = { has_ideology = ideology:ideology_bakufu }
				}
			}
			ig:ig_armed_forces = {
				add_ideology = ideology_monarchist
			}
		}
		
		# HMPS Constitutional Republicanism
		if = {
			limit = {
				c:SWI ?= this
			}
			ig:ig_armed_forces = { 
				add_ideology = ideology_republican
			}
		}
		if = {
			limit = {
				OR = {
					c:USA ?= this
					c:CSA ?= this
				}
			}
			ig:ig_armed_forces = { 
				add_ideology = ideology_american_republicanism
			}
		}
		
		# HMPS Taiping God Worshippers
		if = {
			limit = {
				c:TPG ?= this
			}
			ig:ig_armed_forces = {
				remove_ideology = ideology_monarchist
				remove_ideology = ideology_cultural_chauvinism
				add_ideology = ideology_heavenly_kingdom_theocratic
			}
		}
		
		# HMPS Austrian Hegemonists
		if = {
			limit = {
				c:AUS ?= this
			}
			ig:ig_armed_forces = {
				add_ideology = ideology_austrian_hegemony
				remove_ideology = ideology_cultural_chauvinism
			}
		}
	}
}
