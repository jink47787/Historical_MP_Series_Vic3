ast_commonwealth_progress_bar = {
 	name = "ast_commonwealth_progress_bar_name"
 	desc = "ast_commonwealth_progress_bar_desc"

 	default_green = yes

 	monthly_progress = {
		c:GBR ?= {
			if = {
				limit = {
					any_country = {
						OR = {
							top_overlord = c:GBR
							is_in_same_power_bloc = c:GBR
						}
						country_has_primary_culture = cu:australian
					}
				}
				add = {
					desc = "ast_commonwealth_progress_bar_tt"
					#value = var:ast_commonwealth_target
					
					### GDP
					every_country = {
						limit = {
							OR = {
								top_overlord = c:GBR
								is_in_same_power_bloc = c:GBR
							}
							country_has_primary_culture = cu:australian
						}
						add = gdp
					}
					divide = gdp
					multiply = 100
					
					
					### POPULATION
					multiply = 2000000
					every_country = {
						limit = {
							OR = {
								top_overlord = c:GBR
								is_in_same_power_bloc = c:GBR
							}
							country_has_primary_culture = cu:australian
						}
						add = total_population
					}
					divide = 2000000
					
					### SOL
					multiply = 2
					add = {
						every_country = {
							limit = {
								OR = {
									top_overlord = c:GBR
									is_in_same_power_bloc = c:GBR
								}
								country_has_primary_culture = cu:australian
							}
							add = {
								add = average_sol
								multiply = total_population
							}
						}
						divide = {
							add=1
							every_country = {
								limit = {
									OR = {
										top_overlord = c:GBR
										is_in_same_power_bloc = c:GBR
									}
									country_has_primary_culture = cu:australian
								}
								add = total_population
							}
						}
					}
					divide = 2
					
					### FORMATION
					if = {
						limit = {
							exists = c:AST
							c:AST ?= {
								OR = {
									top_overlord = c:GBR
									is_in_same_power_bloc = c:GBR
								}
							}
						}
						add = 15
					}
					
					
					subtract = global_var:commonwealth_progress_ast
					multiply = 0.05
					min = -1
					max = 1
				}
			}
		}
 	}

	start_value = 6
	min_value = 0
	max_value = 100
}

saf_commonwealth_progress_bar = {
 	name = "saf_commonwealth_progress_bar_name"
 	desc = "saf_commonwealth_progress_bar_desc"

 	monthly_progress = {
		c:GBR ?= {
			if = {
				limit = {
					exists = c:SAF
				}
				add = {
					desc = "saf_commonwealth_progress_bar_tt"
					
					### GDP
					add = c:SAF.gdp
					divide = gdp
					multiply = 100
					
					
					### POPULATION
					multiply = 2000000
					add = c:SAF.total_population
					divide = 2000000
					
					### SOL
					multiply = 2
					add = c:SAF.average_sol
					divide = 2
					
					### FORMATION
					if = {
						limit = {
							exists = c:SAF
							c:SAF ?= {
								country_has_primary_culture = cu:boer
								OR = {
									top_overlord = c:GBR
									is_in_same_power_bloc = c:GBR
								}
							}
						}
						add = 15
					}
					
					
					subtract = global_var:commonwealth_progress_saf
					multiply = 0.05
					min = -1
					max = 1
				}
			}
		}
 	}

 	default = yes

	start_value = 7
	min_value = 0
	max_value = 100
}
