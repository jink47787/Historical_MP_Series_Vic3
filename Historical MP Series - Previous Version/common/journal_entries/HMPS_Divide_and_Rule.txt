je_divide_and_rule = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		any_primary_culture = {
			OR = {
				has_discrimination_trait_group = heritage_group_european
				has_discrimination_trait_group = heritage_group_east_asian
			}
		}
		OR = {
			c:SAF ?= this
			c:DEI ?= this
			capital = {
				is_in_geographic_region = geographic_region_africa
			}
			capital = {
				is_in_geographic_region = geographic_region_south_east_asia
			}
		}
	}

	possible = {
		has_technology_researched = civilizing_mission
	}

	immediate = {
	}

	scripted_button = je_divide_and_rule_sotho_button
	scripted_button = je_divide_and_rule_mande_button
	scripted_button = je_divide_and_rule_hausa_button
	scripted_button = je_divide_and_rule_fluvian_bantu_button
	scripted_button = je_divide_and_rule_oromo_button
	scripted_button = je_divide_and_rule_lacustrine_bantu_button
	scripted_button = je_divide_and_rule_javan_button
	scripted_button = je_divide_and_rule_vietnamese_button

	on_weekly_pulse = {

	}

	on_monthly_pulse = {
		effect = {
		}
	}

	complete = {
		#has_modifier = modifier_has_shown_colonial_favouritism
	}

	on_complete = {
	}

	fail = {
		has_law_or_variant = law_type:law_multicultural
	}

	on_fail = {
		custom_tooltip = {
			text = je_divide_and_rule_fail_tt
			
			if = { limit = { has_modifier = modifier_has_shown_colonial_favouritism }
				remove_modifier = modifier_has_shown_colonial_favouritism }
			if = {
				limit = {
					has_modifier = secondary_culture_javan
					NOT = {
						c:DEI ?= this
					}
				}
				remove_modifier = secondary_culture_javan
			}
			if = { limit = { has_modifier = secondary_culture_sotho }
				remove_modifier = secondary_culture_sotho }
			if = { limit = { has_modifier = secondary_culture_mande }
				remove_modifier = secondary_culture_mande }
			if = { limit = { has_modifier = secondary_culture_hausa }
				remove_modifier = secondary_culture_hausa }
			if = { limit = { has_modifier = secondary_culture_fluvian_bantu }
				remove_modifier = secondary_culture_fluvian_bantu }
			if = { limit = { has_modifier = secondary_culture_oromo }
				remove_modifier = secondary_culture_oromo }
			if = { limit = { has_modifier = secondary_culture_lacustrine_bantu }
				remove_modifier = secondary_culture_lacustrine_bantu }
			if = { limit = { has_modifier = secondary_culture_vietnamese }
				remove_modifier = secondary_culture_vietnamese }
		}
	}

	invalid = {
	}

	weight = 100

	should_be_pinned_by_default = yes
}
